# 7.1. Creación de Tablas

## Módulo 1 - Módulo de Gestión de Reservas:

```sql
CREATE TABLE Cliente (
    ruc CHAR(11) PRIMARY KEY,
    razon_social VARCHAR(150) NOT NULL,
    direccion VARCHAR(200) NOT NULL,
    email VARCHAR(100),
    CONSTRAINT chk_ruc_cliente CHECK (ruc ~ '^[0-9]{11}$'),
    CONSTRAINT chk_email_cliente CHECK (email IS NULL OR email ~* '^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,}$')
);

CREATE TABLE Reserva (
    id_reserva INT PRIMARY KEY,
    codigo VARCHAR(20) NOT NULL UNIQUE,
    fecha_registro DATE NOT NULL DEFAULT CURRENT_DATE,
    id_estado_reserva INT NOT NULL,
    pago_total DECIMAL(10,2),
    ruc_cliente CHAR(11) NOT NULL,
    id_agente_reservas INT NOT NULL,
    matricula_buque VARCHAR(20) NOT NULL,
    id_ruta INT NOT NULL,
    CONSTRAINT fk_reserva_estado FOREIGN KEY (id_estado_reserva) 
        REFERENCES EstadoReserva(id_estado_reserva)
        ON DELETE RESTRICT
        ON UPDATE CASCADE,
    CONSTRAINT fk_reserva_cliente FOREIGN KEY (ruc_cliente) 
        REFERENCES Cliente(ruc)
        ON DELETE RESTRICT
        ON UPDATE CASCADE,
    CONSTRAINT fk_reserva_agente FOREIGN KEY (id_agente_reservas) 
        REFERENCES AgenteReservas(id_agente_reservas)
        ON DELETE RESTRICT
        ON UPDATE CASCADE,
    CONSTRAINT fk_reserva_buque FOREIGN KEY (matricula_buque) 
        REFERENCES Buque(matricula)
        ON DELETE RESTRICT
        ON UPDATE CASCADE,
    CONSTRAINT fk_reserva_ruta FOREIGN KEY (id_ruta) 
        REFERENCES Ruta(id_ruta)
        ON DELETE RESTRICT
        ON UPDATE CASCADE,
    CONSTRAINT chk_pago_total CHECK (pago_total IS NULL OR pago_total >= 0)
);
```

## Módulo 2 - Módulo de Gestión del Personal y Tripulación:

```sql
CREATE TABLE Tripulante (
    id_tripulante INT PRIMARY KEY,
    id_empleado INT NOT NULL UNIQUE, -- Relacion 1:1 con Empleado
    disponibilidad BOOLEAN NOT NULL DEFAULT TRUE,
    nacionalidad VARCHAR(50) NOT NULL,
    id_contrato INT NOT NULL, -- Cada tripulante tiene un contrato
    CONSTRAINT fk_tripulante_empleado FOREIGN KEY (id_empleado) 
        REFERENCES Empleado(id_empleado)
        ON DELETE RESTRICT
        ON UPDATE CASCADE,
    CONSTRAINT fk_tripulante_contrato FOREIGN KEY (id_contrato) 
        REFERENCES Contrato(id_contrato)
        ON DELETE RESTRICT
        ON UPDATE CASCADE
);

CREATE TABLE Buque (
    matricula VARCHAR(20) PRIMARY KEY, -- PK natural
    nombre VARCHAR(100) NOT NULL,
    capacidad INT NOT NULL, -- Capacidad en TEU
    id_estado_embarcacion INT NOT NULL,
    peso DECIMAL(15,2) NOT NULL, -- Peso maximo en toneladas
    ubicacion_actual VARCHAR(100), -- Coordenadas GPS actuales
    CONSTRAINT fk_buque_estado FOREIGN KEY (id_estado_embarcacion) 
        REFERENCES EstadoEmbarcacion(id_estado_embarcacion)
        ON DELETE RESTRICT
        ON UPDATE CASCADE,
    CONSTRAINT chk_capacidad_buque CHECK (capacidad > 0), -- Capacidad mayor a 0
    CONSTRAINT chk_peso_buque CHECK (peso > 0) -- Peso mayor a 0
);

```

## Módulo 3 - Módulo de Gestión de Operaciones Marítimas:

```sql
CREATE TABLE OperacionMaritima (
    id_operacion_maritima INT PRIMARY KEY,
    id_operacion INT NOT NULL UNIQUE, -- Relacion 1:1 con Operacion
    codigo VARCHAR(20) NOT NULL UNIQUE, -- Codigo UK
    cantidad_contenedores INT NOT NULL,
    id_estatus_navegacion INT NOT NULL,
    porcentaje_trayecto DECIMAL(5,2) NOT NULL DEFAULT 0.00, -- Porcentaje del trayecto completado
    matricula VARCHAR(20) NOT NULL, -- Buque asignado
    CONSTRAINT fk_op_maritima_operacion FOREIGN KEY (id_operacion) 
        REFERENCES Operacion(id_operacion)
        ON DELETE RESTRICT
        ON UPDATE CASCADE,
    CONSTRAINT fk_op_maritima_estatus FOREIGN KEY (id_estatus_navegacion) 
        REFERENCES EstatusNavegacion(id_estatus_navegacion)
        ON DELETE RESTRICT
        ON UPDATE CASCADE,
    CONSTRAINT fk_op_maritima_buque FOREIGN KEY (matricula) 
        REFERENCES Buque(matricula)
        ON DELETE RESTRICT
        ON UPDATE CASCADE,
    CONSTRAINT chk_cantidad_contenedores CHECK (cantidad_contenedores >= 0), -- Cantidad mayor o igual a 0
    CONSTRAINT chk_porcentaje CHECK (porcentaje_trayecto BETWEEN 0 AND 100) -- Porcentaje entre 0 y 100
);

CREATE TABLE OperacionPortuaria (
    id_operacion_portuaria INT PRIMARY KEY,
    id_operacion INT NOT NULL UNIQUE, -- Relacion 1:1 con Operacion
    codigo VARCHAR(20) NOT NULL UNIQUE, -- Codigo UK
    id_puerto INT NOT NULL,
    id_muelle INT NOT NULL,
    id_tipo_operacion_portuaria INT NOT NULL,
    matricula VARCHAR(20) NOT NULL, -- Buque asignado
    CONSTRAINT fk_op_portuaria_operacion FOREIGN KEY (id_operacion) 
        REFERENCES Operacion(id_operacion)
        ON DELETE RESTRICT
        ON UPDATE CASCADE,
    CONSTRAINT fk_op_portuaria_puerto FOREIGN KEY (id_puerto) 
        REFERENCES Puerto(id_puerto)
        ON DELETE RESTRICT
        ON UPDATE CASCADE,
    CONSTRAINT fk_op_portuaria_muelle FOREIGN KEY (id_muelle) 
        REFERENCES Muelle(id_muelle)
        ON DELETE RESTRICT
        ON UPDATE CASCADE,
    CONSTRAINT fk_op_portuaria_tipo FOREIGN KEY (id_tipo_operacion_portuaria) 
        REFERENCES TipoOperacionPortuaria(id_tipo_operacion_portuaria)
        ON DELETE RESTRICT
        ON UPDATE CASCADE,
    CONSTRAINT fk_op_portuaria_buque FOREIGN KEY (matricula) 
        REFERENCES Buque(matricula)
        ON DELETE RESTRICT
        ON UPDATE CASCADE
);

CREATE TABLE RutaMaritima (
    id_ruta_maritima INT PRIMARY KEY,
    id_ruta INT NOT NULL UNIQUE, -- Relacion 1:1 con Ruta
    codigo VARCHAR(20) NOT NULL UNIQUE, -- Codigo UK
    id_puerto_origen INT NOT NULL,
    id_puerto_destino INT NOT NULL,
    distancia INT NOT NULL, -- Distancia en millas nauticas
    CONSTRAINT fk_ruta_maritima_ruta FOREIGN KEY (id_ruta) 
        REFERENCES Ruta(id_ruta)
        ON DELETE RESTRICT
        ON UPDATE CASCADE,
    CONSTRAINT fk_ruta_puerto_origen FOREIGN KEY (id_puerto_origen) 
        REFERENCES Puerto(id_puerto)
        ON DELETE RESTRICT
        ON UPDATE CASCADE,
    CONSTRAINT fk_ruta_puerto_destino FOREIGN KEY (id_puerto_destino) 
        REFERENCES Puerto(id_puerto)
        ON DELETE RESTRICT
        ON UPDATE CASCADE,
    CONSTRAINT chk_distancia CHECK (distancia > 0), -- Distancia mayor a 0
    CONSTRAINT chk_puertos_diferentes CHECK (id_puerto_origen != id_puerto_destino) -- Puertos diferentes
);
```

## Módulo 4 - Módulo de Gestión de Operaciones Terrestres:

```sql
 CREATE TABLE operaciones_terrestres.OperacionTerrestre (
     id_operacion_terrestre CHAR(36) PRIMARY KEY DEFAULT gen_random_uuid()::TEXT,
     id_operacion CHAR(36) NOT NULL UNIQUE,
     codigo VARCHAR(20) NOT NULL UNIQUE,
     id_vehiculo CHAR(36) NOT NULL,
     id_ruta_terrestre CHAR(36) NOT NULL,
     id_conductor CHAR(36) NOT NULL,
     costo_operacion_terrestre DECIMAL(10,2) NOT NULL,
     CONSTRAINT fk_op_terrestre_operacion FOREIGN KEY (id_operacion)
         REFERENCES operaciones_terrestres.Operacion(id_operacion)
         ON DELETE RESTRICT
         ON UPDATE CASCADE,
     CONSTRAINT fk_op_terrestre_vehiculo FOREIGN KEY (id_vehiculo)
         REFERENCES operaciones_terrestres.Vehiculo(id_vehiculo)
         ON DELETE RESTRICT
         ON UPDATE CASCADE,
     CONSTRAINT fk_op_terrestre_ruta FOREIGN KEY (id_ruta_terrestre)
         REFERENCES operaciones_terrestres.RutaTerrestre(id_ruta_terrestre)
         ON DELETE RESTRICT
         ON UPDATE CASCADE,
     CONSTRAINT fk_op_terrestre_conductor FOREIGN KEY (id_conductor)
         REFERENCES operaciones_terrestres.Conductor(id_conductor)
         ON DELETE RESTRICT
         ON UPDATE CASCADE,
     CONSTRAINT chk_costo CHECK (costo_operacion_terrestre >= 0)
 );

 CREATE TABLE operaciones_terrestres.Vehiculo (
     id_vehiculo CHAR(36) PRIMARY KEY DEFAULT gen_random_uuid()::TEXT,
     id_activo CHAR(36) NOT NULL UNIQUE,
     placa VARCHAR(20) NOT NULL UNIQUE,
     capacidad_ton DECIMAL(10,2) NOT NULL,
     id_tipo_vehiculo CHAR(36) NOT NULL,
     id_estado_vehiculo CHAR(36) NOT NULL,
     CONSTRAINT fk_vehiculo_activo FOREIGN KEY (id_activo)
         REFERENCES operaciones_terrestres.Activo(id_activo)
         ON DELETE RESTRICT
         ON UPDATE CASCADE,
     CONSTRAINT fk_vehiculo_tipo FOREIGN KEY (id_tipo_vehiculo)
         REFERENCES operaciones_terrestres.TipoVehiculo(id_tipo_vehiculo)
         ON DELETE RESTRICT
         ON UPDATE CASCADE,
     CONSTRAINT fk_vehiculo_estado FOREIGN KEY (id_estado_vehiculo)
         REFERENCES operaciones_terrestres.EstadoVehiculo(id_estado_vehiculo)
         ON DELETE RESTRICT
         ON UPDATE CASCADE,
     CONSTRAINT chk_capacidad CHECK (capacidad_ton > 0)
 );
```

CREATE TABLE mantenimiento_logistico.OperacionMantenimiento (
    id_operacion_mantenimiento CHAR(36) PRIMARY KEY DEFAULT gen_random_uuid()::TEXT,
    id_operacion CHAR(36) NOT NULL UNIQUE,
    id_plan_mantenimiento CHAR(36),
    id_solicitud_mantenimiento CHAR(36),
    CONSTRAINT fk_op_mantenimiento_operacion FOREIGN KEY (id_operacion)
        REFERENCES mantenimiento_logistico.Operacion(id_operacion)
        ON DELETE RESTRICT
        ON UPDATE CASCADE,
    CONSTRAINT fk_op_mantenimiento_plan FOREIGN KEY (id_plan_mantenimiento)
        REFERENCES mantenimiento_logistico.PlanMantenimiento(id_plan_mantenimiento)
        ON DELETE RESTRICT
        ON UPDATE CASCADE,
    CONSTRAINT fk_op_mantenimiento_solicitud FOREIGN KEY (id_solicitud_mantenimiento)
        REFERENCES mantenimiento_logistico.SolicitudMantenimiento(id_solicitud_mantenimiento)
        ON DELETE RESTRICT
        ON UPDATE CASCADE
);

CREATE TABLE mantenimiento_logistico.Repuesto (
    id_repuesto CHAR(36) PRIMARY KEY DEFAULT gen_random_uuid()::TEXT,
    codigo VARCHAR(20) NOT NULL UNIQUE,
    nombre VARCHAR(100) NOT NULL,
    stock INT NOT NULL DEFAULT 0,
    stock_minimo INT NOT NULL DEFAULT 0,
    CONSTRAINT chk_stock CHECK (stock >= 0),
    CONSTRAINT chk_stock_minimo CHECK (stock_minimo >= 0)
);

## Módulo 5 - Módulo de Monitoreo de Entrega:

```SQL
CREATE TABLE Sensor (
    id_sensor INT PRIMARY KEY,
    codigo VARCHAR(20) NOT NULL UNIQUE, -- Codigo UK
    nombre VARCHAR(50) NOT NULL,
    id_tipo_sensor INT NOT NULL,
    id_rol_sensor INT NOT NULL,
    id_contenedor INT NOT NULL,
    CONSTRAINT fk_sensor_tipo FOREIGN KEY (id_tipo_sensor) 
        REFERENCES TipoSensor(id_tipo_sensor)
        ON DELETE RESTRICT
        ON UPDATE CASCADE,
    CONSTRAINT fk_sensor_rol FOREIGN KEY (id_rol_sensor) 
        REFERENCES RolSensor(id_rol_sensor)
        ON DELETE RESTRICT
        ON UPDATE CASCADE,
    CONSTRAINT fk_sensor_contenedor FOREIGN KEY (id_contenedor) 
        REFERENCES Contenedor(id_contenedor)
        ON DELETE CASCADE -- Si se elimina contenedor, se eliminan sus sensores
        ON UPDATE CASCADE
);

CREATE TABLE PosicionContenedor (
    id_posicion INT PRIMARY KEY,
    id_contenedor INT NOT NULL,
    latitud DECIMAL(9,6) NOT NULL, -- Coordenada latitud
    longitud DECIMAL(9,6) NOT NULL, -- Coordenada longitud
    fecha_hora TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP, -- TIMESTAMP del registro GPS
    CONSTRAINT fk_posicion_contenedor FOREIGN KEY (id_contenedor) 
        REFERENCES Contenedor(id_contenedor)
        ON DELETE CASCADE -- Si se elimina contenedor, se elimina su historial
        ON UPDATE CASCADE,
    CONSTRAINT chk_latitud_cont CHECK (latitud BETWEEN -90 AND 90), -- Latitud valida
    CONSTRAINT chk_longitud_cont CHECK (longitud BETWEEN -180 AND 180) -- Longitud valida
);

CREATE TABLE Notificacion (
    id_notificacion INT PRIMARY KEY,
    codigo VARCHAR(20) NOT NULL UNIQUE, -- Codigo UK
    id_tipo_notificacion INT NOT NULL,
    fecha_hora TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP, -- TIMESTAMP para precision exacta
    valor DECIMAL(10,2) NOT NULL, -- Valor medido por el sensor
    id_sensor INT NOT NULL,
    id_reporte INT NOT NULL,
    CONSTRAINT fk_notificacion_tipo FOREIGN KEY (id_tipo_notificacion) 
        REFERENCES TipoNotificacion(id_tipo_notificacion)
        ON DELETE RESTRICT
        ON UPDATE CASCADE,
    CONSTRAINT fk_notificacion_sensor FOREIGN KEY (id_sensor) 
        REFERENCES Sensor(id_sensor)
        ON DELETE CASCADE -- Si se elimina sensor, se eliminan sus notificaciones
        ON UPDATE CASCADE,
    CONSTRAINT fk_notificacion_reporte FOREIGN KEY (id_reporte) 
        REFERENCES Reporte(id_reporte)
        ON DELETE CASCADE -- Si se elimina reporte, se eliminan sus notificaciones
        ON UPDATE CASCADE
);
```
