- **Cargar Lista de Clientes:**

   Al cargar la pantalla, se obtienen los datos necesarios para los campos desplegables y selectores.
   
```sql
SELECT 
    id_cliente,
    CONCAT(razon_social, ' (RUC: ', ruc, ')') AS nombre_completo

ORDER BY razon_social ASC;
```

- **Cargar Lista de Contenedores Disponibles:**

   El dropdown "Seleccionar contenedor" debe mostrar solo contenedores que estén disponibles.
   
```sql
SELECT 
    c.id_contenedor,
    CONCAT(c.codigo, ' - ', tc.nombre, ' (', c.dimensiones, ')') AS descripcion_contenedor
FROM gestion_reserva.Contenedor c
INNER JOIN gestion_reserva.TipoContenedor tc 
    ON c.id_tipo_contenedor = tc.id_tipo_contenedor
INNER JOIN gestion_reserva.EstadoContenedor ec 
    ON c.id_estado_contenedor = ec.id_estado_contenedor
WHERE ec.nombre = 'Disponible'
ORDER BY tc.nombre, c.codigo ASC;
```

- **Cargar Lista de Rutas Activas:**

   El dropdown "Seleccionar Ruta" debe mostrar las rutas disponibles con información de origen, destino y tarifa.
   
```sql
SELECT 
    id_ruta,
    CONCAT(codigo, ' - ', origen, ' → ', destino, ' (', duracion, ' días aprox. - $', tarifa, ')') AS descripcion_ruta
FROM gestion_reserva.Ruta
ORDER BY origen, destino ASC;
```

- **Botón "Registrar Reserva":**

   Al presionar el botón "Registrar Reserva", se ejecutarán múltiples operaciones transaccionales:
   
```sql
--PASO 1 (Generar Código de Reserva Automático)
SELECT 
    COALESCE(
        'RES-' || LPAD((MAX(CAST(SUBSTRING(codigo FROM 5) AS INTEGER)) + 1)::TEXT, 3, '0'),
        'RES-001'
    ) AS nuevo_codigo
FROM gestion_reserva.Reserva
WHERE codigo LIKE 'RES-%';

--PASO 2 (Obtener Matrícula de Buque Disponible para la Ruta)
SELECT 
    b.matricula,
    b.id_buque
FROM gestion_reserva.Buque b
INNER JOIN gestion_reserva.EstadoEmbarcacion ee 
    ON b.id_estado_embarcacion = ee.id_estado_embarcacion
WHERE ee.nombre = 'Disponible'
ORDER BY b.capacidad DESC
LIMIT 1;

--PASO 3 (Obtener ID del Agente de Reservas Actual)
SELECT 
    ar.id_agente_reservas
FROM gestion_reserva.AgenteReservas ar
INNER JOIN gestion_reserva.Empleado e 
    ON ar.id_empleado = e.id_empleado
WHERE e.codigo = '<codigo_empleado_sesion>'
LIMIT 1;

--PASO 4 (Calcular Pago Total (Tarifa de Ruta + Costo de Contenedor))
SELECT 
    r.tarifa + tc.costo AS pago_total
FROM gestion_reserva.Ruta r
CROSS JOIN gestion_reserva.TipoContenedor tc
INNER JOIN gestion_reserva.Contenedor c 
    ON tc.id_tipo_contenedor = c.id_tipo_contenedor
WHERE r.id_ruta = <1>  -- ID de ruta seleccionada
  AND c.id_contenedor = <2>;  -- ID de contenedor seleccionado

--PASO 5 (Insertar Nueva Reserva)
INSERT INTO gestion_reserva.Reserva (
    id_reserva,
    codigo,
    fecha_registro,
    id_estado_reserva,
    pago_total,
    ruc_cliente,
    id_agente_reservas,
    matricula_buque,
    id_ruta
)
VALUES (
    gen_random_uuid(),
    <nuevo_codigo>,  -- Obtenido en 4.1
    CURRENT_DATE,
    (SELECT id_estado_reserva FROM gestion_reserva.EstadoReserva WHERE nombre = 'Confirmada'),
    <pago_total>,  -- Calculado en 4.4
    (SELECT ruc FROM gestion_reserva.Cliente WHERE id_cliente = <1>),  -- Cliente seleccionado
    <id_agente_reservas>,  -- Obtenido en 4.3
    <matricula_buque>,  -- Obtenido en 4.2
    <2>  -- ID de ruta seleccionada
)
RETURNING id_reserva;

--PASO 6 (Asociar Contenedor a la Reserva)
INSERT INTO gestion_reserva.ReservaContenedor (
    id_reserva_contenedor,
    id_reserva,
    id_contenedor,
    fecha_asignacion,
    cantidad
)
VALUES (
    gen_random_uuid(),
    <id_reserva>,  -- ID retornado en 4.5
    <3>,  -- ID de contenedor seleccionado
    CURRENT_DATE,
    1  -- Cantidad por defecto
);

--PASO 7 (Actualizar Estado del Contenedor a "En tránsito")
UPDATE gestion_reserva.Contenedor
SET id_estado_contenedor = (
    SELECT id_estado_contenedor 
    FROM gestion_reserva.EstadoContenedor 
    WHERE nombre = 'En tránsito'
)
WHERE id_contenedor = <3>;  -- ID de contenedor seleccionado

--PASO 8(Registrar Atención al Cliente)
INSERT INTO gestion_reserva.AtencionCliente (
    id_atencion_cliente,
    id_cliente,
    id_agente_reservas,
    fecha_atencion
)
VALUES (
    gen_random_uuid(),
    <1>,  -- ID de cliente seleccionado
    <id_agente_reservas>,  -- Obtenido en 4.3
    CURRENT_TIMESTAMP
);
```

- **Listar Todos los Clientes:**

   Al cargar la página, se debe mostrar la tabla principal con todos los clientes registrados.

```sql
SELECT 
    c.razon_social AS nombre,
    c.ruc,
    CASE 
        WHEN EXISTS (
            SELECT 1
            FROM gestion_reserva.Reserva r
            INNER JOIN gestion_reserva.EstadoReserva er 
                ON r.id_estado_reserva = er.id_estado_reserva
            WHERE r.ruc_cliente = c.ruc
              AND er.nombre IN ('Activa', 'Confirmada')
        )
        THEN 'Activo'
        ELSE 'Inactivo'
    END AS estado_cliente,
    (
        SELECT MAX(r2.fecha_registro)
        FROM gestion_reserva.Reserva r2
        WHERE r2.ruc_cliente = c.ruc
    ) AS ultima_reserva
FROM gestion_reserva.Cliente c
ORDER BY ultima_reserva DESC;
```

- **Tarjeta "Total Reservas":**

   Obtener el total de reservas activas y completadas.
   
```sql
SELECT 
    COUNT(*) AS total_reservas,
    COUNT(CASE WHEN er.nombre IN ('Activa') THEN 1 END) AS reservas_activas,
    COUNT(CASE WHEN er.nombre = 'Finalizada' THEN 1 END) AS reservas_completadas
FROM gestion_reserva.Reserva r
INNER JOIN gestion_reserva.EstadoReserva er 
    ON r.id_estado_reserva = er.id_estado_reserva;
```

- **Tarjeta "Total Contenedores":**

   Obtener el total de contenedores disponibles y ocupados.

```sql
SELECT 
    COUNT(*) AS total_contenedores,
    COUNT(CASE WHEN ec.nombre = 'Disponible' THEN 1 END) AS contenedores_disponibles,
    COUNT(CASE WHEN ec.nombre IN ('En tránsito', 'En mantenimiento') THEN 1 END) AS contenedores_ocupados
FROM gestion_reserva.Contenedor c
INNER JOIN gestion_reserva.EstadoContenedor ec 
    ON c.id_estado_contenedor = ec.id_estado_contenedor;
```

- **Tarjeta "Buques en Operación":**

   Obtener el total de buques y cuántos están en operación vs. en puerto.

```sql
SELECT 
    COUNT(*) AS total_buques,
    COUNT(CASE WHEN ee.nombre = 'En operación' THEN 1 END) AS buques_en_operacion,
    COUNT(CASE WHEN ee.nombre IN ('Disponible', 'Mantenimiento') THEN 1 END) AS buques_en_puerto
FROM gestion_reserva.Buque b
INNER JOIN gestion_reserva.EstadoEmbarcacion ee 
    ON b.id_estado_embarcacion = ee.id_estado_embarcacion;
```

- **Tarjeta "Rutas Activas":**

```sql
SELECT 
    COUNT(*) AS total_rutas
```

- **Gráfico "Buques en Operación y Progreso del Trayecto":**

   Obtener 5 buques en operación con su porcentaje de trayecto.

```sql
SELECT 
    b.nombre AS nombre_buque,
    om.porcentaje_trayecto
FROM gestion_reserva.Buque b
INNER JOIN gestion_reserva.OperacionMaritima om 
    ON b.id_buque = om.id_buque
INNER JOIN gestion_reserva.Operacion o 
    ON om.id_operacion = o.id_operacion
INNER JOIN gestion_reserva.EstadoOperacion eo 
    ON o.id_estado_operacion = eo.id_estado_operacion
WHERE eo.nombre = 'En Progreso'
ORDER BY om.porcentaje_trayecto DESC
LIMIT 5;
```

- **Panel "Top Rutas Más Usadas":**

   Obtener las 5 rutas con mayor número de reservas.

```sql
SELECT 
    CONCAT(r.origen, ' → ', r.destino) AS ruta,
    COUNT(res.id_reserva) AS total_reservas
FROM gestion_reserva.Ruta r
INNER JOIN gestion_reserva.Reserva res 
    ON r.id_ruta = res.id_ruta
GROUP BY r.id_ruta, r.codigo, r.origen, r.destino
ORDER BY total_reservas DESC
LIMIT 5;
```

- **Tabla "Clientes con Más Reservas":**

   Obtener los clientes con mayor número de reservas activas..

```sql
SELECT 
    c.razon_social AS cliente,
    COUNT(DISTINCT r.id_reserva) AS reservas_activas,
    MAX(r.fecha_registro) AS ultima_reserva,
    SUM(rc.cantidad) AS total_contenedores
FROM gestion_reserva.Cliente c
INNER JOIN gestion_reserva.Reserva r 
    ON c.ruc = r.ruc_cliente
INNER JOIN gestion_reserva.EstadoReserva er 
    ON r.id_estado_reserva = er.id_estado_reserva
LEFT JOIN gestion_reserva.ReservaContenedor rc 
    ON r.id_reserva = rc.id_reserva
WHERE er.nombre IN ('Activa', 'Confirmada')
GROUP BY c.id_cliente, c.razon_social, c.ruc
ORDER BY reservas_activas DESC, ultima_reserva DESC
LIMIT 10;
```
