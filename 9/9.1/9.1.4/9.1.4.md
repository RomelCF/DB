## **Planificación de Rutas de Transporte**

### **Eventos:**

#### **Carga inicial del Dashboard:**

```sql
-- Query 1.1: Obtener órdenes pendientes de transporte terrestre
SELECT 
    ot.codigo AS numero_orden,
    r.origen,
    r.destino,
    v.capacidad_ton AS tipo_carga,
    o.fecha_inicio,
    o.fecha_fin,
    eo.nombre AS estado_operacion
FROM operaciones_terrestres.OperacionTerrestre ot
JOIN operaciones_terrestres.Operacion o ON ot.id_operacion = o.id_operacion
JOIN operaciones_terrestres.RutaTerrestre rt ON ot.id_ruta_terrestre = rt.id_ruta_terrestre
JOIN operaciones_terrestres.Ruta r ON rt.id_ruta = r.id_ruta
JOIN operaciones_terrestres.Vehiculo v ON ot.id_vehiculo = v.id_vehiculo
JOIN operaciones_terrestres.EstadoOperacion eo ON o.id_estado_operacion = eo.id_estado_operacion
WHERE eo.nombre = 'Pendiente' -- En curso o Activo
ORDER BY o.fecha_inicio DESC;

-- Query 1.2: Obtener conteo de órdenes por estado
SELECT 
    eo.nombre AS estado,
    COUNT(ot.id_operacion_terrestre) AS cantidad
FROM operaciones_terrestres.OperacionTerrestre ot
JOIN operaciones_terrestres.Operacion o ON ot.id_operacion = o.id_operacion
JOIN operaciones_terrestres.EstadoOperacion eo ON o.id_estado_operacion = eo.id_estado_operacion
GROUP BY eo.nombre;
```

## **Asignación de Vehículos y Conductores**

### **Eventos:**

#### **Carga inicial de vehículos y conductores disponibles:**

```sql
-- Query 2.1: Obtener vehículos disponibles con filtros
SELECT 
    v.placa,
    tv.nombre AS tipo_vehiculo,
    v.capacidad_ton,
    ev.nombre AS estado_vehiculo,
    a.nombre AS nombre_activo,
    ea.nombre AS estado_activo
FROM operaciones_terrestres.Vehiculo v
JOIN operaciones_terrestres.TipoVehiculo tv ON v.id_tipo_vehiculo = tv.id_tipo_vehiculo
JOIN operaciones_terrestres.EstadoVehiculo ev ON v.id_estado_vehiculo = ev.id_estado_vehiculo
JOIN operaciones_terrestres.Activo a ON v.id_activo = a.id_activo

   AND tv.nombre = 'Tacto Camión'  -- Filtro por tipo de vehículo (dinámico)
ORDER BY v.capacidad_ton DESC;

-- Query 2.2: Obtener conductores disponibles
SELECT 
    e.nombre,
    e.apellido,
    c.licencia,
    c.categoria,
    c.disponibilidad,

FROM operaciones_terrestres.Conductor c
JOIN operaciones_terrestres.Empleado e ON c.id_empleado = e.id_empleado
JOIN operaciones_terrestres.EspecialidadEmpleado ee ON e.id_especialidad_empleado = ee.id_especialidad_empleado
```

```sql
-- Query 3.9: Calcular progreso del checklist (0% - 100%)
SELECT 
    CASE 
        WHEN cd.combustible IS NOT NULL AND cd.combustible != '' THEN 1 ELSE 0 
    END +
    CASE 
        WHEN cd.frenos IS NOT NULL AND cd.frenos != '' THEN 1 ELSE 0

        CASE WHEN cd.frenos IS NOT NULL AND cd.frenos != '' THEN 1 ELSE 0 END +
        CASE WHEN cd.observaciones IS NOT NULL THEN 1 ELSE 0 END
    ) * 100.0 / 3, 2) AS porcentaje_completado
```

## **Seguimiento en Tiempo Real de Flota**

### **Eventos:**

#### **Carga inicial de métricas del dashboard:**

```sql
-- Query 4.1: Obtener métricas generales de envíos
SELECT 
    COUNT(*) AS total_envios,
    COUNT(CASE WHEN eo.nombre = 'En Progreso' AND o.fecha_fin < CURRENT_TIMESTAMP THEN 1 END) AS con_retraso,
    COUNT(CASE WHEN EXISTS (
        SELECT 1 FROM operaciones_terrestres.Incidencia i 
        WHERE i.id_operacion = o.id_operacion 
        AND i.id_estado_incidencia = (SELECT id_estado_incidencia FROM operaciones_terrestres.EstadoIncidencia WHERE nombre = 'Activa')
    ) THEN 1 END) AS en_incidencia
FROM operaciones_terrestres.OperacionTerrestre ot
JOIN operaciones_terrestres.Operacion o ON ot.id_operacion = o.id_operacion
JOIN operaciones_terrestres.EstadoOperacion eo ON o.id_estado_operacion = eo.id_estado_operacion
WHERE eo.nombre IN ('En Progreso', 'Pendiente');
```

#### **Obtener alertas activas del sistema:**

```sql
-- Query 4.2: Obtener alertas de retraso (> 20% del tiempo estimado)
SELECT 
    ot.codigo AS numero_orden,
    v.placa AS vehiculo,
    r.duracion AS duracion_estimada,
    EXTRACT(EPOCH FROM (CURRENT_TIMESTAMP - o.fecha_inicio)) / 3600 AS horas_transcurridas,
    ROUND((EXTRACT(EPOCH FROM (CURRENT_TIMESTAMP - o.fecha_inicio)) / 3600 / r.duracion - 1) * 100, 2) AS porcentaje_retraso
FROM operaciones_terrestres.OperacionTerrestre ot
JOIN operaciones_terrestres.Operacion o ON ot.id_operacion = o.id_operacion
JOIN operaciones_terrestres.RutaTerrestre rt ON ot.id_ruta_terrestre = rt.id_ruta_terrestre
JOIN operaciones_terrestres.Ruta r ON rt.id_ruta = r.id_ruta
JOIN operaciones_terrestres.Vehiculo v ON ot.id_vehiculo = v.id_vehiculo
JOIN operaciones_terrestres.EstadoOperacion eo ON o.id_estado_operacion = eo.id_estado_operacion
WHERE eo.nombre = 'En Progreso'
  AND (EXTRACT(EPOCH FROM (CURRENT_TIMESTAMP - o.fecha_inicio)) / 3600 / r.duracion - 1) * 100 > 20;
```

#### **Obtener vehículos activos con su estado actual:**

```sql
-- Query 4.3: Obtener lista de vehículos activos en operaciones
SELECT 
    v.placa,
    CONCAT(e.nombre, ' ', e.apellido) AS conductor,
    r.origen,
    r.destino,
    r.duracion,
    ROUND((EXTRACT(EPOCH FROM (CURRENT_TIMESTAMP - o.fecha_inicio)) / 3600 / r.duracion) * 100, 2) AS progreso_porcentaje,
    CASE 
        WHEN (EXTRACT(EPOCH FROM (CURRENT_TIMESTAMP - o.fecha_inicio)) / 3600 / r.duracion - 1) * 100 > 20 THEN 'Retraso'
        WHEN EXISTS (
            SELECT 1 FROM operaciones_terrestres.Incidencia i 
            WHERE i.id_operacion = o.id_operacion 
            AND i.id_estado_incidencia = (SELECT id_estado_incidencia FROM operaciones_terrestres.EstadoIncidencia WHERE nombre = 'Activa')
        ) THEN 'Incidencia'
        ELSE 'En Tiempo'
    END AS estado_actual
FROM operaciones_terrestres.OperacionTerrestre ot
JOIN operaciones_terrestres.Operacion o ON ot.id_operacion = o.id_operacion
JOIN operaciones_terrestres.RutaTerrestre rt ON ot.id_ruta_terrestre = rt.id_ruta_terrestre
JOIN operaciones_terrestres.Ruta r ON rt.id_ruta = r.id_ruta
JOIN operaciones_terrestres.Vehiculo v ON ot.id_vehiculo = v.id_vehiculo
JOIN operaciones_terrestres.Conductor c ON ot.id_conductor = c.id_conductor
JOIN operaciones_terrestres.Empleado e ON c.id_empleado = e.id_empleado
JOIN operaciones_terrestres.EstadoOperacion eo ON o.id_estado_operacion = eo.id_estado_operacion
WHERE eo.nombre = 'En Progreso'
ORDER BY estado_actual DESC, o.fecha_inicio DESC;
```

```sql
-- Query 4.5: Detectar vehículos posiblemente detenidos
-- Nota: Esta consulta asume que tienes una tabla de ubicaciones GPS
-- Si no la tienes, sería necesario implementar un sistema de tracking
SELECT 
    v.placa,
    ot.codigo AS operacion,
    r.origen,
    r.destino,
    EXTRACT(EPOCH FROM (CURRENT_TIMESTAMP - ultima_ubicacion.fecha_hora)) / 60 AS minutos_sin_movimiento

JOIN operaciones_terrestres.EstadoOperacion eo ON o.id_estado_operacion = eo.id_estado_operacion
-- Esta subconsulta simularía obtener la última ubicación del GPS
CROSS JOIN LATERAL (
    SELECT CURRENT_TIMESTAMP - INTERVAL '45 minutes' AS fecha_hora
) AS ultima_ubicacion
WHERE eo.nombre = 'En Progreso'
  AND EXTRACT(EPOCH FROM (CURRENT_TIMESTAMP - ultima_ubicacion.fecha_hora)) / 60 > 30;
```

## **Planificación de Mantenimiento Preventivo**

### **Eventos:**

#### **Carga inicial de activos que requieren mantenimiento:**

```sql
-- Query 1.1: Obtener activos que requieren mantenimiento urgente
SELECT 
    a.codigo AS id_activo,
    a.nombre AS nombre_activo,
    ta.nombre AS tipo_activo,
    ea.nombre AS estado_activo,
    a.ubicacion,
    pm.fecha_inicio AS proximo_mantenimiento,
    CASE 
        WHEN pm.fecha_inicio <= CURRENT_DATE THEN 'Vencido'
        WHEN pm.fecha_inicio <= CURRENT_DATE + INTERVAL '7 days' THEN '7 Días Restantes'
        WHEN pm.fecha_inicio <= CURRENT_DATE + INTERVAL '30 days' THEN '27 Días Restantes'
        ELSE 'Programado'
    END AS estado_mantenimiento,
    (pm.fecha_inicio - CURRENT_DATE) AS dias_restantes
FROM mantenimiento_logistico.Activo a
JOIN mantenimiento_logistico.TipoActivo ta ON a.id_tipo_activo = ta.id_tipo_activo
JOIN mantenimiento_logistico.EstadoActivo ea ON a.id_estado_activo = ea.id_estado_activo
LEFT JOIN mantenimiento_logistico.PlanMantenimiento pm ON a.id_activo = pm.id_activo
WHERE ea.nombre IN ('En mantenimiento', 'Fuera de servicio')
   OR (pm.fecha_inicio IS NOT NULL AND pm.fecha_inicio <= CURRENT_DATE + INTERVAL '30 days')
ORDER BY pm.fecha_inicio ASC NULLS LAST;
```

---

#### **Búsqueda y filtrado de activos:**

```sql
-- Query 1.2: Buscar activos con filtros
SELECT 
    a.codigo AS id_activo,
    a.nombre AS nombre_activo,
    ta.nombre AS tipo_activo,
    ea.nombre AS estado_activo,
    a.ubicacion,
    pm.fecha_inicio AS proximo_mantenimiento,
    pm.frecuencia
FROM mantenimiento_logistico.Activo a
JOIN mantenimiento_logistico.TipoActivo ta ON a.id_tipo_activo = ta.id_tipo_activo
JOIN mantenimiento_logistico.EstadoActivo ea ON a.id_estado_activo = ea.id_estado_activo
LEFT JOIN mantenimiento_logistico.PlanMantenimiento pm ON a.id_activo = pm.id_activo
WHERE 
    (a.codigo ILIKE '%[BUSQUEDA]%' OR a.nombre ILIKE '%[BUSQUEDA]%')
    AND (ta.nombre = '[TIPO_FILTRO]' OR '[TIPO_FILTRO]' = 'Todos')
    AND (ea.nombre = '[ESTADO_FILTRO]' OR '[ESTADO_FILTRO]' = 'Todos')
    AND (a.ubicacion = '[UBICACION_FILTRO]' OR '[UBICACION_FILTRO]' = 'Todos')
ORDER BY a.nombre;
```

#### **Calendario de mantenimientos programados:**

```sql
-- Query 1.3: Obtener mantenimientos programados para calendario
SELECT 
    a.codigo AS id_activo,
    a.nombre AS nombre_activo,
    pm.fecha_inicio AS fecha_programada,
    ep.nombre AS estado_plan,
    pm.frecuencia,
    ta.nombre AS tipo_activo
FROM mantenimiento_logistico.PlanMantenimiento pm
JOIN mantenimiento_logistico.Activo a ON pm.id_activo = a.id_activo
JOIN mantenimiento_logistico.TipoActivo ta ON a.id_tipo_activo = ta.id_tipo_activo
JOIN mantenimiento_logistico.EstadoPlan ep ON pm.id_estado_plan = ep.id_estado_plan
WHERE pm.fecha_inicio BETWEEN '[FECHA_INICIO_MES]' AND '[FECHA_FIN_MES]'
  AND ep.nombre = 'Activo'
ORDER BY pm.fecha_inicio;
```

# **Gestión de Inventario de Repuestos**


### **Query 4.1: Métricas del dashboard de inventario**
```sql
SELECT 
    COUNT(*) as total_repuestos,
    COUNT(CASE WHEN stock <= stock_minimo THEN 1 END) as stock_critico,
    SUM(stock * 150) as valor_inventario_estimado
FROM mantenimiento_logistico.Repuesto;
```

---

### **Query 4.2: Lista de todos los repuestos con estado**
```sql
SELECT 
    codigo,
    nombre,
    'Categoría' as categoria, -- Campo temporal, ajustar según esquema
    stock,
    stock_minimo,
    'Almacén A-1' as ubicacion, -- Campo temporal
    CASE 
        WHEN stock <= stock_minimo THEN 'Crítico'
        WHEN stock <= stock_minimo * 1.5 THEN 'Bajo'
        ELSE 'Satisfactorio'
    END as estado
FROM mantenimiento_logistico.Repuesto
ORDER BY
```

```sql
    SUM(omr.cantidad * omr.precio_unitario) as costo_total
FROM mantenimiento_logistico.Activo a
JOIN mantenimiento_logistico.TipoActivo ta ON a.id_tipo_activo = ta.id_tipo_activo
JOIN mantenimiento_logistico.OrdenMantenimiento om ON (
    EXISTS (
        SELECT 1 FROM mantenimiento_logistico.OperacionMantenimiento opm 
        WHERE (opm.id_plan_mantenimiento IS NOT NULL AND 
               (SELECT id_activo FROM mantenimiento_logistico.PlanMantenimiento WHERE id_plan_mantenimiento = opm.id_plan_mantenimiento) = a.id_activo)
        OR (opm.id_solicitud_mantenimiento IS NOT NULL AND 
            (SELECT id_activo FROM mantenimiento_logistico.SolicitudMantenimiento WHERE id_solicitud_mantenimiento = opm.id_solicitud_mantenimiento) = a.id_activo)
        AND opm.id_operacion_mantenimiento = om.id_operacion_mantenimiento
    )
)
JOIN mantenimiento_logistico.OperacionMantenimientoRepuesto omr ON (
    EXISTS (
        SELECT 1 FROM mantenimiento_logistico.OperacionMantenimiento opm 
        WHERE opm.id_operacion_mantenimiento = omr.id_operacion_mantenimiento
        AND ((opm.id_plan_mantenimiento IS NOT NULL AND 
              (SELECT id_activo FROM mantenimiento_logistico.PlanMantenimiento WHERE id_plan_mantenimiento = opm.id_plan_mantenimiento) = a.id_activo)
        OR (opm.id_solicitud_mantenimiento IS NOT NULL AND 
            (SELECT id_activo FROM mantenimiento_logistico.SolicitudMantenimiento WHERE id_solicitud_mantenimiento = opm.id_solicitud_mantenimiento) = a.id_activo))
    )
)
WHERE om.fecha_generada BETWEEN '2023-10-01' AND CURRENT_DATE
GROUP BY a.nombre, ta.nombre
ORDER BY total_mantenimientos DESC
LIMIT 10;
```

### **Query 7.6: Eficiencia de técnicos**
```sql
SELECT 
    CONCAT(e.nombre, ' ', e.apellido) as tecnico,
    t.especialidad,
    COUNT(DISTINCT omt.id_operacion_mantenimiento) as ordenes_atendidas,
    COUNT(DISTINCT tm.id_tarea) as tareas_completadas,
    ROUND(AVG(EXTRACT(DAYS FROM (om.fecha_cierre - om.fecha_generada))), 1) as tiempo_promedio
FROM mantenimiento_logistico.Tecnico t
JOIN mantenimiento_logistico.Empleado e ON t.id_empleado = e.id_empleado
JOIN mantenimiento_logistico.OperacionMantenimientoTecnico omt ON t.id_tecnico = omt.id_tecnico
JOIN mantenimiento_logistico.OrdenMantenimiento om ON (
    SELECT opm.id_operacion_mantenimiento 
    FROM mantenimiento_logistico.OperacionMantenimiento opm 
    WHERE opm.id_operacion_mantenimiento = omt.id_operacion_mantenimiento
) = om.id_operacion_mantenimiento
LEFT JOIN mantenimiento_logistico.TareaMantenimiento tm ON om.id_orden = tm.id_orden
WHERE om.fecha_generada BETWEEN '2023-10-01' AND CURRENT_DATE
GROUP BY e.id_empleado, e.nombre, e.apellido, t.especialidad
ORDER BY ordenes_atendidas DESC;
```

### **Query 8.1: Estado de sincronización general**
```sql
SELECT 
    'CMMS' as sistema,
    CASE 
        WHEN MAX(last_sync) >= CURRENT_TIMESTAMP - INTERVAL '10 minutes' THEN 'Connected'
        ELSE 'Error'
    END as estado,
    MAX(last_sync) as ultima_sincronizacion
FROM (
    SELECT MAX(fecha_uso) as last_sync FROM mantenimiento_logistico.OperacionMantenimientoRepuesto
    UNION ALL
    SELECT MAX(fecha_asignacion) as last_sync FROM mantenimiento_logistico.OperacionMantenimientoTecnico
    UNION ALL
    SELECT MAX(fecha_generada) as last_sync FROM mantenimiento_logistico.OrdenMantenimiento
) as sync_dates;
```

### **Query 8.6: Forzar sincronización de datos pendientes**
```sql
-- Marcar órdenes completadas para sincronización
UPDATE mantenimiento_logistico.Operacion 
SET id_estado_operacion = (
    SELECT id_estado_operacion FROM mantenimiento_logistico.EstadoOperacion WHERE nombre = 'Completada'
)
WHERE id_operacion IN (
    SELECT opm.id_operacion 
    FROM mantenimiento_logistico.OperacionMantenimiento opm
    JOIN mantenimiento_logistico.OrdenMantenimiento om ON opm.id_operacion_mantenimiento = om.id_operacion_mantenimiento
    WHERE om.id_estado_orden = (
        SELECT id_estado_orden FROM mantenimiento_logistico.EstadoOrden WHERE nombre = 'Completada'
    )
    AND om.fecha_cierre IS NOT NULL
)
AND fecha_fin IS NULL;

-- Actualizar stock de repuestos después de sincronización
UPDATE mantenimiento_logistico.Repuesto r
SET stock = stock - (
    SELECT COALESCE(SUM(omr.cantidad), 0)
    FROM mantenimiento_logistico.OperacionMantenimientoRepuesto omr
    WHERE omr.id_repuesto = r.id_repuesto
    AND omr.fecha_uso >= CURRENT_DATE - INTERVAL '1 day'
)
WHERE r.id_repuesto IN (
    SELECT DISTINCT id_repuesto 
    FROM mantenimiento_logistico.OperacionMantenimientoRepuesto 
    WHERE fecha_uso >= CURRENT_DATE - INTERVAL '1 day'
);
```

