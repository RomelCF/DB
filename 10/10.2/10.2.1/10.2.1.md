# 10.2.1. Módulo 1 / Integrante 1

## Módulo de Gestión de Reservas

- **Nota:** Utilizar los [scripts de creación de tablas para este módulo](/scripts/ddl/tablas/integrante1.sql) y el [poblamiento inicial de datos](/scripts/dml/datos-iniciales/integrante1.sql)

### Creación de vistas

## 1. Vista: v_dashboard_reservas

**Pantalla:** P05 - Dashboard de Reservas

**Descripción:** Proporciona un resumen ejecutivo con métricas de reservas activas, ingresos, ocupación de buques y estadísticas principales para el dashboard.

**Tablas incluidas:** Reserva, EstadoReserva, Cliente, Buque, Ruta, ReservaContenedor, Contenedor

### Script de creación de vista

```sql
CREATE OR REPLACE VIEW gestion_reserva.v_dashboard_reservas AS
SELECT 
    -- Métricas de reservas
    COUNT(DISTINCT r.id_reserva) AS total_reservas,
    COUNT(DISTINCT CASE WHEN er.nombre = 'Activa' THEN r.id_reserva END) AS reservas_activas,
    COUNT(DISTINCT CASE WHEN er.nombre = 'Confirmada' THEN r.id_reserva END) AS reservas_confirmadas,
    COUNT(DISTINCT CASE WHEN er.nombre = 'Finalizada' THEN r.id_reserva END) AS reservas_finalizadas,
    COUNT(DISTINCT CASE WHEN er.nombre = 'Cancelada' THEN r.id_reserva END) AS reservas_canceladas,
    COUNT(DISTINCT CASE WHEN er.nombre = 'Pendiente' THEN r.id_reserva END) AS reservas_pendientes,
    
    -- Métricas financieras
    COALESCE(SUM(CASE WHEN er.nombre = 'Finalizada' THEN r.pago_total END), 0) AS ingresos_totales,
    COALESCE(SUM(CASE WHEN er.nombre IN ('Activa', 'Confirmada') THEN r.pago_total END), 0) AS ingresos_proyectados,
    COALESCE(AVG(CASE WHEN er.nombre = 'Finalizada' THEN r.pago_total END), 0)::DECIMAL(10,2) AS ticket_promedio,
    COALESCE(MAX(r.pago_total), 0) AS ticket_maximo,
    
    -- Métricas de clientes
    COUNT(DISTINCT r.ruc_cliente) AS clientes_activos,
    (SELECT COUNT(*) FROM gestion_reserva.Cliente) AS total_clientes,
    
    -- Métricas de buques
    COUNT(DISTINCT r.matricula_buque) AS buques_en_uso,
    (SELECT COUNT(*) FROM gestion_reserva.Buque WHERE id_estado_embarcacion = 
        (SELECT id_estado_embarcacion FROM gestion_reserva.EstadoEmbarcacion WHERE nombre = 'Disponible')
    ) AS buques_disponibles,
    
    -- Métricas de contenedores
    COALESCE(SUM(rc.cantidad), 0) AS total_contenedores_reservados,
    COUNT(DISTINCT rc.id_contenedor) AS tipos_contenedores_usados,
    
    -- Métricas de rutas
    COUNT(DISTINCT r.id_ruta) AS rutas_activas,
    (SELECT COUNT(*) FROM gestion_reserva.Ruta) AS total_rutas,
    
    -- Métricas temporales
    COUNT(DISTINCT CASE 
        WHEN r.fecha_registro >= CURRENT_DATE - INTERVAL '7 days' THEN r.id_reserva 
    END) AS reservas_ultima_semana,
    COUNT(DISTINCT CASE 
        WHEN r.fecha_registro >= CURRENT_DATE - INTERVAL '30 days' THEN r.id_reserva 
    END) AS reservas_ultimo_mes,
    
    -- Ocupación promedio de buques
    (
        SELECT AVG(porcentaje_ocupacion)::DECIMAL(5,2)
        FROM (
            SELECT 
                r2.matricula_buque,
                (SUM(rc2.cantidad)::DECIMAL / NULLIF(b2.capacidad, 0) * 100) AS porcentaje_ocupacion
            FROM gestion_reserva.Reserva r2
            INNER JOIN gestion_reserva.ReservaContenedor rc2 ON r2.id_reserva = rc2.id_reserva
            INNER JOIN gestion_reserva.Buque b2 ON r2.matricula_buque = b2.matricula
            INNER JOIN gestion_reserva.EstadoReserva er2 ON r2.id_estado_reserva = er2.id_estado_reserva
            WHERE er2.nombre IN ('Activa', 'Confirmada')
            GROUP BY r2.matricula_buque, b2.capacidad
        ) AS ocupaciones
    ) AS ocupacion_promedio_buques,
    
    -- Timestamp
    CURRENT_TIMESTAMP AS fecha_consulta
    
FROM gestion_reserva.Reserva r
INNER JOIN gestion_reserva.EstadoReserva er ON r.id_estado_reserva = er.id_estado_reserva
LEFT JOIN gestion_reserva.ReservaContenedor rc ON r.id_reserva = rc.id_reserva;
```

## 2. Vista: v_listado_reservas

**Pantalla:** P02 - Listado de Reservas

**Descripción:** Proporciona el listado completo de reservas con información del cliente, buque, ruta, estado y detalles de contenedores para visualización en tabla.

**Tablas incluidas:** Reserva, EstadoReserva, Cliente, Buque, Ruta, ReservaContenedor, Contenedor, TipoContenedor

### Script de creación de vista

```sql
CREATE OR REPLACE VIEW gestion_reserva.v_listado_reservas AS
SELECT 
    -- Información básica de la reserva
    r.id_reserva,
    r.codigo,
    r.fecha_registro,
    r.pago_total,
    er.nombre AS estado_reserva,
    
    -- Información del cliente
    c.ruc AS ruc_cliente,
    c.razon_social AS cliente,
    c.email AS email_cliente,
    c.direccion AS direccion_cliente,
    
    -- Información del buque
    b.matricula AS matricula_buque,
    b.nombre AS nombre_buque,
    b.capacidad AS capacidad_buque,
    b.ubicacion_actual AS ubicacion_buque,
    eeb.nombre AS estado_buque,
    
    -- Información de la ruta
    rt.codigo AS codigo_ruta,
    rt.origen,
    rt.destino,
    rt.duracion AS duracion_dias,
    rt.tarifa AS tarifa_ruta,
    
    -- Información de contenedores
    COUNT(DISTINCT rc.id_contenedor) AS tipos_contenedores,
    COALESCE(SUM(rc.cantidad), 0) AS total_contenedores,
    STRING_AGG(DISTINCT tc.nombre, ', ') AS tipos_contenedores_nombres,
    
    -- Cálculos
    ROUND((COALESCE(SUM(rc.cantidad), 0)::DECIMAL / NULLIF(b.capacidad, 0) * 100)::NUMERIC, 2) AS porcentaje_ocupacion,
    
    -- Días desde registro
    (CURRENT_DATE - r.fecha_registro)::INTEGER AS dias_desde_registro,
    
    -- Agente de reservas
    CONCAT(e.nombre, ' ', e.apellido) AS agente_reservas,
    
    -- Estado operacional
    CASE 
        WHEN er.nombre = 'Cancelada' THEN 'Cancelada'
        WHEN er.nombre = 'Finalizada' THEN 'Completada'
        WHEN er.nombre = 'Activa' THEN 'En curso'
        WHEN er.nombre = 'Confirmada' THEN 'Confirmada'
        ELSE 'Pendiente'
    END AS estado_operacional
    
FROM gestion_reserva.Reserva r
INNER JOIN gestion_reserva.EstadoReserva er ON r.id_estado_reserva = er.id_estado_reserva
INNER JOIN gestion_reserva.Cliente c ON r.ruc_cliente = c.ruc
INNER JOIN gestion_reserva.Buque b ON r.matricula_buque = b.matricula
INNER JOIN gestion_reserva.EstadoEmbarcacion eeb ON b.id_estado_embarcacion = eeb.id_estado_embarcacion
INNER JOIN gestion_reserva.Ruta rt ON r.id_ruta = rt.id_ruta
INNER JOIN gestion_reserva.AgenteReservas ar ON r.id_agente_reservas = ar.id_agente_reservas
INNER JOIN gestion_reserva.Empleado e ON ar.id_empleado = e.id_empleado
LEFT JOIN gestion_reserva.ReservaContenedor rc ON r.id_reserva = rc.id_reserva
LEFT JOIN gestion_reserva.Contenedor ct ON rc.id_contenedor = ct.id_contenedor
LEFT JOIN gestion_reserva.TipoContenedor tc ON ct.id_tipo_contenedor = tc.id_tipo_contenedor
GROUP BY r.id_reserva, r.codigo, r.fecha_registro, r.pago_total, er.nombre, 
         c.ruc, c.razon_social, c.email, c.direccion,
         b.matricula, b.nombre, b.capacidad, b.ubicacion_actual, eeb.nombre,
         rt.codigo, rt.origen, rt.destino, rt.duracion, rt.tarifa,
         e.nombre, e.apellido
ORDER BY r.fecha_registro DESC;
```

## 3. Vista: v_registro_reserva_completo

**Pantalla:** P01 - Registro de Reservas

**Descripción:** Contiene toda la información necesaria para el formulario de registro y edición de reservas, incluyendo catálogos de estados, rutas disponibles, buques con capacidad y contenedores.

**Tablas incluidas:** Reserva, EstadoReserva, Cliente, Buque, Ruta, Contenedor, TipoContenedor, EstadoContenedor, EstadoEmbarcacion

### Script de creación de vista

```sql
CREATE OR REPLACE VIEW gestion_reserva.v_registro_reserva_completo AS
SELECT 
    -- Información de la reserva
    r.id_reserva,
    r.codigo,
    r.fecha_registro,
    r.pago_total,
    r.id_estado_reserva,
    er.nombre AS estado_reserva,
    
    -- Cliente
    r.ruc_cliente,
    c.razon_social AS cliente,
    c.email AS email_cliente,
    c.direccion AS direccion_cliente,
    
    -- Buque
    r.matricula_buque,
    b.nombre AS nombre_buque,
    b.capacidad AS capacidad_buque,
    b.peso AS peso_maximo_buque,
    b.ubicacion_actual,
    eeb.nombre AS estado_buque,
    
    -- Capacidad disponible del buque
    (
        b.capacidad - COALESCE(
            (
                SELECT SUM(rc_uso.cantidad)
                FROM gestion_reserva.Reserva r_uso
                INNER JOIN gestion_reserva.ReservaContenedor rc_uso ON r_uso.id_reserva = rc_uso.id_reserva
                INNER JOIN gestion_reserva.EstadoReserva er_uso ON r_uso.id_estado_reserva = er_uso.id_estado_reserva
                WHERE r_uso.matricula_buque = r.matricula_buque
                AND er_uso.nombre IN ('Confirmada', 'Activa')
                AND r_uso.id_reserva != r.id_reserva
            ), 0
        )
    ) AS espacios_disponibles,
    
    -- Ruta
    r.id_ruta,
    rt.codigo AS codigo_ruta,
    rt.origen,
    rt.destino,
    rt.duracion AS duracion_dias,
    rt.tarifa AS tarifa_ruta,
    
    -- Contenedores de la reserva
    STRING_AGG(DISTINCT tc.nombre || ' x' || rc.cantidad, ', ') AS contenedores_reservados,
    COUNT(DISTINCT rc.id_contenedor) AS tipos_contenedores,
    COALESCE(SUM(rc.cantidad), 0) AS total_contenedores,
    
    -- Agente de reservas
    r.id_agente_reservas,
    CONCAT(e.nombre, ' ', e.apellido) AS agente_reservas,
    e.codigo AS codigo_agente,
    
    -- Indicadores
    CASE 
        WHEN b.capacidad - COALESCE(
            (
                SELECT SUM(rc_uso.cantidad)
                FROM gestion_reserva.Reserva r_uso
                INNER JOIN gestion_reserva.ReservaContenedor rc_uso ON r_uso.id_reserva = rc_uso.id_reserva
                INNER JOIN gestion_reserva.EstadoReserva er_uso ON r_uso.id_estado_reserva = er_uso.id_estado_reserva
                WHERE r_uso.matricula_buque = r.matricula_buque
                AND er_uso.nombre IN ('Confirmada', 'Activa')
            ), 0
        ) > 0 THEN TRUE
        ELSE FALSE
    END AS buque_disponible,
    
    CASE 
        WHEN r.fecha_registro >= CURRENT_DATE - INTERVAL '7 days' THEN TRUE
        ELSE FALSE
    END AS puede_modificar

FROM gestion_reserva.Reserva r
INNER JOIN gestion_reserva.EstadoReserva er ON r.id_estado_reserva = er.id_estado_reserva
INNER JOIN gestion_reserva.Cliente c ON r.ruc_cliente = c.ruc
INNER JOIN gestion_reserva.Buque b ON r.matricula_buque = b.matricula
INNER JOIN gestion_reserva.EstadoEmbarcacion eeb ON b.id_estado_embarcacion = eeb.id_estado_embarcacion
INNER JOIN gestion_reserva.Ruta rt ON r.id_ruta = rt.id_ruta
INNER JOIN gestion_reserva.AgenteReservas ar ON r.id_agente_reservas = ar.id_agente_reservas
INNER JOIN gestion_reserva.Empleado e ON ar.id_empleado = e.id_empleado
LEFT JOIN gestion_reserva.ReservaContenedor rc ON r.id_reserva = rc.id_reserva
LEFT JOIN gestion_reserva.Contenedor ct ON rc.id_contenedor = ct.id_contenedor
LEFT JOIN gestion_reserva.TipoContenedor tc ON ct.id_tipo_contenedor = tc.id_tipo_contenedor
GROUP BY r.id_reserva, r.codigo, r.fecha_registro, r.pago_total, 
         r.id_estado_reserva, er.nombre, r.ruc_cliente, c.razon_social, 
         c.email, c.direccion, r.matricula_buque, b.nombre, b.capacidad, 
         b.peso, b.ubicacion_actual, eeb.nombre, r.id_ruta, rt.codigo, rt.origen, 
         rt.destino, rt.duracion, rt.tarifa,
         r.id_agente_reservas, e.nombre, e.apellido, e.codigo;
```

## 4. Vista: v_clientes_historial

**Pantalla:** P03 - Gestión de Clientes e Historial

**Descripción:** Muestra el perfil completo de clientes con su historial de reservas, estadísticas de servicios y métricas de negocio.

**Tablas incluidas:** Cliente, ClienteTelefono, TipoTelefono, Reserva, EstadoReserva, AtencionCliente, AgenteReservas, Empleado

### Script de creación de vista

```sql
CREATE OR REPLACE VIEW gestion_reserva.v_clientes_historial AS
SELECT 
    -- Información básica del cliente
    c.id_cliente,
    c.ruc,
    c.razon_social,
    c.direccion,
    c.email,
    
    -- Teléfonos adicionales
    STRING_AGG(DISTINCT ct.telefono || ' (' || tt.nombre || ')', ', ') AS telefonos,
    
    -- Estadísticas de reservas
    COUNT(DISTINCT r.id_reserva) AS total_reservas,
    COUNT(DISTINCT CASE WHEN er.nombre = 'Finalizada' THEN r.id_reserva END) AS reservas_completadas,
    COUNT(DISTINCT CASE WHEN er.nombre = 'Confirmada' THEN r.id_reserva END) AS reservas_confirmadas,
    COUNT(DISTINCT CASE WHEN er.nombre = 'Activa' THEN r.id_reserva END) AS reservas_activas,
    COUNT(DISTINCT CASE WHEN er.nombre = 'Cancelada' THEN r.id_reserva END) AS reservas_canceladas,
    
    -- Métricas financieras
    COALESCE(SUM(CASE WHEN er.nombre = 'Finalizada' THEN r.pago_total END), 0) AS ingresos_generados,
    COALESCE(AVG(CASE WHEN er.nombre = 'Finalizada' THEN r.pago_total END), 0)::DECIMAL(10,2) AS ticket_promedio,
    COALESCE(MAX(r.pago_total), 0) AS ticket_maximo,
    
    -- Fechas importantes
    MIN(r.fecha_registro) AS primera_reserva,
    MAX(r.fecha_registro) AS ultima_reserva,
    
    -- Rutas más utilizadas
    (
        SELECT STRING_AGG(DISTINCT rt.origen || ' → ' || rt.destino, ', ')
        FROM gestion_reserva.Reserva r2
        INNER JOIN gestion_reserva.Ruta rt ON r2.id_ruta = rt.id_ruta
        WHERE r2.ruc_cliente = c.ruc
        LIMIT 5
    ) AS rutas_frecuentes,
    
    -- Atenciones recibidas
    COUNT(DISTINCT ac.id_atencion_cliente) AS atenciones_recibidas,
    MAX(ac.fecha_atencion) AS ultima_atencion,
    STRING_AGG(DISTINCT CONCAT(emp.nombre, ' ', emp.apellido), ', ') AS agentes_asignados,
    
    -- Buques utilizados
    COUNT(DISTINCT r.matricula_buque) AS buques_diferentes,
    
    -- Estado del cliente
    CASE 
        WHEN MAX(r.fecha_registro) >= CURRENT_DATE - INTERVAL '90 days' THEN 'Activo'
        WHEN MAX(r.fecha_registro) >= CURRENT_DATE - INTERVAL '180 days' THEN 'Regular'
        WHEN MAX(r.fecha_registro) IS NOT NULL THEN 'Inactivo'
        ELSE 'Nuevo'
    END AS clasificacion_cliente,
    
    -- Nivel de cliente basado en ingresos
    CASE 
        WHEN COALESCE(SUM(CASE WHEN er.nombre = 'Finalizada' THEN r.pago_total END), 0) >= 100000 THEN 'Premium'
        WHEN COALESCE(SUM(CASE WHEN er.nombre = 'Finalizada' THEN r.pago_total END), 0) >= 50000 THEN 'Gold'
        WHEN COALESCE(SUM(CASE WHEN er.nombre = 'Finalizada' THEN r.pago_total END), 0) >= 10000 THEN 'Silver'
        ELSE 'Standard'
    END AS nivel_cliente

FROM gestion_reserva.Cliente c
LEFT JOIN gestion_reserva.ClienteTelefono ct ON c.id_cliente = ct.id_cliente
LEFT JOIN gestion_reserva.TipoTelefono tt ON ct.id_tipo_telefono = tt.id_tipo_telefono
LEFT JOIN gestion_reserva.Reserva r ON c.ruc = r.ruc_cliente
LEFT JOIN gestion_reserva.EstadoReserva er ON r.id_estado_reserva = er.id_estado_reserva
LEFT JOIN gestion_reserva.AtencionCliente ac ON c.id_cliente = ac.id_cliente
LEFT JOIN gestion_reserva.AgenteReservas ar ON ac.id_agente_reservas = ar.id_agente_reservas
LEFT JOIN gestion_reserva.Empleado emp ON ar.id_empleado = emp.id_empleado
GROUP BY c.id_cliente, c.ruc, c.razon_social, c.direccion, c.email
ORDER BY ingresos_generados DESC;
```
