# 10.2.5. Módulo 5 / Integrante 5

## Módulo de Monitoreo de Entrega

- **Nota:** Utilizar los [scripts de creación de tablas para este módulo](/scripts/ddl/tablas/integrante5.sql) y el [poblamiento inicial de datos](/scripts/dml/datos-iniciales/integrante5.sql)

### Creación de vistas

1. **Pantalla: P03 - Dashboard principal de monitoreo**
    - Vista 1: v_dashboard_kpis
    - Tablas incluidas: Operacion, Contenedor, Sensor, Notificacion, Incidencia

    **Script de creación de vista**

```sql
CREATE OR REPLACE VIEW monitoreo.v_dashboard_kpis AS
SELECT 
    -- Métricas de operaciones
    COUNT(DISTINCT o.id_operacion) AS total_operaciones,
    COUNT(DISTINCT CASE WHEN eo.nombre = 'En curso' THEN o.id_operacion END) AS operaciones_en_curso,
    COUNT(DISTINCT CASE WHEN eo.nombre = 'Completada' THEN o.id_operacion END) AS operaciones_completadas,
    COUNT(DISTINCT CASE WHEN eo.nombre = 'Pendiente' THEN o.id_operacion END) AS operaciones_pendientes,
    
    -- Métricas de contenedores
    (SELECT COUNT(*) FROM monitoreo.Contenedor) AS total_contenedores,
    (SELECT COUNT(*) FROM monitoreo.Contenedor c 
     JOIN monitoreo.EstadoContenedor ec ON c.id_estado_contenedor = ec.id_estado_contenedor 
     WHERE ec.nombre = 'En tránsito') AS contenedores_en_transito,
    (SELECT COUNT(*) FROM monitoreo.Contenedor c 
     JOIN monitoreo.EstadoContenedor ec ON c.id_estado_contenedor = ec.id_estado_contenedor 
     WHERE ec.nombre = 'Disponible') AS contenedores_disponibles,
    (SELECT COUNT(*) FROM monitoreo.Contenedor c 
     JOIN monitoreo.EstadoContenedor ec ON c.id_estado_contenedor = ec.id_estado_contenedor 
     WHERE ec.nombre = 'Entregado') AS contenedores_entregados,
    
    -- Métricas de sensores
    (SELECT COUNT(*) FROM monitoreo.Sensor) AS total_sensores_activos,
    (SELECT COUNT(*) FROM monitoreo.Sensor s 
     JOIN monitoreo.TipoSensor ts ON s.id_tipo_sensor = ts.id_tipo_sensor 
     WHERE ts.nombre = 'Temperatura') AS sensores_temperatura,
    (SELECT COUNT(*) FROM monitoreo.Sensor s 
     JOIN monitoreo.TipoSensor ts ON s.id_tipo_sensor = ts.id_tipo_sensor 
     WHERE ts.nombre = 'GPS') AS sensores_gps,
    
    -- Notificaciones últimas 24 horas
    (SELECT COUNT(*) FROM monitoreo.Notificacion n
     JOIN monitoreo.TipoNotificacion tn ON n.id_tipo_notificacion = tn.id_tipo_notificacion
     WHERE tn.nombre = 'Crítica' 
     AND n.fecha_hora >= CURRENT_TIMESTAMP - INTERVAL '24 hours') AS notificaciones_criticas_24h,
    (SELECT COUNT(*) FROM monitoreo.Notificacion n
     JOIN monitoreo.TipoNotificacion tn ON n.id_tipo_notificacion = tn.id_tipo_notificacion
     WHERE tn.nombre = 'Advertencia' 
     AND n.fecha_hora >= CURRENT_TIMESTAMP - INTERVAL '24 hours') AS notificaciones_advertencia_24h,
    
    -- Incidencias
    (SELECT COUNT(*) FROM monitoreo.Incidencia i
     JOIN monitoreo.EstadoIncidencia ei ON i.id_estado_incidencia = ei.id_estado_incidencia
     WHERE ei.nombre = 'Abierta') AS incidencias_abiertas,
    (SELECT COUNT(*) FROM monitoreo.Incidencia i
     JOIN monitoreo.EstadoIncidencia ei ON i.id_estado_incidencia = ei.id_estado_incidencia
     WHERE ei.nombre = 'En proceso') AS incidencias_en_proceso,
    
    -- Entregas
    (SELECT COUNT(*) FROM monitoreo.Entrega) AS total_entregas,
    (SELECT COUNT(*) FROM monitoreo.Entrega e
     JOIN monitoreo.EstadoEntrega ee ON e.id_estado_entrega = ee.id_estado_entrega
     WHERE ee.nombre = 'Completada') AS entregas_completadas,
    
    -- Vehículos y buques
    (SELECT COUNT(*) FROM monitoreo.Vehiculo) AS total_vehiculos,
    (SELECT COUNT(*) FROM monitoreo.Buque) AS total_buques,
    
    -- Duración promedio
    AVG(EXTRACT(EPOCH FROM (COALESCE(o.fecha_fin, CURRENT_TIMESTAMP) - o.fecha_inicio)) / 3600)::DECIMAL(10,2) AS duracion_promedio_horas,
    
    -- Timestamp
    CURRENT_TIMESTAMP AS fecha_consulta
    
FROM monitoreo.Operacion o
JOIN monitoreo.EstadoOperacion eo ON o.id_estado_operacion = eo.id_estado_operacion;
```

2. **Pantalla: P05, P06 - Vista detallada de contenedor**
    - Vista 2: v_contenedores_completo
    - Tablas incluidas: Contenedor, Sensor, Notificacion, Documentacion

    **Script de creación de vista**

```sql
CREATE OR REPLACE VIEW monitoreo.v_contenedores_completo AS
SELECT 
    -- Información básica del contenedor
    c.id_contenedor,
    c.codigo,
    c.peso,
    c.capacidad,
    c.dimensiones,
    
    -- Tipo de contenedor
    tc.codigo AS codigo_tipo,
    tc.nombre AS tipo_contenedor,
    tc.costo AS costo_tipo,
    
    -- Estado
    ec.nombre AS estado_contenedor,
    
    -- Porcentaje de carga
    ROUND((c.peso / NULLIF(c.capacidad, 0) * 100)::NUMERIC, 2) AS porcentaje_carga,
    
    -- Estado operacional consolidado
    CASE 
        WHEN ec.nombre = 'No disponible' THEN 'No disponible'
        WHEN ec.nombre = 'En mantenimiento' THEN 'En mantenimiento'
        WHEN ec.nombre = 'Entregado' THEN 'Entregado'
        WHEN ec.nombre = 'En tránsito' THEN 'En tránsito'
        WHEN ec.nombre = 'Disponible' THEN 'Disponible'
        ELSE 'Otro'
    END AS estado_operacional,
    
    -- Sensores (subconsultas)
    (SELECT COUNT(*) 
     FROM monitoreo.Sensor s 
     WHERE s.id_contenedor = c.id_contenedor) AS total_sensores,
     
    (SELECT COUNT(*) 
     FROM monitoreo.Sensor s 
     JOIN monitoreo.TipoSensor ts ON s.id_tipo_sensor = ts.id_tipo_sensor
     WHERE s.id_contenedor = c.id_contenedor 
     AND ts.nombre = 'Temperatura') AS sensores_temperatura,
     
    (SELECT COUNT(*) 
     FROM monitoreo.Sensor s 
     JOIN monitoreo.TipoSensor ts ON s.id_tipo_sensor = ts.id_tipo_sensor
     WHERE s.id_contenedor = c.id_contenedor 
     AND ts.nombre = 'Humedad') AS sensores_humedad,
     
    (SELECT COUNT(*) 
     FROM monitoreo.Sensor s 
     JOIN monitoreo.TipoSensor ts ON s.id_tipo_sensor = ts.id_tipo_sensor
     WHERE s.id_contenedor = c.id_contenedor 
     AND ts.nombre = 'GPS') AS sensores_gps,
     
    (SELECT COUNT(*) 
     FROM monitoreo.Sensor s 
     JOIN monitoreo.TipoSensor ts ON s.id_tipo_sensor = ts.id_tipo_sensor
     WHERE s.id_contenedor = c.id_contenedor 
     AND ts.nombre = 'Vibración') AS sensores_vibracion,
    
    -- Notificaciones
    (SELECT COUNT(*) 
     FROM monitoreo.Notificacion n
     JOIN monitoreo.Sensor s ON n.id_sensor = s.id_sensor
     WHERE s.id_contenedor = c.id_contenedor) AS total_notificaciones,
     
    (SELECT COUNT(*) 
     FROM monitoreo.Notificacion n
     JOIN monitoreo.Sensor s ON n.id_sensor = s.id_sensor
     JOIN monitoreo.TipoNotificacion tn ON n.id_tipo_notificacion = tn.id_tipo_notificacion
     WHERE s.id_contenedor = c.id_contenedor 
     AND tn.nombre = 'Crítica') AS notificaciones_criticas,
     
    (SELECT MAX(n.fecha_hora) 
     FROM monitoreo.Notificacion n
     JOIN monitoreo.Sensor s ON n.id_sensor = s.id_sensor
     WHERE s.id_contenedor = c.id_contenedor) AS fecha_ultima_notificacion,
    
    -- Documentación
    (SELECT COUNT(*) 
     FROM monitoreo.DocumentacionContenedor dc 
     WHERE dc.id_contenedor = c.id_contenedor) AS total_documentos,
     
    (SELECT STRING_AGG(d.nombre, ', ') 
     FROM monitoreo.DocumentacionContenedor dc
     JOIN monitoreo.Documentacion d ON dc.id_documentacion = d.id_documentacion
     WHERE dc.id_contenedor = c.id_contenedor) AS documentos_nombres,
    
    -- Entrega (si existe)
    ent.codigo AS codigo_entrega,
    ent.fecha_entrega,
    ent.lugar_entrega,
    ee.nombre AS estado_entrega,
    imp.razon_social AS importador,
    imp.ruc AS ruc_importador
    
FROM monitoreo.Contenedor c
JOIN monitoreo.TipoContenedor tc ON c.id_tipo_contenedor = tc.id_tipo_contenedor
JOIN monitoreo.EstadoContenedor ec ON c.id_estado_contenedor = ec.id_estado_contenedor
LEFT JOIN monitoreo.Entrega ent ON ent.id_contenedor = c.id_contenedor
LEFT JOIN monitoreo.EstadoEntrega ee ON ent.id_estado_entrega = ee.id_estado_entrega
LEFT JOIN monitoreo.Importador imp ON ent.id_importador = imp.id_importador;
```

3. **Pantalla: P09 - Generación de reportes y auditoría**
    - Vista 3: v_operaciones_reporte
    - Tablas incluidas: Operacion, Incidencia, Reporte, Notificacion

    **Script de creación de vista**

```sql
CREATE OR REPLACE VIEW monitoreo.v_operaciones_reporte AS
SELECT 
    -- Información básica
    o.id_operacion,
    o.codigo AS codigo_operacion,
    o.fecha_inicio,
    o.fecha_fin,
    eo.nombre AS estado_operacion,
    
    -- Duración
    CASE 
        WHEN o.fecha_fin IS NOT NULL THEN 
            EXTRACT(EPOCH FROM (o.fecha_fin - o.fecha_inicio)) / 3600
        ELSE 
            EXTRACT(EPOCH FROM (CURRENT_TIMESTAMP - o.fecha_inicio)) / 3600
    END AS duracion_horas,
    
    -- Estado temporal
    CASE 
        WHEN o.fecha_fin IS NULL THEN 'Activa'
        ELSE 'Finalizada'
    END AS estado_temporal,
    
    -- Días de duración
    CASE 
        WHEN o.fecha_fin IS NOT NULL THEN 
            EXTRACT(DAY FROM (o.fecha_fin - o.fecha_inicio))
        ELSE 
            EXTRACT(DAY FROM (CURRENT_TIMESTAMP - o.fecha_inicio))
    END::INTEGER AS dias_duracion,
    
    -- Incidencias
    (SELECT COUNT(*) 
     FROM monitoreo.Incidencia i 
     WHERE i.id_operacion = o.id_operacion) AS total_incidencias,
     
    (SELECT COUNT(*) 
     FROM monitoreo.Incidencia i
     JOIN monitoreo.EstadoIncidencia ei ON i.id_estado_incidencia = ei.id_estado_incidencia
     WHERE i.id_operacion = o.id_operacion 
     AND ei.nombre = 'Abierta') AS incidencias_abiertas,
     
    (SELECT COUNT(*) 
     FROM monitoreo.Incidencia i
     JOIN monitoreo.EstadoIncidencia ei ON i.id_estado_incidencia = ei.id_estado_incidencia
     WHERE i.id_operacion = o.id_operacion 
     AND ei.nombre = 'Resuelta') AS incidencias_resueltas,
     
    (SELECT STRING_AGG(DISTINCT ti.nombre, ', ') 
     FROM monitoreo.Incidencia i
     JOIN monitoreo.TipoIncidencia ti ON i.id_tipo_incidencia = ti.id_tipo_incidencia
     WHERE i.id_operacion = o.id_operacion) AS tipos_incidencias,
     
    (SELECT AVG(i.grado_severidad)::DECIMAL(3,1) 
     FROM monitoreo.Incidencia i 
     WHERE i.id_operacion = o.id_operacion) AS severidad_promedio,
     
    (SELECT MAX(i.grado_severidad) 
     FROM monitoreo.Incidencia i 
     WHERE i.id_operacion = o.id_operacion) AS severidad_maxima,
    
    -- Reportes
    (SELECT COUNT(DISTINCT r.id_reporte)
     FROM monitoreo.Incidencia i
     JOIN monitoreo.IncidenciaReporte ir ON ir.id_incidencia = i.id_incidencia
     JOIN monitoreo.Reporte r ON r.id_reporte = ir.id_reporte
     WHERE i.id_operacion = o.id_operacion) AS total_reportes,
     
    (SELECT STRING_AGG(DISTINCT r.codigo, ', ' ORDER BY r.codigo)
     FROM monitoreo.Incidencia i
     JOIN monitoreo.IncidenciaReporte ir ON ir.id_incidencia = i.id_incidencia
     JOIN monitoreo.Reporte r ON r.id_reporte = ir.id_reporte
     WHERE i.id_operacion = o.id_operacion) AS codigos_reportes,
    
    -- Usuarios involucrados
    (SELECT STRING_AGG(DISTINCT u.correo_electronico, ', ')
     FROM monitoreo.Incidencia i
     JOIN monitoreo.Usuario u ON i.id_usuario = u.id_usuario
     WHERE i.id_operacion = o.id_operacion) AS usuarios_reportaron,
    
    -- Calificación automática
    CASE 
        WHEN (SELECT COUNT(*) FROM monitoreo.Incidencia i WHERE i.id_operacion = o.id_operacion) = 0 THEN 'Excelente'
        WHEN (SELECT COUNT(*) FROM monitoreo.Incidencia i WHERE i.id_operacion = o.id_operacion) <= 2 
             AND COALESCE((SELECT AVG(i.grado_severidad) FROM monitoreo.Incidencia i WHERE i.id_operacion = o.id_operacion), 0) < 5 
             THEN 'Bueno'
        WHEN (SELECT COUNT(*) FROM monitoreo.Incidencia i WHERE i.id_operacion = o.id_operacion) <= 5 
             AND COALESCE((SELECT AVG(i.grado_severidad) FROM monitoreo.Incidencia i WHERE i.id_operacion = o.id_operacion), 0) < 7 
             THEN 'Regular'
        ELSE 'Deficiente'
    END AS calificacion_operacion,
    
    -- Cumplimiento de plazo (7 días estándar)
    CASE 
        WHEN o.fecha_fin IS NOT NULL AND o.fecha_fin <= o.fecha_inicio + INTERVAL '7 days' THEN TRUE
        WHEN o.fecha_fin IS NULL AND CURRENT_TIMESTAMP <= o.fecha_inicio + INTERVAL '7 days' THEN TRUE
        ELSE FALSE
    END AS cumplimiento_plazo
    
FROM monitoreo.Operacion o
JOIN monitoreo.EstadoOperacion eo ON o.id_estado_operacion = eo.id_estado_operacion;
```
