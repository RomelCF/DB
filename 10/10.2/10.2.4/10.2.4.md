# 10.2.4. Módulo 4 / Integrante 4


## Módulo de Gestión de Operaciones Terrestres

- **Nota:** Utilizar los [scripts de creación de tablas para este módulo](/scripts/ddl/tablas/integrante4.sql) y el [poblamiento inicial de datos](/scripts/dml/datos-iniciales/integrante4.sql)

### Creación de vistas

## 1. Vista: v_dashboard_operaciones_terrestres

**Pantalla:** P04 - Seguimiento en Tiempo Real de Flota

**Descripción:** Proporciona un resumen ejecutivo con métricas de operaciones terrestres, vehículos, rutas y estadísticas principales para el dashboard de monitoreo.

**Tablas incluidas:** Operacion, OperacionTerrestre, EstadoOperacion, Vehiculo, EstadoVehiculo, Ruta, RutaTerrestre, Incidencia

### Script de creación de vista

```sql
CREATE OR REPLACE VIEW operaciones_terrestres.v_dashboard_operaciones_terrestres AS
SELECT 
    -- Métricas de operaciones
    COUNT(DISTINCT o.id_operacion) AS total_operaciones,
    COUNT(DISTINCT CASE WHEN eo.nombre = 'En Progreso' THEN o.id_operacion END) AS operaciones_en_curso,
    COUNT(DISTINCT CASE WHEN eo.nombre = 'Completada' THEN o.id_operacion END) AS operaciones_completadas,
    COUNT(DISTINCT CASE WHEN eo.nombre = 'Pendiente' THEN o.id_operacion END) AS operaciones_pendientes,
    COUNT(DISTINCT CASE WHEN eo.nombre = 'Cancelada' THEN o.id_operacion END) AS operaciones_canceladas,
    
    -- Métricas de vehículos
    COUNT(DISTINCT v.id_vehiculo) AS total_vehiculos,
    COUNT(DISTINCT CASE WHEN ev.nombre = 'En ruta' THEN v.id_vehiculo END) AS vehiculos_en_ruta,
    COUNT(DISTINCT CASE WHEN ev.nombre = 'Disponible' THEN v.id_vehiculo END) AS vehiculos_disponibles,
    COUNT(DISTINCT CASE WHEN ev.nombre = 'Mantenimiento' THEN v.id_vehiculo END) AS vehiculos_mantenimiento,
    COUNT(DISTINCT CASE WHEN ev.nombre = 'Fuera de servicio' THEN v.id_vehiculo END) AS vehiculos_fuera_servicio,
    
    -- Métricas de conductores
    COUNT(DISTINCT c.id_conductor) AS total_conductores,
    COUNT(DISTINCT CASE WHEN c.disponibilidad = TRUE THEN c.id_conductor END) AS conductores_disponibles,
    COUNT(DISTINCT CASE WHEN c.disponibilidad = FALSE THEN c.id_conductor END) AS conductores_ocupados,
    
    -- Métricas de rutas
    COUNT(DISTINCT r.id_ruta) AS rutas_activas,
    (SELECT COUNT(*) FROM operaciones_terrestres.Ruta) AS total_rutas,
    
    -- Métricas de incidencias
    COUNT(DISTINCT i.id_incidencia) AS total_incidencias,
    COUNT(DISTINCT CASE WHEN ei.nombre = 'Reportada' THEN i.id_incidencia END) AS incidencias_reportadas,
    COUNT(DISTINCT CASE WHEN ei.nombre = 'En investigación' THEN i.id_incidencia END) AS incidencias_investigacion,
    COUNT(DISTINCT CASE WHEN ei.nombre = 'Resuelta' THEN i.id_incidencia END) AS incidencias_resueltas,
    COUNT(DISTINCT CASE WHEN ei.nombre = 'Cerrada' THEN i.id_incidencia END) AS incidencias_cerradas,
    
    -- Métricas de puntualidad
    COUNT(DISTINCT CASE 
        WHEN eo.nombre = 'Completada' AND o.fecha_fin <= o.fecha_inicio + (r.duracion || ' hours')::INTERVAL 
        THEN o.id_operacion 
    END) AS operaciones_a_tiempo,
    
    ROUND(
        COUNT(DISTINCT CASE 
            WHEN eo.nombre = 'Completada' AND o.fecha_fin <= o.fecha_inicio + (r.duracion || ' hours')::INTERVAL 
            THEN o.id_operacion 
        END) * 100.0 / 
        NULLIF(COUNT(DISTINCT CASE WHEN eo.nombre = 'Completada' THEN o.id_operacion END), 0),
    2) AS porcentaje_puntualidad,
    
    -- Métricas financieras
    COALESCE(SUM(ot.costo_operacion_terrestre), 0) AS costos_totales,
    COALESCE(SUM(r.tarifa), 0) AS ingresos_totales,
    COALESCE(SUM(r.tarifa - ot.costo_operacion_terrestre), 0) AS utilidad_neta,
    
    -- Métricas de capacidad
    COALESCE(SUM(v.capacidad_ton), 0) AS capacidad_total_ton,
    COALESCE(AVG(v.capacidad_ton), 0)::DECIMAL(10,2) AS capacidad_promedio_ton,
    
    -- Alertas activas
    COUNT(DISTINCT CASE 
        WHEN eo.nombre = 'En Progreso' 
        AND (EXTRACT(EPOCH FROM (CURRENT_TIMESTAMP - o.fecha_inicio)) / 3600 / r.duracion - 1) * 100 > 20 
        THEN o.id_operacion 
    END) AS operaciones_con_retraso,
    
    -- Timestamp de consulta
    CURRENT_TIMESTAMP AS fecha_consulta
    
FROM operaciones_terrestres.Operacion o
INNER JOIN operaciones_terrestres.EstadoOperacion eo ON o.id_estado_operacion = eo.id_estado_operacion
LEFT JOIN operaciones_terrestres.OperacionTerrestre ot ON o.id_operacion = ot.id_operacion
LEFT JOIN operaciones_terrestres.Vehiculo v ON ot.id_vehiculo = v.id_vehiculo
LEFT JOIN operaciones_terrestres.EstadoVehiculo ev ON v.id_estado_vehiculo = ev.id_estado_vehiculo
LEFT JOIN operaciones_terrestres.Conductor c ON ot.id_conductor = c.id_conductor
LEFT JOIN operaciones_terrestres.RutaTerrestre rt ON ot.id_ruta_terrestre = rt.id_ruta_terrestre
LEFT JOIN operaciones_terrestres.Ruta r ON rt.id_ruta = r.id_ruta
LEFT JOIN operaciones_terrestres.Incidencia i ON o.id_operacion = i.id_operacion
LEFT JOIN operaciones_terrestres.EstadoIncidencia ei ON i.id_estado_incidencia = ei.id_estado_incidencia;
```

## 2. Vista: v_listado_operaciones_terrestres

**Pantalla:** P01 - Planificación de Rutas de Transporte

**Descripción:** Proporciona el listado completo de operaciones terrestres con información de vehículos, rutas, conductores y estados para visualización en tabla.

**Tablas incluidas:** Operacion, OperacionTerrestre, EstadoOperacion, Vehiculo, TipoVehiculo, EstadoVehiculo, Conductor, Empleado, Ruta, RutaTerrestre

### Script de creación de vista

```sql
CREATE OR REPLACE VIEW operaciones_terrestres.v_listado_operaciones_terrestres AS
SELECT 
    -- Información básica de la operación
    o.id_operacion,
    o.codigo AS codigo_operacion,
    o.fecha_inicio,
    o.fecha_fin,
    eo.nombre AS estado_operacion,
    
    -- Información de operación terrestre
    ot.id_operacion_terrestre,
    ot.codigo AS codigo_operacion_terrestre,
    ot.costo_operacion_terrestre,
    
    -- Información del vehículo
    v.id_vehiculo,
    v.placa AS placa_vehiculo,
    a.nombre AS nombre_vehiculo,
    tv.nombre AS tipo_vehiculo,
    v.capacidad_ton,
    ev.nombre AS estado_vehiculo,
    
    -- Información del conductor
    c.id_conductor,
    CONCAT(e.nombre, ' ', e.apellido) AS nombre_conductor,
    c.licencia,
    c.categoria,
    c.disponibilidad,
    ee.nombre AS especialidad_conductor,
    
    -- Información de la ruta
    r.id_ruta,
    r.codigo AS codigo_ruta,
    r.origen,
    r.destino,
    r.duracion AS duracion_estimada_horas,
    r.tarifa AS tarifa_ruta,
    
    -- Cálculos y métricas
    CASE 
        WHEN o.fecha_fin IS NOT NULL THEN 
            EXTRACT(EPOCH FROM (o.fecha_fin - o.fecha_inicio)) / 3600
        ELSE 
            EXTRACT(EPOCH FROM (CURRENT_TIMESTAMP - o.fecha_inicio)) / 3600
    END::DECIMAL(10,2) AS duracion_real_horas,
    
    CASE 
        WHEN o.fecha_fin IS NOT NULL AND r.duracion > 0 THEN
            ROUND((EXTRACT(EPOCH FROM (o.fecha_fin - o.fecha_inicio)) / 3600 / r.duracion) * 100, 2)
        WHEN o.fecha_fin IS NULL AND r.duracion > 0 THEN
            ROUND((EXTRACT(EPOCH FROM (CURRENT_TIMESTAMP - o.fecha_inicio)) / 3600 / r.duracion) * 100, 2)
        ELSE 0
    END AS progreso_porcentaje,
    
    CASE 
        WHEN o.fecha_fin IS NOT NULL AND o.fecha_fin <= o.fecha_inicio + (r.duracion || ' hours')::INTERVAL THEN 
            'A Tiempo'
        WHEN o.fecha_fin IS NOT NULL AND o.fecha_fin > o.fecha_inicio + (r.duracion || ' hours')::INTERVAL THEN 
            'Con Retraso'
        WHEN o.fecha_fin IS NULL AND CURRENT_TIMESTAMP > o.fecha_inicio + (r.duracion || ' hours')::INTERVAL THEN 
            'En Retraso'
        ELSE 'En Tiempo'
    END AS estado_puntualidad,
    
    -- Checklist de despacho
    cd.codigo AS codigo_checklist,
    cd.combustible,
    cd.frenos,
    cd.observaciones AS observaciones_checklist,
    
    -- Incidencias relacionadas
    (SELECT COUNT(*) FROM operaciones_terrestres.Incidencia i 
     WHERE i.id_operacion = o.id_operacion) AS total_incidencias,
    
    (SELECT COUNT(*) FROM operaciones_terrestres.Incidencia i 
     WHERE i.id_operacion = o.id_operacion 
     AND i.id_estado_incidencia = (SELECT id_estado_incidencia FROM operaciones_terrestres.EstadoIncidencia WHERE nombre = 'Activa')) AS incidencias_activas,
    
    -- Documentación
    (SELECT COUNT(*) FROM operaciones_terrestres.OperacionDocumento od 
     WHERE od.id_operacion = o.id_operacion) AS total_documentos,
    
    -- Reportes de transporte
    (SELECT COUNT(*) FROM operaciones_terrestres.ReporteTransporte rt 
     WHERE rt.id_operacion_terrestre = ot.id_operacion_terrestre) AS total_reportes

FROM operaciones_terrestres.Operacion o
INNER JOIN operaciones_terrestres.EstadoOperacion eo ON o.id_estado_operacion = eo.id_estado_operacion
INNER JOIN operaciones_terrestres.OperacionTerrestre ot ON o.id_operacion = ot.id_operacion
LEFT JOIN operaciones_terrestres.Vehiculo v ON ot.id_vehiculo = v.id_vehiculo
LEFT JOIN operaciones_terrestres.Activo a ON v.id_activo = a.id_activo
LEFT JOIN operaciones_terrestres.TipoVehiculo tv ON v.id_tipo_vehiculo = tv.id_tipo_vehiculo
LEFT JOIN operaciones_terrestres.EstadoVehiculo ev ON v.id_estado_vehiculo = ev.id_estado_vehiculo
LEFT JOIN operaciones_terrestres.Conductor c ON ot.id_conductor = c.id_conductor
LEFT JOIN operaciones_terrestres.Empleado e ON c.id_empleado = e.id_empleado
LEFT JOIN operaciones_terrestres.EspecialidadEmpleado ee ON e.id_especialidad_empleado = ee.id_especialidad_empleado
LEFT JOIN operaciones_terrestres.RutaTerrestre rtt ON ot.id_ruta_terrestre = rtt.id_ruta_terrestre
LEFT JOIN operaciones_terrestres.Ruta r ON rtt.id_ruta = r.id_ruta
LEFT JOIN operaciones_terrestres.ChecklistDespacho cd ON ot.id_operacion_terrestre = cd.id_operacion_terrestre
ORDER BY o.fecha_inicio DESC;
```

## 3. Vista: v_registro_operacion_completo

**Pantalla:** P02 - Asignación de Vehículos y Conductores

**Descripción:** Contiene toda la información necesaria para el formulario de registro y edición de operaciones terrestres, incluyendo catálogos de vehículos disponibles, conductores y rutas.

**Tablas incluidas:** Operacion, OperacionTerrestre, EstadoOperacion, Vehiculo, TipoVehiculo, EstadoVehiculo, Conductor, Empleado, Ruta, RutaTerrestre, ChecklistDespacho

### Script de creación de vista

```sql
CREATE OR REPLACE VIEW operaciones_terrestres.v_registro_operacion_completo AS
SELECT 
    -- Información de la operación base
    o.id_operacion,
    o.codigo AS codigo_operacion,
    o.fecha_inicio,
    o.fecha_fin,
    o.id_estado_operacion,
    eo.nombre AS estado_operacion,
    
    -- Información de operación terrestre
    ot.id_operacion_terrestre,
    ot.codigo AS codigo_operacion_terrestre,
    ot.costo_operacion_terrestre,
    ot.id_vehiculo,
    ot.id_ruta_terrestre,
    ot.id_conductor,
    
    -- Información del vehículo seleccionado
    v.placa AS placa_vehiculo,
    a.nombre AS nombre_vehiculo,
    a.codigo AS codigo_activo_vehiculo,
    tv.nombre AS tipo_vehiculo,
    v.capacidad_ton,
    ev.nombre AS estado_vehiculo,
    a.ubicacion AS ubicacion_vehiculo,
    
    -- Información del conductor seleccionado
    CONCAT(e.nombre, ' ', e.apellido) AS nombre_conductor,
    e.codigo AS codigo_empleado_conductor,
    c.licencia,
    c.categoria,
    c.disponibilidad,
    ee.nombre AS especialidad_conductor,
    
    -- Información de la ruta seleccionada
    r.id_ruta,
    r.codigo AS codigo_ruta,
    r.origen,
    r.destino,
    r.duracion,
    r.tarifa,
    
    -- Checklist de despacho
    cd.id_checklist,
    cd.codigo AS codigo_checklist,
    cd.observaciones AS observaciones_checklist,
    cd.combustible,
    cd.frenos,
    
    -- Validaciones de disponibilidad
    CASE 
        WHEN ev.nombre = 'Disponible' THEN TRUE
        ELSE FALSE
    END AS vehiculo_disponible,
    
    CASE 
        WHEN c.disponibilidad = TRUE THEN TRUE
        ELSE FALSE
    END AS conductor_disponible,
    
    CASE 
        WHEN o.fecha_inicio >= CURRENT_DATE - INTERVAL '7 days' 
        AND eo.nombre IN ('Pendiente', 'En curso') THEN TRUE
        ELSE FALSE
    END AS puede_modificar,
    
    -- Cálculo de costo estimado
    CASE 
        WHEN ot.costo_operacion_terrestre IS NOT NULL THEN ot.costo_operacion_terrestre
        ELSE r.tarifa * 0.7  -- 70% de la tarifa como costo estimado
    END AS costo_estimado,
    
    -- Información para asignaciones
    (SELECT COUNT(*) FROM operaciones_terrestres.OperacionTerrestre ot2 
     WHERE ot2.id_vehiculo = ot.id_vehiculo 
     AND ot2.id_operacion_terrestre != COALESCE(ot.id_operacion_terrestre, '00000000-0000-0000-0000-000000000000')
     AND EXISTS (SELECT 1 FROM operaciones_terrestres.Operacion o2 
                 WHERE o2.id_operacion = ot2.id_operacion 
                 AND o2.id_estado_operacion = (SELECT id_estado_operacion FROM operaciones_terrestres.EstadoOperacion WHERE nombre = 'En Progreso')))
    AS vehiculo_en_uso,
    
    (SELECT COUNT(*) FROM operaciones_terrestres.OperacionTerrestre ot2 
     WHERE ot2.id_conductor = ot.id_conductor 
     AND ot2.id_operacion_terrestre != COALESCE(ot.id_operacion_terrestre, '00000000-0000-0000-0000-000000000000')
     AND EXISTS (SELECT 1 FROM operaciones_terrestres.Operacion o2 
                 WHERE o2.id_operacion = ot2.id_operacion 
                 AND o2.id_estado_operacion = (SELECT id_estado_operacion FROM operaciones_terrestres.EstadoOperacion WHERE nombre = 'En Progreso')))
    AS conductor_en_uso,
    
    -- Documentación asociada
    (SELECT STRING_AGG(doc.nombre, ', ') 
     FROM operaciones_terrestres.OperacionDocumento od
     JOIN operaciones_terrestres.DocumentacionOperacion doc ON od.id_documento = doc.id_documento
     WHERE od.id_operacion = o.id_operacion) AS documentos_asociados,
    
    -- Reportes generados
    (SELECT STRING_AGG(rt.codigo, ', ') 
     FROM operaciones_terrestres.ReporteTransporte rt
     WHERE rt.id_operacion_terrestre = ot.id_operacion_terrestre) AS reportes_generados

FROM operaciones_terrestres.Operacion o
INNER JOIN operaciones_terrestres.EstadoOperacion eo ON o.id_estado_operacion = eo.id_estado_operacion
LEFT JOIN operaciones_terrestres.OperacionTerrestre ot ON o.id_operacion = ot.id_operacion
LEFT JOIN operaciones_terrestres.Vehiculo v ON ot.id_vehiculo = v.id_vehiculo
LEFT JOIN operaciones_terrestres.Activo a ON v.id_activo = a.id_activo
LEFT JOIN operaciones_terrestres.TipoVehiculo tv ON v.id_tipo_vehiculo = tv.id_tipo_vehiculo
LEFT JOIN operaciones_terrestres.EstadoVehiculo ev ON v.id_estado_vehiculo = ev.id_estado_vehiculo
LEFT JOIN operaciones_terrestres.Conductor c ON ot.id_conductor = c.id_conductor
LEFT JOIN operaciones_terrestres.Empleado e ON c.id_empleado = e.id_empleado
LEFT JOIN operaciones_terrestres.EspecialidadEmpleado ee ON e.id_especialidad_empleado = ee.id_especialidad_empleado
LEFT JOIN operaciones_terrestres.RutaTerrestre rtt ON ot.id_ruta_terrestre = rtt.id_ruta_terrestre
LEFT JOIN operaciones_terrestres.Ruta r ON rtt.id_ruta = r.id_ruta
LEFT JOIN operaciones_terrestres.ChecklistDespacho cd ON ot.id_operacion_terrestre = cd.id_operacion_terrestre;
```

## 4. Vista: v_gestor_incidencias_operaciones

**Pantalla:** P05 - Gestión de Incidencias en Transporte

**Descripción:** Muestra el listado completo de incidencias con información detallada de la operación, vehículo, conductor y estados para su gestión.

**Tablas incluidas:** Incidencia, TipoIncidencia, EstadoIncidencia, Operacion, OperacionTerrestre, Vehiculo, Conductor, Empleado, Usuario, Ruta

### Script de creación de vista

```sql
CREATE OR REPLACE VIEW operaciones_terrestres.v_gestor_incidencias_operaciones AS
SELECT 
    -- Información de la incidencia
    i.id_incidencia,
    i.codigo AS codigo_incidencia,
    i.descripcion,
    i.grado_severidad,
    i.fecha_hora,
    
    -- Tipo y estado de incidencia
    ti.id_tipo_incidencia,
    ti.nombre AS tipo_incidencia,
    ei.id_estado_incidencia,
    ei.nombre AS estado_incidencia,
    
    -- Información de la operación relacionada
    o.id_operacion,
    o.codigo AS codigo_operacion,
    o.fecha_inicio AS fecha_inicio_operacion,
    o.fecha_fin AS fecha_fin_operacion,
    eo.nombre AS estado_operacion,
    
    -- Información de operación terrestre
    ot.codigo AS codigo_operacion_terrestre,
    
    -- Información del vehículo
    v.placa AS placa_vehiculo,
    a.nombre AS nombre_vehiculo,
    tv.nombre AS tipo_vehiculo,
    ev.nombre AS estado_vehiculo,
    
    -- Información del conductor
    CONCAT(e.nombre, ' ', e.apellido) AS nombre_conductor,
    c.licencia,
    c.categoria,
    
    -- Información de la ruta
    r.origen,
    r.destino,
    r.duracion AS duracion_estimada_horas,
    
    -- Información del usuario que reportó
    u.id_usuario,
    u.correo_electronico AS email_usuario,
    CONCAT(eu.nombre, ' ', eu.apellido) AS nombre_usuario,
    ru.nombre AS rol_usuario,
    
    -- Cálculos y clasificaciones
    CASE 
        WHEN i.grado_severidad = 1 THEN 'Mínima'
        WHEN i.grado_severidad = 2 THEN 'Baja'
        WHEN i.grado_severidad = 3 THEN 'Media'
        WHEN i.grado_severidad = 4 THEN 'Alta'
        WHEN i.grado_severidad = 5 THEN 'Crítica'
        ELSE 'No especificada'
    END AS clasificacion_severidad,
    
    CASE 
        WHEN ei.nombre = 'Reportada' THEN 'Pendiente'
        WHEN ei.nombre = 'En investigación' THEN 'En Proceso'
        WHEN ei.nombre = 'Resuelta' THEN 'Resuelta'
        WHEN ei.nombre = 'Cerrada' THEN 'Cerrada'
        ELSE ei.nombre
    END AS estado_operacional,
    
    -- Tiempo transcurrido desde el reporte
    EXTRACT(EPOCH FROM (CURRENT_TIMESTAMP - i.fecha_hora)) / 3600 AS horas_desde_reporte,
    
    CASE 
        WHEN EXTRACT(EPOCH FROM (CURRENT_TIMESTAMP - i.fecha_hora)) / 3600 < 24 THEN '< 24h'
        WHEN EXTRACT(EPOCH FROM (CURRENT_TIMESTAMP - i.fecha_hora)) / 3600 < 72 THEN '1-3 días'
        ELSE '> 3 días'
    END AS antiguedad_incidencia,
    
    -- Indicadores de urgencia
    CASE 
        WHEN i.grado_severidad >= 4 AND ei.nombre IN ('Reportada', 'En investigación') THEN TRUE
        ELSE FALSE
    END AS requiere_atencion_urgente,
    
    CASE 
        WHEN EXTRACT(EPOCH FROM (CURRENT_TIMESTAMP - i.fecha_hora)) / 3600 > 48 
        AND ei.nombre IN ('Reportada', 'En investigación') THEN TRUE
        ELSE FALSE
    END AS atencion_retrasada,
    
    -- Información de progreso de la operación al momento de la incidencia
    CASE 
        WHEN i.fecha_hora BETWEEN o.fecha_inicio AND COALESCE(o.fecha_fin, CURRENT_TIMESTAMP) THEN
            ROUND((EXTRACT(EPOCH FROM (i.fecha_hora - o.fecha_inicio)) / 3600 / r.duracion) * 100, 2)
        ELSE NULL
    END AS progreso_operacion_al_incidente,
    
    -- Otras incidencias en la misma operación
    (SELECT COUNT(*) FROM operaciones_terrestres.Incidencia i2 
     WHERE i2.id_operacion = i.id_operacion 
     AND i2.id_incidencia != i.id_incidencia) AS otras_incidencias_misma_operacion,
    
    -- Historial de incidencias del vehículo
    (SELECT COUNT(*) FROM operaciones_terrestres.Incidencia i2 
     JOIN operaciones_terrestres.Operacion o2 ON i2.id_operacion = o2.id_operacion
     JOIN operaciones_terrestres.OperacionTerrestre ot2 ON o2.id_operacion = ot2.id_operacion
     WHERE ot2.id_vehiculo = v.id_vehiculo 
     AND i2.id_incidencia != i.id_incidencia
     AND i2.fecha_hora >= CURRENT_DATE - INTERVAL '30 days') AS incidencias_vehiculo_30_dias,
    
    -- Historial de incidencias del conductor
    (SELECT COUNT(*) FROM operaciones_terrestres.Incidencia i2 
     JOIN operaciones_terrestres.Operacion o2 ON i2.id_operacion = o2.id_operacion
     JOIN operaciones_terrestres.OperacionTerrestre ot2 ON o2.id_operacion = ot2.id_operacion
     WHERE ot2.id_conductor = c.id_conductor 
     AND i2.id_incidencia != i.id_incidencia
     AND i2.fecha_hora >= CURRENT_DATE - INTERVAL '30 days') AS incidencias_conductor_30_dias

FROM operaciones_terrestres.Incidencia i
INNER JOIN operaciones_terrestres.TipoIncidencia ti ON i.id_tipo_incidencia = ti.id_tipo_incidencia
INNER JOIN operaciones_terrestres.EstadoIncidencia ei ON i.id_estado_incidencia = ei.id_estado_incidencia
INNER JOIN operaciones_terrestres.Operacion o ON i.id_operacion = o.id_operacion
INNER JOIN operaciones_terrestres.EstadoOperacion eo ON o.id_estado_operacion = eo.id_estado_operacion
LEFT JOIN operaciones_terrestres.OperacionTerrestre ot ON o.id_operacion = ot.id_operacion
LEFT JOIN operaciones_terrestres.Vehiculo v ON ot.id_vehiculo = v.id_vehiculo
LEFT JOIN operaciones_terrestres.Activo a ON v.id_activo = a.id_activo
LEFT JOIN operaciones_terrestres.TipoVehiculo tv ON v.id_tipo_vehiculo = tv.id_tipo_vehiculo
LEFT JOIN operaciones_terrestres.EstadoVehiculo ev ON v.id_estado_vehiculo = ev.id_estado_vehiculo
LEFT JOIN operaciones_terrestres.Conductor c ON ot.id_conductor = c.id_conductor
LEFT JOIN operaciones_terrestres.Empleado e ON c.id_empleado = e.id_empleado
```

**Descripción:** Proporciona un resumen ejecutivo con métricas de mantenimiento, disponibilidad de activos, costos y estadísticas principales para el dashboard.

**Tablas incluidas:** Activo, EstadoActivo, OrdenMantenimiento, EstadoOrden, OperacionMantenimientoRepuesto, TipoMantenimiento

### Script de creación de vista

```sql
CREATE OR REPLACE VIEW mantenimiento_logistico.v_dashboard_mantenimiento AS
SELECT 
    -- Métricas de activos
    COUNT(DISTINCT a.id_activo) AS total_activos,
    COUNT(DISTINCT CASE WHEN ea.nombre = 'Operativo' THEN a.id_activo END) AS activos_operativos,
    COUNT(DISTINCT CASE WHEN ea.nombre = 'En mantenimiento' THEN a.id_activo END) AS activos_en_mantenimiento,
    COUNT(DISTINCT CASE WHEN ea.nombre = 'Fuera de servicio' THEN a.id_activo END) AS activos_fuera_servicio,
    
    -- Métricas de disponibilidad
    ROUND(
        (COUNT(DISTINCT CASE WHEN ea.nombre = 'Operativo' THEN a.id_activo END) * 100.0 / 
        NULLIF(COUNT(DISTINCT a.id_activo), 0)), 2
    ) AS porcentaje_disponibilidad,
    
    -- Métricas de órdenes de mantenimiento
    COUNT(DISTINCT om.id_orden) AS total_ordenes,
    COUNT(DISTINCT CASE WHEN eo.nombre = 'Abierta' THEN om.id_orden END) AS ordenes_abiertas,
    COUNT(DISTINCT CASE WHEN eo.nombre = 'En ejecución' THEN om.id_orden END) AS ordenes_en_ejecucion,
    COUNT(DISTINCT CASE WHEN eo.nombre = 'Completada' THEN om.id_orden END) AS ordenes_completadas,
    COUNT(DISTINCT CASE WHEN eo.nombre = 'Cancelada' THEN om.id_orden END) AS ordenes_canceladas,
    
    -- Métricas de costos
    COALESCE(SUM(omr.cantidad * omr.precio_unitario), 0) AS costos_materiales_totales,
    COALESCE(SUM(omr.cantidad * omr.precio_unitario * 0.5), 0) AS costos_mano_obra_estimados,
    COALESCE(SUM(omr.cantidad * omr.precio_unitario * 1.5), 0) AS costos_totales_estimados,
    
    -- Métricas de tipos de mantenimiento
    COUNT(DISTINCT CASE WHEN tm.nombre = 'Preventivo' THEN om.id_orden END) AS mantenimientos_preventivos,
    COUNT(DISTINCT CASE WHEN tm.nombre = 'Correctivo' THEN om.id_orden END) AS mantenimientos_correctivos,
    COUNT(DISTINCT CASE WHEN tm.nombre = 'Predictivo' THEN om.id_orden END) AS mantenimientos_predictivos,
    COUNT(DISTINCT CASE WHEN tm.nombre = 'Emergencia' THEN om.id_orden END) AS mantenimientos_emergencia,
    
    -- Métricas de eficiencia
    ROUND(
        AVG(CASE 
            WHEN om.fecha_cierre IS NOT NULL AND om.fecha_generada IS NOT NULL THEN
                EXTRACT(EPOCH FROM (om.fecha_cierre - om.fecha_generada)) / 86400.0
            ELSE NULL
        END)::NUMERIC, 2
    ) AS tiempo_promedio_resolucion_dias,
    
    COUNT(DISTINCT CASE 
        WHEN om.fecha_cierre IS NOT NULL AND om.fecha_generada IS NOT NULL AND
             EXTRACT(EPOCH FROM (om.fecha_cierre - om.fecha_generada)) / 86400.0 <= 7 
        THEN om.id_orden 
    END) AS ordenes_completadas_menos_7_dias,
    
    -- Métricas de repuestos
    (SELECT COUNT(*) FROM mantenimiento_logistico.Repuesto) AS total_repuestos,
    (SELECT COUNT(*) FROM mantenimiento_logistico.Repuesto WHERE stock <= stock_minimo) AS repuestos_stock_critico,
    
    -- Métricas de técnicos
    (SELECT COUNT(*) FROM mantenimiento_logistico.Tecnico) AS total_tecnicos,
    
    -- Solicitudes de mantenimiento
    (SELECT COUNT(*) FROM mantenimiento_logistico.SolicitudMantenimiento) AS total_solicitudes,
    (SELECT COUNT(*) FROM mantenimiento_logistico.SolicitudMantenimiento sm
     JOIN mantenimiento_logistico.EstadoSolicitud es ON sm.id_estado_solicitud = es.id_estado_solicitud
     WHERE es.nombre = 'Pendiente') AS solicitudes_pendientes,
    
    -- Planes de mantenimiento
    (SELECT COUNT(*) FROM mantenimiento_logistico.PlanMantenimiento) AS total_planes,
    (SELECT COUNT(*) FROM mantenimiento_logistico.PlanMantenimiento pm
     JOIN mantenimiento_logistico.EstadoPlan ep ON pm.id_estado_plan = ep.id_estado_plan

    -- Timestamp de consulta
    CURRENT_TIMESTAMP AS fecha_consulta
    
FROM mantenimiento_logistico.Activo a
INNER JOIN mantenimiento_logistico.EstadoActivo ea ON a.id_estado_activo = ea.id_estado_activo
LEFT JOIN mantenimiento_logistico.PlanMantenimiento pm ON a.id_activo = pm.id_activo
LEFT JOIN mantenimiento_logistico.OperacionMantenimiento opm ON (
    opm.id_plan_mantenimiento = pm.id_plan_mantenimiento OR 
    opm.id_solicitud_mantenimiento IN (
        SELECT id_solicitud_mantenimiento FROM mantenimiento_logistico.SolicitudMantenimiento 
        WHERE id_activo = a.id_activo
    )
)
LEFT JOIN mantenimiento_logistico.OrdenMantenimiento om ON opm.id_operacion_mantenimiento = om.id_operacion_mantenimiento
LEFT JOIN mantenimiento_logistico.EstadoOrden eo ON om.id_estado_orden = eo.id_estado_orden
LEFT JOIN mantenimiento_logistico.TipoMantenimiento tm ON om.id_tipo_mantenimiento = tm.id_tipo_mantenimiento
```

## 4. Vista: v_gestor_inventario_repuestos

**Pantalla:** P04 - Gestión de Inventario de Repuestos

**Descripción:** Muestra el inventario completo de repuestos con información de stock, estados críticos y estadísticas de uso para su gestión.

**Tablas incluidas:** Repuesto, OperacionMantenimientoRepuesto, OperacionMantenimiento, OrdenMantenimiento

### Script de creación de vista

```sql
CREATE OR REPLACE VIEW mantenimiento_logistico.v_gestor_inventario_repuestos AS
SELECT 
    -- Información básica del repuesto
    r.id_repuesto,
    r.codigo AS codigo_repuesto,
    r.nombre AS nombre_repuesto,
    r.stock,
    r.stock_minimo,
    
    -- Cálculos de stock
    (r.stock - r.stock_minimo) AS margen_seguridad,
    CASE 
        WHEN r.stock = 0 THEN 'Agotado'
        WHEN r.stock <= r.stock_minimo THEN 'Crítico'
        WHEN r.stock <= r.stock_minimo * 1.5 THEN 'Bajo'
        ELSE 'Satisfactorio'
    END AS estado_stock,
    
    CASE 
        WHEN r.stock <= r.stock_minimo THEN TRUE
        ELSE FALSE
    END AS requiere_reabastecimiento,
    
    -- Estadísticas de uso
    COALESCE(SUM(omr.cantidad), 0) AS total_utilizado,
    COALESCE(COUNT(DISTINCT omr.id_operacion_mantenimiento), 0) AS veces_utilizado,
    
    -- Uso reciente (últimos 30 días)
    COALESCE(SUM(CASE 
        WHEN omr.fecha_uso >= CURRENT_DATE - INTERVAL '30 days' THEN omr.cantidad 
        ELSE 0 
    END), 0) AS utilizado_ultimos_30_dias,
    
    COALESCE(COUNT(DISTINCT CASE 
        WHEN omr.fecha_uso >= CURRENT_DATE - INTERVAL '30 days' THEN omr.id_operacion_mantenimiento 
        ELSE NULL 
    END), 0) AS veces_utilizado_ultimos_30_dias,
    
    -- Costos
    COALESCE(AVG(omr.precio_unitario), 0)::DECIMAL(10,2) AS precio_promedio,
    COALESCE(SUM(omr.cantidad * omr.precio_unitario), 0) AS costo_total_utilizado,
    
    -- Fechas importantes
    MAX(omr.fecha_uso) AS fecha_ultimo_uso,
    MIN(omr.fecha_uso) AS fecha_primer_uso,
    
    -- Órdenes donde se ha utilizado
    (SELECT STRING_AGG(DISTINCT om.codigo, ', ' ORDER BY om.codigo)
     FROM mantenimiento_logistico.OperacionMantenimientoRepuesto omr2
     JOIN mantenimiento_logistico.OperacionMantenimiento opm ON omr2.id_operacion_mantenimiento = opm.id_operacion_mantenimiento
     JOIN mantenimiento_logistico.OrdenMantenimiento om ON opm.id_operacion_mantenimiento = om.id_operacion_mantenimiento
     WHERE omr2.id_repuesto = r.id_repuesto
     LIMIT 5) AS ordenes_recientes,
    
    -- Tipos de mantenimiento donde se usa
    (SELECT STRING_AGG(DISTINCT tm.nombre, ', ')
     FROM mantenimiento_logistico.OperacionMantenimientoRepuesto omr2
     JOIN mantenimiento_logistico.OperacionMantenimiento opm ON omr2.id_operacion_mantenimiento = opm.id_operacion_mantenimiento
     JOIN mantenimiento_logistico.OrdenMantenimiento om ON opm.id_operacion_mantenimiento = om.id_operacion_mantenimiento
     JOIN mantenimiento_logistico.TipoMantenimiento tm ON om.id_tipo_mantenimiento = tm.id_tipo_mantenimiento
     WHERE omr2.id_repuesto = r.id_repuesto) AS tipos_mantenimiento_utilizados,
    
    -- Tasa de uso mensual promedio
    CASE 
        WHEN COUNT(DISTINCT DATE_TRUNC('month', omr.fecha_uso)) > 0 THEN
            COALESCE(SUM(omr.cantidad), 0) / COUNT(DISTINCT DATE_TRUNC('month', omr.fecha_uso))
        ELSE 0
    END::DECIMAL(10,2) AS uso_mensual_promedio,
    
    -- Proyección de stock
    CASE 
        WHEN (COALESCE(SUM(omr.cantidad), 0) / NULLIF(COUNT(DISTINCT DATE_TRUNC('month', omr.fecha_uso)), 0)) > 0 THEN
            r.stock / (COALESCE(SUM(omr.cantidad), 0) / NULLIF(COUNT(DISTINCT DATE_TRUNC('month', omr.fecha_uso)), 0))
        ELSE NULL
    END::DECIMAL(10,1) AS meses_restantes_stock,
    
    -- Alertas
    CASE 
        WHEN r.stock = 0 THEN 'Stock Agotado - Pedido Urgente'
        WHEN r.stock <= r.stock_minimo THEN 'Stock Crítico - Reabastecer'
        WHEN r.stock <= r.stock_minimo * 2 THEN 'Stock Bajo - Monitorear'
        ELSE 'Stock Adecuado'
    END AS alerta_inventario,
    
    -- Información para reorden
    CASE 
        WHEN r.stock <= r.stock_minimo THEN GREATEST(r.stock_minimo * 3 - r.stock, 10)
        ELSE 0
    END AS cantidad_sugerida_reorden,
    
    -- Valor total en inventario
    (r.stock * COALESCE(AVG(omr.precio_unitario), 0))::DECIMAL(10,2) AS valor_total_inventario

FROM mantenimiento_logistico.Repuesto r
LEFT JOIN mantenimiento_logistico.OperacionMantenimientoRepuesto omr ON r.id_repuesto = omr.id_repuesto
GROUP BY r.id_repuesto, r.codigo, r.nombre, r.stock, r.stock_minimo
ORDER BY 
    CASE 
        WHEN r.stock = 0 THEN 1
        WHEN r.stock <= r.stock_minimo THEN 2
        WHEN r.stock <= r.stock_minimo * 1.5 THEN 3
        ELSE 4
    END,
    r.codigo;
```
