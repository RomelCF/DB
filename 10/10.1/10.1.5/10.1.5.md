```sql
-- Insertar un contrato base
INSERT INTO monitoreo.Contrato (id_contrato, fecha_emision, fecha_vencimiento, id_estado_contrato)
SELECT 
    gen_random_uuid(),
    CURRENT_DATE - INTERVAL '1 year',
    CURRENT_DATE + INTERVAL '2 years',
    id_estado_contrato

LIMIT 1

-- Insertar empleados base
DO $$
DECLARE
    v_id_contrato UUID;
    v_id_rol_operador UUID;
BEGIN
    SELECT id_contrato INTO v_id_contrato FROM monitoreo.Contrato LIMIT 1;
    SELECT id_especialidad_empleado INTO v_id_rol_operador FROM monitoreo.EspecialidadEmpleado WHERE nombre = 'Operador' LIMIT 1;
    
    FOR i IN 1..10 LOOP
        INSERT INTO monitoreo.Empleado (
            id_empleado, codigo, dni, nombre, apellido, 
            direccion, id_especialidad_empleado, id_contrato
        )
        VALUES (
            gen_random_uuid(),
            'EMP-' || LPAD(i::TEXT, 4, '0'),
            LPAD((10000000 + i)::TEXT, 8, '0'),
            'Empleado' || i,
            'Apellido' || i,
            'Dirección ' || i,
            v_id_rol_operador,
            floor(random() * 10 + 1),
            v_id_contrato
        );
    END LOOP;
    
    RAISE NOTICE 'Empleados base insertados: 10';
END $$;
```

- **Script para poblar datos masivos de prueba**

```sql
DO $$
DECLARE
    v_start_time TIMESTAMP := clock_timestamp();
    v_id_estado_en_curso UUID;
    v_id_estado_disponible UUID;
    v_id_estado_transito UUID;
    v_id_rol_sensor UUID;
    v_id_operador UUID;
    v_tipos_contenedor UUID[];
    v_tipos_sensor UUID[];
BEGIN
    RAISE NOTICE '========================================';
    RAISE NOTICE 'CARGA REDUCIDA - INICIO';
    RAISE NOTICE '========================================';
    
    -- Obtener IDs de catálogo
    SELECT id_estado_operacion INTO v_id_estado_en_curso 
    FROM monitoreo.EstadoOperacion WHERE nombre = 'En curso' LIMIT 1;
    
    SELECT id_estado_contenedor INTO v_id_estado_disponible 
    FROM monitoreo.EstadoContenedor WHERE nombre = 'Disponible' LIMIT 1;
    
    SELECT id_estado_contenedor INTO v_id_estado_transito 
    FROM monitoreo.EstadoContenedor WHERE nombre = 'En tránsito' LIMIT 1;
    
    SELECT id_rol_sensor INTO v_id_rol_sensor 
    FROM monitoreo.RolSensor WHERE nombre = 'Principal' LIMIT 1;
    
    SELECT id_operador INTO v_id_operador 
    FROM monitoreo.Operador LIMIT 1;
    
    SELECT ARRAY_AGG(id_tipo_contenedor) INTO v_tipos_contenedor
    FROM monitoreo.TipoContenedor;
    
    SELECT ARRAY_AGG(id_tipo_sensor) INTO v_tipos_sensor
    FROM monitoreo.TipoSensor;
    
    -- =====================================
    -- 1. Operaciones (2,000)
    -- =====================================
    RAISE NOTICE '1/4: Insertando 2,000 operaciones...';
    
    INSERT INTO monitoreo.Operacion (id_operacion, codigo, fecha_inicio, id_estado_operacion)
    SELECT 
        gen_random_uuid(),
        'OP-2024-' || LPAD(i::TEXT, 6, '0'),
        CURRENT_TIMESTAMP - (random() * INTERVAL '365 days'),
        v_id_estado_en_curso
    FROM generate_series(1, 2000) i;
    
    INSERT INTO monitoreo.OperacionMonitoreo (id_operacion_monitoreo, id_operacion)
    SELECT gen_random_uuid(), id_operacion
    FROM monitoreo.Operacion
    WHERE codigo LIKE 'OP-2024-%';
    
    INSERT INTO monitoreo.OperadorOperacionMonitoreo (id_operador_operacion_monitoreo, id_operador, id_operacion_monitoreo, fecha_realizacion)
    SELECT gen_random_uuid(), v_id_operador, id_operacion_monitoreo, CURRENT_DATE
    FROM monitoreo.OperacionMonitoreo om
    WHERE EXISTS (SELECT 1 FROM monitoreo.Operacion o WHERE o.id_operacion = om.id_operacion AND o.codigo LIKE 'OP-2024-%');
    
    RAISE NOTICE '  ✓ Operaciones completadas (%.1f seg)', 
        EXTRACT(EPOCH FROM (clock_timestamp() - v_start_time));
    
    -- =====================================
    -- 2. Contenedores (10,000)
    -- =====================================
    RAISE NOTICE '2/4: Insertando 10,000 contenedores...';
    
    -- Generar contenedores con peso <= capacidad (respeta constraint)
    INSERT INTO monitoreo.Contenedor (
        id_contenedor, codigo, peso, capacidad, dimensiones,
        id_estado_contenedor, id_tipo_contenedor
    )
    SELECT 
        gen_random_uuid(),
        'CONT-' || LPAD(i::TEXT, 6, '0'),
        capacidad_temp * (0.3 + random() * 0.6), -- Peso entre 30% y 90% de capacidad
        capacidad_temp, -- Capacidad
        '20x8x8',
        CASE WHEN random() < 0.5 THEN v_id_estado_disponible ELSE v_id_estado_transito END,
        v_tipos_contenedor[1 + floor(random() * array_length(v_tipos_contenedor, 1))]
    FROM (
        SELECT 
            i,
            (10000 + random() * 20000)::DECIMAL AS capacidad_temp
        FROM generate_series(1, 10000) i
    ) subquery;
    
    RAISE NOTICE '  ✓ Contenedores completados (%.1f seg)', 
        EXTRACT(EPOCH FROM (clock_timestamp() - v_start_time));
    
    -- =====================================
    -- 3. Sensores (30,000 = 3 por contenedor)
    -- =====================================
    RAISE NOTICE '3/4: Insertando 30,000 sensores...';
    
    INSERT INTO monitoreo.Sensor (id_sensor, codigo, nombre, id_tipo_sensor, id_rol_sensor, id_contenedor)
    SELECT 
        gen_random_uuid(),
        c.codigo || '-S' || s.n,
        'Sensor ' || s.n,
        v_tipos_sensor[1 + floor(random() * array_length(v_tipos_sensor, 1))],
        v_id_rol_sensor,
        c.id_contenedor
    FROM monitoreo.Contenedor c
    CROSS JOIN (SELECT generate_series(1, 3) AS n) s
    WHERE c.codigo LIKE 'CONT-%';
    
    RAISE NOTICE '  ✓ Sensores completados (%.1f seg)', 
        EXTRACT(EPOCH FROM (clock_timestamp() - v_start_time));
    
    -- =====================================
    -- 4. Posiciones GPS (100,000 = 10 por contenedor)
    -- =====================================
    RAISE NOTICE '4/4: Insertando 100,000 posiciones GPS...';
    
    INSERT INTO monitoreo.PosicionContenedor (id_contenedor, latitud, longitud, fecha_hora)
    SELECT 
        c.id_contenedor,
        -12.0 + (random() * 0.5),
        -77.0 + (random() * 0.5),
        CURRENT_TIMESTAMP - ((pos.n * 30 + random() * 30) * INTERVAL '1 minute')
    FROM monitoreo.Contenedor c
    CROSS JOIN (SELECT generate_series(1, 10) AS n) pos
    WHERE c.codigo LIKE 'CONT-%';
    
    RAISE NOTICE '  ✓ Posiciones GPS completadas (%.1f seg)', 
        EXTRACT(EPOCH FROM (clock_timestamp() - v_start_time));
    
    RAISE NOTICE '========================================';
    RAISE NOTICE 'CARGA COMPLETADA EN: %.1f segundos', 
        EXTRACT(EPOCH FROM (clock_timestamp() - v_start_time));
    RAISE NOTICE '========================================';
    RAISE NOTICE 'Datos insertados:';
    RAISE NOTICE '  - Operaciones: 2,000';
    RAISE NOTICE '  - Contenedores: 10,000';
    RAISE NOTICE '  - Sensores: 30,000';
    RAISE NOTICE '  - Posiciones GPS: 100,000';
    RAISE NOTICE '========================================';
    
END $$;
```

>En este caso ingrese 2000 operaciones, 10000 contenedores, 30000 sensores y 100000 posiciones GPS

**1. CONSULTA 1: Últimas posiciones de contenedores (P04 - Query 4.1)**

- **Script de análisis**

```sql
-- Script de análisis
EXPLAIN (ANALYZE, BUFFERS, FORMAT TEXT)
WITH UltimasPosicionesContenedor AS (
    SELECT DISTINCT ON (id_contenedor)
        id_contenedor,
        latitud,
        longitud,
        fecha_hora
    FROM monitoreo.PosicionContenedor
    ORDER BY id_contenedor, fecha_hora DESC
)
SELECT 
    c.id_contenedor,
    c.codigo,
    ec.nombre AS estado,
    tc.nombre AS tipo,
    upc.latitud,
    upc.longitud,
    upc.fecha_hora AS ultima_actualizacion
FROM monitoreo.Contenedor c
JOIN monitoreo.EstadoContenedor ec ON c.id_estado_contenedor = ec.id_estado_contenedor
JOIN monitoreo.TipoContenedor tc ON c.id_tipo_contenedor = tc.id_tipo_contenedor
JOIN UltimasPosicionesContenedor upc ON upc.id_contenedor = c.id_contenedor
WHERE ec.nombre IN ('En tránsito', 'En preparación')
ORDER BY c.codigo
LIMIT 100;
```

- **Creación de Indices**

```sql
CREATE INDEX IF NOT EXISTS idx_posicioncontenedor_id_fecha 
ON monitoreo.PosicionContenedor (id_contenedor, fecha_hora DESC);

DROP INDEX monitoreo.idx_posicioncontenedor_id_fecha CASCADE;
```
