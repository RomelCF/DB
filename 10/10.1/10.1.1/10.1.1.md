- **Script para poblar datos masivos de prueba**

```sql
-- ============================================================
-- CARGA MASIVA DE DATOS - MÓDULO GESTIÓN DE RESERVAS
-- Schema: gestion_reserva
-- ============================================================

DO $$
DECLARE
    v_start_time TIMESTAMP := clock_timestamp();
    v_id_estado_confirmada UUID;
    v_id_estado_activa UUID;
    v_id_estado_disponible UUID;
    v_id_estado_transito UUID;
    v_id_estado_progreso UUID;
    v_id_estado_completada UUID;
    v_clientes TEXT[];
    v_agentes UUID[];
    v_buques TEXT[];
    v_rutas UUID[];
    v_tipos_contenedor UUID[];
    v_total_reservas INT := 5000;
    v_total_contenedores INT := 15000;
    v_total_operaciones INT := 8000;
BEGIN
    RAISE NOTICE '========================================';
    RAISE NOTICE 'CARGA MASIVA - GESTIÓN DE RESERVAS';
    RAISE NOTICE '========================================';
    
    -- Obtener IDs de catálogos
    SELECT id_estado_reserva INTO v_id_estado_confirmada 
    FROM gestion_reserva.EstadoReserva WHERE nombre = 'Confirmada' LIMIT 1;
    
    SELECT id_estado_reserva INTO v_id_estado_activa 
    FROM gestion_reserva.EstadoReserva WHERE nombre = 'Activa' LIMIT 1;
    
    SELECT id_estado_contenedor INTO v_id_estado_disponible 
    FROM gestion_reserva.EstadoContenedor WHERE nombre = 'Disponible' LIMIT 1;
    
    SELECT id_estado_contenedor INTO v_id_estado_transito 
    FROM gestion_reserva.EstadoContenedor WHERE nombre = 'En tránsito' LIMIT 1;
    
    SELECT id_estado_operacion INTO v_id_estado_progreso 
    FROM gestion_reserva.EstadoOperacion WHERE nombre = 'En Progreso' LIMIT 1;
    
    SELECT id_estado_operacion INTO v_id_estado_completada 
    FROM gestion_reserva.EstadoOperacion WHERE nombre = 'Completada' LIMIT 1;
    
    -- Obtener arrays de IDs
    SELECT ARRAY_AGG(ruc) INTO v_clientes FROM gestion_reserva.Cliente;
    SELECT ARRAY_AGG(id_agente_reservas) INTO v_agentes FROM gestion_reserva.AgenteReservas;
    SELECT ARRAY_AGG(matricula) INTO v_buques FROM gestion_reserva.Buque;
    SELECT ARRAY_AGG(id_ruta) INTO v_rutas FROM gestion_reserva.Ruta;
    SELECT ARRAY_AGG(id_tipo_contenedor) INTO v_tipos_contenedor FROM gestion_reserva.TipoContenedor;
    
    RAISE NOTICE 'Catálogos cargados: % clientes, % agentes, % buques, % rutas', 
        array_length(v_clientes, 1), array_length(v_agentes, 1), 
        array_length(v_buques, 1), array_length(v_rutas, 1);
    
    -- =====================================
    -- 1. CONTENEDORES (15,000)
    -- =====================================
    RAISE NOTICE '1/5: Insertando % contenedores...', v_total_contenedores;
    
    INSERT INTO gestion_reserva.Contenedor (
        id_contenedor, codigo, peso, capacidad, dimensiones,
        id_estado_contenedor, id_tipo_contenedor
    )
    SELECT 
        gen_random_uuid(),
        'CONT-' || LPAD(i::TEXT, 6, '0'),
        capacidad_temp * (0.2 + random() * 0.6),
        capacidad_temp,
        CASE (random() * 3)::INT
            WHEN 0 THEN '20x8x8.5 ft'
            WHEN 1 THEN '40x8x8.5 ft'
            ELSE '40x8x9.5 ft'
        END,
        CASE WHEN random() < 0.7 THEN v_id_estado_disponible ELSE v_id_estado_transito END,
        v_tipos_contenedor[1 + floor(random() * array_length(v_tipos_contenedor, 1))]
    FROM (
        SELECT 
            i,
            (15000 + random() * 15000)::DECIMAL AS capacidad_temp
        FROM generate_series(1, v_total_contenedores) i
    ) subquery;
    
    -- Insertar mercancías para contenedores
    INSERT INTO gestion_reserva.ContenedorMercancia (
        id_contenedor_mercancia, id_contenedor, tipo_mercancia
    )
    SELECT 
        gen_random_uuid(),
        id_contenedor,
        CASE (random() * 6)::INT
            WHEN 0 THEN 'Electrónicos'
            WHEN 1 THEN 'Textiles'
            WHEN 2 THEN 'Alimentos'
            WHEN 3 THEN 'Maquinaria'
            WHEN 4 THEN 'Productos Químicos'
            ELSE 'Materiales de Construcción'
        END
    FROM gestion_reserva.Contenedor
    WHERE codigo LIKE 'CONT-%'
    AND random() < 0.8;
    
    RAISE NOTICE '  ✓ Contenedores completados (%.2f seg)', 
        EXTRACT(EPOCH FROM (clock_timestamp() - v_start_time));
    
    -- =====================================
    -- 2. OPERACIONES (8,000)
    -- =====================================
    RAISE NOTICE '2/5: Insertando % operaciones...', v_total_operaciones;
    
    INSERT INTO gestion_reserva.Operacion (
        id_operacion, codigo, fecha_inicio, fecha_fin, id_estado_operacion
    )
    SELECT 
        gen_random_uuid(),
        'OP-2024-' || LPAD(i::TEXT, 6, '0'),
        fecha_inicio_temp,
        CASE 
            WHEN random() < 0.4 THEN fecha_inicio_temp + (random() * INTERVAL '30 days')
            ELSE NULL
        END,
        CASE 
            WHEN random() < 0.4 THEN v_id_estado_completada
            ELSE v_id_estado_progreso
        END
    FROM (
        SELECT 
            i,
            CURRENT_TIMESTAMP - (random() * INTERVAL '180 days') AS fecha_inicio_temp
        FROM generate_series(1, v_total_operaciones) i
    ) subquery;
    
    -- Insertar Operaciones Marítimas (50% de operaciones)
    INSERT INTO gestion_reserva.OperacionMaritima (
        id_operacion_maritima, id_operacion, codigo, cantidad_contenedores,
        porcentaje_trayecto, id_buque
    )
    SELECT 
        gen_random_uuid(),
        o.id_operacion,
        'OM-' || o.codigo,
        (50 + random() * 200)::INT,
        (random() * 100)::DECIMAL(5,2),
        (SELECT id_buque FROM gestion_reserva.Buque ORDER BY random() LIMIT 1)
    FROM gestion_reserva.Operacion o
    WHERE o.codigo LIKE 'OP-2024-%'
    AND random() < 0.5;
    
    -- Insertar Operaciones Terrestres (50% de operaciones)
    INSERT INTO gestion_reserva.OperacionTerrestre (
        id_operacion_terrestre, id_operacion, codigo, costo_operacion_terrestre
    )
    SELECT 
        gen_random_uuid(),
        o.id_operacion,
        'OT-' || o.codigo,
        (500 + random() * 2000)::DECIMAL(10,2)
    FROM gestion_reserva.Operacion o
    WHERE o.codigo LIKE 'OP-2024-%'
    AND NOT EXISTS (
        SELECT 1 FROM gestion_reserva.OperacionMaritima om 
        WHERE om.id_operacion = o.id_operacion
    );
    
    RAISE NOTICE '  ✓ Operaciones completadas (%.2f seg)', 
        EXTRACT(EPOCH FROM (clock_timestamp() - v_start_time));
    
    -- =====================================
    -- 3. RESERVAS (5,000)
    -- =====================================
    RAISE NOTICE '3/5: Insertando % reservas...', v_total_reservas;
    
    INSERT INTO gestion_reserva.Reserva (
        id_reserva, codigo, fecha_registro, id_estado_reserva,
        pago_total, ruc_cliente, id_agente_reservas, matricula_buque, id_ruta
    )
    SELECT 
        gen_random_uuid(),
        'RES-' || TO_CHAR(fecha_temp, 'YYYYMMDD') || '-' || LPAD(i::TEXT, 4, '0'),
        fecha_temp,
        CASE WHEN random() < 0.7 THEN v_id_estado_confirmada ELSE v_id_estado_activa END,
        (5000 + random() * 45000)::DECIMAL(10,2),
        v_clientes[1 + floor(random() * array_length(v_clientes, 1))],
        v_agentes[1 + floor(random() * array_length(v_agentes, 1))],
        v_buques[1 + floor(random() * array_length(v_buques, 1))],
        v_rutas[1 + floor(random() * array_length(v_rutas, 1))]
    FROM (
        SELECT 
            i,
            CURRENT_DATE - (random() * 90)::INT AS fecha_temp
        FROM generate_series(1, v_total_reservas) i
    ) subquery;
    
    RAISE NOTICE '  ✓ Reservas completadas (%.2f seg)', 
        EXTRACT(EPOCH FROM (clock_timestamp() - v_start_time));
    
    -- =====================================
    -- 4. RESERVA CONTENEDOR (15,000)
    -- =====================================
    RAISE NOTICE '4/5: Vinculando contenedores con reservas...';
    
    INSERT INTO gestion_reserva.ReservaContenedor (
        id_reserva_contenedor, id_reserva, id_contenedor,
        fecha_asignacion, cantidad
    )
    SELECT 
        gen_random_uuid(),
        r.id_reserva,
        c.id_contenedor,
        r.fecha_registro,
        (1 + random() * 3)::INT
    FROM gestion_reserva.Reserva r
    CROSS JOIN LATERAL (
        SELECT id_contenedor 
        FROM gestion_reserva.Contenedor 
        ORDER BY random() 
        LIMIT (1 + random() * 5)::INT
    ) c
    WHERE r.codigo LIKE 'RES-%';
    
    RAISE NOTICE '  ✓ Vinculación de contenedores completada (%.2f seg)', 
        EXTRACT(EPOCH FROM (clock_timestamp() - v_start_time));
    
    -- =====================================
    -- 5. VINCULACIONES DE RESERVAS
    -- =====================================
    RAISE NOTICE '5/5: Vinculando reservas con operaciones...';
    
    -- Vincular con operaciones marítimas
    INSERT INTO gestion_reserva.ReservaOperacionMaritima (
        id_reserva_operacion_maritima, id_reserva, id_operacion_maritima, fecha_vinculacion
    )
    SELECT 
        gen_random_uuid(),
        r.id_reserva,
        om.id_operacion_maritima,
        r.fecha_registro
    FROM gestion_reserva.Reserva r
    CROSS JOIN LATERAL (
        SELECT id_operacion_maritima 
        FROM gestion_reserva.OperacionMaritima 
        ORDER BY random() 
        LIMIT 1
    ) om
    WHERE r.codigo LIKE 'RES-%'
    AND random() < 0.6;
    
    -- Vincular con operaciones terrestres
    INSERT INTO gestion_reserva.ReservaOperacionTerrestre (
        id_reserva_operacion_terrestre, id_reserva, id_operacion_terrestre, fecha_vinculacion
    )
    SELECT 
        gen_random_uuid(),
        r.id_reserva,
        ot.id_operacion_terrestre,
        r.fecha_registro
    FROM gestion_reserva.Reserva r
    CROSS JOIN LATERAL (
        SELECT id_operacion_terrestre 
        FROM gestion_reserva.OperacionTerrestre 
        ORDER BY random() 
        LIMIT 1
    ) ot
    WHERE r.codigo LIKE 'RES-%'
    AND random() < 0.5;
    
    RAISE NOTICE '  ✓ Vinculaciones completadas (%.2f seg)', 
        EXTRACT(EPOCH FROM (clock_timestamp() - v_start_time));
    
    -- =====================================
    -- RESUMEN FINAL
    -- =====================================
    RAISE NOTICE '========================================';
    RAISE NOTICE 'CARGA COMPLETADA EN: %.2f segundos', 
        EXTRACT(EPOCH FROM (clock_timestamp() - v_start_time));
    RAISE NOTICE '========================================';
    RAISE NOTICE 'Datos insertados:';
    RAISE NOTICE '  - Contenedores: %', v_total_contenedores;
    RAISE NOTICE '  - Operaciones: %', v_total_operaciones;
    RAISE NOTICE '  - Reservas: %', v_total_reservas;
    RAISE NOTICE '  - ReservaContenedor: ~%', (SELECT COUNT(*) FROM gestion_reserva.ReservaContenedor);
    RAISE NOTICE '========================================';
    
END $$;
```

## **1. CONSULTA 1: Panel de Reservas Activas con Agregaciones Complejas**

**Descripción**: Consulta que obtiene todas las reservas activas con información detallada del cliente, buque, ruta y contenedores asociados, incluyendo agregaciones y concatenaciones.

### **Script de análisis**

```sql
EXPLAIN (ANALYZE, BUFFERS, FORMAT TEXT)
SELECT 
    r.codigo AS codigo_reserva,
    c.razon_social AS cliente,
    c.ruc,
    er.nombre AS estado_reserva,
    r.fecha_registro,
    r.pago_total,
    b.nombre AS nombre_buque,
    b.matricula,
    b.capacidad AS capacidad_buque,
    rt.codigo AS codigo_ruta,
    rt.origen,
    rt.destino,
    rt.duracion AS duracion_dias,
    rt.tarifa AS tarifa_ruta,
    COUNT(DISTINCT rc.id_contenedor) AS cantidad_contenedores,
    SUM(rc.cantidad) AS total_unidades_contenedor,
    STRING_AGG(DISTINCT tc.nombre, ', ') AS tipos_contenedor,
    AVG(tc.costo)::DECIMAL(10,2) AS costo_promedio_contenedor,
    COUNT(DISTINCT rom.id_operacion_maritima) AS operaciones_maritimas,
    COUNT(DISTINCT rot.id_operacion_terrestre) AS operaciones_terrestres
FROM gestion_reserva.Reserva r
INNER JOIN gestion_reserva.Cliente c ON r.ruc_cliente = c.ruc
INNER JOIN gestion_reserva.EstadoReserva er ON r.id_estado_reserva = er.id_estado_reserva
INNER JOIN gestion_reserva.Buque b ON r.matricula_buque = b.matricula
INNER JOIN gestion_reserva.Ruta rt ON r.id_ruta = rt.id_ruta
LEFT JOIN gestion_reserva.ReservaContenedor rc ON r.id_reserva = rc.id_reserva
LEFT JOIN gestion_reserva.Contenedor cnt ON rc.id_contenedor = cnt.id_contenedor
LEFT JOIN gestion_reserva.TipoContenedor tc ON cnt.id_tipo_contenedor = tc.id_tipo_contenedor
LEFT JOIN gestion_reserva.ReservaOperacionMaritima rom ON r.id_reserva = rom.id_reserva
LEFT JOIN gestion_reserva.ReservaOperacionTerrestre rot ON r.id_reserva = rot.id_reserva
WHERE er.nombre IN ('Confirmada', 'Activa')
AND r.fecha_registro >= CURRENT_DATE - INTERVAL '60 days'
GROUP BY r.id_reserva, r.codigo, c.razon_social, c.ruc, er.nombre, 
         r.fecha_registro, r.pago_total, b.nombre, b.matricula, b.capacidad,
         rt.codigo, rt.origen, rt.destino, rt.duracion, rt.tarifa
HAVING COUNT(DISTINCT rc.id_contenedor) > 0
ORDER BY r.fecha_registro DESC, r.pago_total DESC
LIMIT 100;
```

## **2. CONSULTA 2: Análisis Detallado de Clientes con Historial de Reservas**

**Descripción**: Consulta compleja que analiza el perfil completo de clientes con estadísticas de reservas, rutas utilizadas, atenciones recibidas y métricas de ingresos generados.

### **Script de análisis**

```sql
EXPLAIN (ANALYZE, BUFFERS, FORMAT TEXT)
SELECT 
    c.ruc,
    c.razon_social,
    c.direccion,
    c.email,
    STRING_AGG(DISTINCT ct.telefono || ' (' || tt.nombre || ')', ', ') AS telefonos,
    COUNT(DISTINCT r.id_reserva) AS total_reservas,
    COUNT(DISTINCT CASE WHEN er.nombre = 'Finalizada' THEN r.id_reserva END) AS reservas_completadas,
    COUNT(DISTINCT CASE WHEN er.nombre = 'Confirmada' THEN r.id_reserva END) AS reservas_confirmadas,
    COUNT(DISTINCT CASE WHEN er.nombre = 'Activa' THEN r.id_reserva END) AS reservas_activas,
    COUNT(DISTINCT CASE WHEN er.nombre = 'Cancelada' THEN r.id_reserva END) AS reservas_canceladas,
    COALESCE(SUM(CASE WHEN er.nombre = 'Finalizada' THEN r.pago_total END), 0) AS ingresos_totales_generados,
    COALESCE(AVG(CASE WHEN er.nombre = 'Finalizada' THEN r.pago_total END), 0)::DECIMAL(10,2) AS promedio_pago_por_reserva,
    MAX(r.fecha_registro) AS ultima_reserva,
    STRING_AGG(DISTINCT rt.origen || ' → ' || rt.destino, '; ') AS rutas_utilizadas,
    COUNT(DISTINCT ac.id_atencion_cliente) AS atenciones_recibidas,
    MAX(ac.fecha_atencion) AS ultima_atencion,
    COUNT(DISTINCT r.matricula_buque) AS buques_diferentes_utilizados,
    COUNT(DISTINCT rc.id_contenedor) AS total_contenedores_reservados,
    COALESCE(SUM(rc.cantidad), 0) AS total_unidades_contenedores,
    (
        SELECT COUNT(DISTINCT rom.id_operacion_maritima)
        FROM gestion_reserva.Reserva r2
        INNER JOIN gestion_reserva.ReservaOperacionMaritima rom ON r2.id_reserva = rom.id_reserva
        WHERE r2.ruc_cliente = c.ruc
    ) AS operaciones_maritimas_asociadas,
    (
        SELECT COUNT(DISTINCT rot.id_operacion_terrestre)
        FROM gestion_reserva.Reserva r3
        INNER JOIN gestion_reserva.ReservaOperacionTerrestre rot ON r3.id_reserva = rot.id_reserva
        WHERE r3.ruc_cliente = c.ruc
    ) AS operaciones_terrestres_asociadas
FROM gestion_reserva.Cliente c
LEFT JOIN gestion_reserva.ClienteTelefono ct ON c.id_cliente = ct.id_cliente
LEFT JOIN gestion_reserva.TipoTelefono tt ON ct.id_tipo_telefono = tt.id_tipo_telefono
LEFT JOIN gestion_reserva.Reserva r ON c.ruc = r.ruc_cliente
LEFT JOIN gestion_reserva.EstadoReserva er ON r.id_estado_reserva = er.id_estado_reserva
LEFT JOIN gestion_reserva.Ruta rt ON r.id_ruta = rt.id_ruta
LEFT JOIN gestion_reserva.AtencionCliente ac ON c.id_cliente = ac.id_cliente
LEFT JOIN gestion_reserva.ReservaContenedor rc ON r.id_reserva = rc.id_reserva
GROUP BY c.id_cliente, c.ruc, c.razon_social, c.direccion, c.email
HAVING COUNT(DISTINCT r.id_reserva) > 0
ORDER BY ingresos_totales_generados DESC, total_reservas DESC
LIMIT 100;
```
