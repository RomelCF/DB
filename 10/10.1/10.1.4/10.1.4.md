- **Script para poblar datos masivos de prueba**

```sql
DO $$
DECLARE
    v_start_time TIMESTAMP := clock_timestamp();
    v_id_tipo_vehiculo UUID;
    v_id_estado_activo_operativo UUID;
    v_id_estado_vehiculo_disponible UUID;
    v_id_estado_operacion_en_curso UUID;
    v_id_estado_operacion_completada UUID;
    v_conductores UUID[];
    v_usuarios UUID[];
    v_total_activos INT := 100;
    v_total_rutas INT := 50;
    v_total_operaciones INT := 300;
    v_total_operaciones_terrestres INT := 200;
BEGIN
    RAISE NOTICE '========================================';
    RAISE NOTICE 'CARGA MASIVA - OPERACIONES TERRESTRES';
    RAISE NOTICE '========================================';
    
    -- Obtener IDs de catálogos
    SELECT id_tipo_vehiculo INTO v_id_tipo_vehiculo 
    FROM operaciones_terrestres.TipoVehiculo WHERE nombre = 'Camión rígido' LIMIT 1;
    
    SELECT id_estado_activo INTO v_id_estado_activo_operativo 
    FROM operaciones_terrestres.EstadoActivo WHERE nombre = 'Operativo' LIMIT 1;
    
    SELECT id_estado_vehiculo INTO v_id_estado_vehiculo_disponible 
    FROM operaciones_terrestres.EstadoVehiculo WHERE nombre = 'Disponible' LIMIT 1;
    
    SELECT id_estado_operacion INTO v_id_estado_operacion_en_curso 
    FROM operaciones_terrestres.EstadoOperacion WHERE nombre = 'En curso' LIMIT 1;
    
    SELECT id_estado_operacion INTO v_id_estado_operacion_completada 
    FROM operaciones_terrestres.EstadoOperacion WHERE nombre = 'Completada' LIMIT 1;
    
    SELECT ARRAY_AGG(id_conductor) INTO v_conductores 
    FROM operaciones_terrestres.Conductor WHERE disponibilidad = TRUE;
    
    SELECT ARRAY_AGG(id_usuario) INTO v_usuarios 
    FROM operaciones_terrestres.Usuario;
    
    RAISE NOTICE 'Catálogos cargados: % conductores, % usuarios', 
        array_length(v_conductores, 1), array_length(v_usuarios, 1);
    
    -- =====================================
    -- 1. ACTIVOS Y VEHÍCULOS (100)
    -- =====================================
    RAISE NOTICE '1/5: Insertando % activos y vehículos...', v_total_activos;
    
    INSERT INTO operaciones_terrestres.Activo (
        id_activo, codigo, nombre, id_tipo_activo, id_estado_activo, ubicacion
    )
    SELECT 
        gen_random_uuid(),
        'ACT-T-' || LPAD(i::TEXT, 4, '0'),
        'Vehículo de Transporte ' || i,
        (SELECT id_tipo_activo FROM operaciones_terrestres.TipoActivo WHERE nombre = 'Vehículo' LIMIT 1),
        v_id_estado_activo_operativo,
        CASE (random() * 3)::INT
            WHEN 0 THEN 'Patio Principal'
            WHEN 1 THEN 'Terminal Norte'
            ELSE 'Terminal Sur'
        END
    FROM generate_series(1, v_total_activos) i;
    
    -- Insertar vehículos asociados
    INSERT INTO operaciones_terrestres.Vehiculo (
        id_vehiculo, id_activo, placa, capacidad_ton, id_tipo_vehiculo, id_estado_vehiculo
    )
    SELECT 
        gen_random_uuid(),
        a.id_activo,
        'PLA-' || LPAD((ROW_NUMBER() OVER ())::TEXT, 4, '0'),
        (10 + random() * 25)::DECIMAL(10,2),
        v_id_tipo_vehiculo,
        CASE WHEN random() < 0.85 THEN v_id_estado_vehiculo_disponible 
             ELSE (SELECT id_estado_vehiculo FROM operaciones_terrestres.EstadoVehiculo WHERE nombre = 'Mantenimiento' LIMIT 1) 
        END
    FROM operaciones_terrestres.Activo a
    WHERE a.codigo LIKE 'ACT-T-%';
    
    RAISE NOTICE '  ✓ Activos y vehículos completados (%.2f seg)', 
        EXTRACT(EPOCH FROM (clock_timestamp() - v_start_time));
    
    -- =====================================
    -- 2. RUTAS (50)
    -- =====================================
    RAISE NOTICE '2/5: Insertando % rutas...', v_total_rutas;
    
    INSERT INTO operaciones_terrestres.Ruta (
        id_ruta, codigo, origen, destino, duracion, tarifa
    )
    SELECT 
        gen_random_uuid(),
        'RUT-T-' || LPAD(i::TEXT, 4, '0'),
        CASE (random() * 5)::INT
            WHEN 0 THEN 'Callao'
            WHEN 1 THEN 'Lima Centro'
            WHEN 2 THEN 'Arequipa'
            WHEN 3 THEN 'Trujillo'
            ELSE 'Cusco'
        END,
        CASE (random() * 5)::INT
            WHEN 0 THEN 'Ica'
            WHEN 1 THEN 'Chiclayo'
            WHEN 2 THEN 'Piura'
            WHEN 3 THEN 'Huancayo'
            ELSE 'Tacna'
        END,
        (1 + random() * 8)::INT,
        (100 + random() * 2000)::DECIMAL(10,2)
    FROM generate_series(1, v_total_rutas) i;
    
    -- Insertar rutas terrestres
    INSERT INTO operaciones_terrestres.RutaTerrestre (id_ruta_terrestre, id_ruta)
    SELECT gen_random_uuid(), id_ruta
    FROM operaciones_terrestres.Ruta
    WHERE codigo LIKE 'RUT-T-%';
    
    RAISE NOTICE '  ✓ Rutas completadas (%.2f seg)', 
        EXTRACT(EPOCH FROM (clock_timestamp() - v_start_time));
    
    -- =====================================
    -- 3. OPERACIONES (300)
    -- =====================================
    RAISE NOTICE '3/5: Insertando % operaciones...', v_total_operaciones;
    
    INSERT INTO operaciones_terrestres.Operacion (
        id_operacion, codigo, fecha_inicio, fecha_fin, id_estado_operacion
    )
    SELECT 
        gen_random_uuid(),
        'OP-T-' || LPAD(i::TEXT, 6, '0'),
        CURRENT_TIMESTAMP - (random() * INTERVAL '90 days'),
        CASE 
            WHEN random() < 0.7 THEN CURRENT_TIMESTAMP - (random() * INTERVAL '30 days')
            ELSE NULL
        END,
        CASE 
            WHEN random() < 0.6 THEN v_id_estado_operacion_completada
            ELSE v_id_estado_operacion_en_curso
        END
    FROM generate_series(1, v_total_operaciones) i;
    
    RAISE NOTICE '  ✓ Operaciones completadas (%.2f seg)', 
        EXTRACT(EPOCH FROM (clock_timestamp() - v_start_time));
    
    -- =====================================
    -- 4. OPERACIONES TERRESTRES (200)
    -- =====================================
    RAISE NOTICE '4/5: Insertando % operaciones terrestres...', v_total_operaciones_terrestres;
    
    INSERT INTO operaciones_terrestres.OperacionTerrestre (
        id_operacion_terrestre, id_operacion, codigo, id_vehiculo, id_ruta_terrestre, id_conductor, costo_operacion_terrestre
    )
    SELECT 
        gen_random_uuid(),
        o.id_operacion,
        'OT-' || o.codigo,
        (SELECT id_vehiculo FROM operaciones_terrestres.Vehiculo ORDER BY random() LIMIT 1),
        (SELECT id_ruta_terrestre FROM operaciones_terrestres.RutaTerrestre ORDER BY random() LIMIT 1),
        v_conductores[1 + floor(random() * array_length(v_conductores, 1))],
        (500 + random() * 5000)::DECIMAL(10,2)
    FROM operaciones_terrestres.Operacion o
    WHERE o.codigo LIKE 'OP-T-%'
    LIMIT v_total_operaciones_terrestres;
    
    RAISE NOTICE '  ✓ Operaciones terrestres completadas (%.2f seg)', 
        EXTRACT(EPOCH FROM (clock_timestamp() - v_start_time));
    
    -- =====================================
    -- 5. CHECKLISTS E INCIDENCIAS
    -- =====================================
    RAISE NOTICE '5/5: Insertando checklists e incidencias...';
    
    -- Checklists de despacho
    INSERT INTO operaciones_terrestres.ChecklistDespacho (
        id_checklist, codigo, observaciones, combustible, frenos, id_operacion_terrestre
    )
    SELECT 
        gen_random_uuid(),
        'CHK-' || ot.codigo,
        CASE (random() * 4)::INT
            WHEN 0 THEN 'Vehículo en óptimas condiciones'
            WHEN 1 THEN 'Revisión completa realizada'
            WHEN 2 THEN 'Documentación verificada'
            ELSE 'Sistemas operativos'
        END,
        CASE (random() * 3)::INT
            WHEN 0 THEN 'Completo'
            WHEN 1 THEN '3/4 tanque'
            ELSE '1/2 tanque'
        END,
        'Aprobado',
        ot.id_operacion_terrestre
    FROM operaciones_terrestres.OperacionTerrestre ot
    WHERE ot.codigo LIKE 'OT-%'
    AND random() < 0.8;
    
    -- Incidencias
    INSERT INTO operaciones_terrestres.Incidencia (
        id_incidencia, codigo, id_tipo_incidencia, descripcion, grado_severidad, fecha_hora, id_estado_incidencia, id_operacion, id_usuario
    )
    SELECT 
        gen_random_uuid(),
        'INC-' || LPAD(i::TEXT, 6, '0'),
        (SELECT id_tipo_incidencia FROM operaciones_terrestres.TipoIncidencia ORDER BY random() LIMIT 1),
        'Incidencia reportada en operación ' || i,
        (1 + random() * 4)::INT,
        CURRENT_TIMESTAMP - (random() * INTERVAL '60 days'),
        (SELECT id_estado_incidencia FROM operaciones_terrestres.EstadoIncidencia ORDER BY random() LIMIT 1),
        (SELECT id_operacion FROM operaciones_terrestres.Operacion ORDER BY random() LIMIT 1),
        v_usuarios[1 + floor(random() * array_length(v_usuarios, 1))]
    FROM generate_series(1, 100) i;
    
    RAISE NOTICE '  ✓ Checklists e incidencias completadas (%.2f seg)', 
        EXTRACT(EPOCH FROM (clock_timestamp() - v_start_time));
    
    -- =====================================
    -- RESUMEN FINAL
    -- =====================================
    RAISE NOTICE '========================================';
    RAISE NOTICE 'CARGA COMPLETADA EN: %.2f segundos', 
        EXTRACT(EPOCH FROM (clock_timestamp() - v_start_time));
    RAISE NOTICE '========================================';
    RAISE NOTICE 'Datos insertados:';
    RAISE NOTICE '  - Activos y Vehículos: %', v_total_activos;
    RAISE NOTICE '  - Rutas: %', v_total_rutas;
    RAISE NOTICE '  - Operaciones: %', v_total_operaciones;
    RAISE NOTICE '  - Operaciones Terrestres: %', v_total_operaciones_terrestres;
    RAISE NOTICE '  - Checklists: %', (SELECT COUNT(*) FROM operaciones_terrestres.ChecklistDespacho);
    RAISE NOTICE '  - Incidencias: %', (SELECT COUNT(*) FROM operaciones_terrestres.Incidencia);
    RAISE NOTICE '========================================';
    
END $$;
```

## **1. CONSULTA 1: Dashboard de Operaciones Terrestres Activas**

**Descripción**: Consulta para el dashboard que muestra operaciones terrestres activas con información detallada de vehículos, conductores y rutas.

### **Script de análisis**

```sql
EXPLAIN (ANALYZE, BUFFERS, FORMAT TEXT)
SELECT 
    ot.codigo AS codigo_operacion,
    o.fecha_inicio,
    o.fecha_fin,
    eo.nombre AS estado_operacion,
    v.placa,
    a.nombre AS vehiculo,
    tv.nombre AS tipo_vehiculo,
    ev.nombre AS estado_vehiculo,
    CONCAT(emp.nombre, ' ', emp.apellido) AS conductor,
    c.licencia,
    c.categoria,
    r.origen,
    r.destino,
    r.duracion,
    r.tarifa,
    ot.costo_operacion_terrestre,
    cd.combustible,
    cd.frenos,
    cd.observaciones
FROM operaciones_terrestres.OperacionTerrestre ot
JOIN operaciones_terrestres.Operacion o ON ot.id_operacion = o.id_operacion
JOIN operaciones_terrestres.EstadoOperacion eo ON o.id_estado_operacion = eo.id_estado_operacion
JOIN operaciones_terrestres.Vehiculo v ON ot.id_vehiculo = v.id_vehiculo
JOIN operaciones_terrestres.Activo a ON v.id_activo = a.id_activo
JOIN operaciones_terrestres.TipoVehiculo tv ON v.id_tipo_vehiculo = tv.id_tipo_vehiculo
JOIN operaciones_terrestres.EstadoVehiculo ev ON v.id_estado_vehiculo = ev.id_estado_vehiculo
JOIN operaciones_terrestres.Conductor c ON ot.id_conductor = c.id_conductor
JOIN operaciones_terrestres.Empleado emp ON c.id_empleado = emp.id_empleado
JOIN operaciones_terrestres.RutaTerrestre rt ON ot.id_ruta_terrestre = rt.id_ruta_terrestre
JOIN operaciones_terrestres.Ruta r ON rt.id_ruta = r.id_ruta
LEFT JOIN operaciones_terrestres.ChecklistDespacho cd ON ot.id_operacion_terrestre = cd.id_operacion_terrestre
WHERE eo.nombre = 'En curso'
AND o.fecha_inicio >= CURRENT_DATE - INTERVAL '7 days'
ORDER BY o.fecha_inicio DESC, ot.codigo
LIMIT 50;
```

```sql
-- Índice para operaciones por estado y fecha
CREATE INDEX idx_operacion_estado_fecha

CREATE INDEX idx_op_terrestre_operacion

CREATE INDEX idx_vehiculo_estado

-- Índice para checklist por operación terrestre
CREATE INDEX idx_checklist_op_terrestre 
ON operaciones_terrestres.ChecklistDespacho (id_operacion_terrestre) 
INCLUDE (combustible, frenos, observaciones);
```

**Descripción**: Consulta para generar reportes de incidencias con filtros por tipo, severidad y período de tiempo.

### **Script de análisis**

```sql
EXPLAIN (ANALYZE, BUFFERS, FORMAT TEXT)
SELECT 
    i.codigo AS codigo_incidencia,
    ti.nombre AS tipo_incidencia,
    ei.nombre AS estado_incidencia,
    i.descripcion,
    i.grado_severidad,
    i.fecha_hora,
    o.codigo AS codigo_operacion,
    ot.codigo AS codigo_operacion_terrestre,
    CONCAT(emp.nombre, ' ', emp.apellido) AS conductor,
    v.placa AS vehiculo,
    r.origen,
    r.destino,
    CONCAT(u_rep.nombre, ' ', u_rep.apellido) AS usuario_reporta,
    ur.nombre AS rol_usuario
FROM operaciones_terrestres.Incidencia i
JOIN operaciones_terrestres.TipoIncidencia ti ON i.id_tipo_incidencia = ti.id_tipo_incidencia
JOIN operaciones_terrestres.EstadoIncidencia ei ON i.id_estado_incidencia = ei.id_estado_incidencia
JOIN operaciones_terrestres.Operacion o ON i.id_operacion = o.id_operacion
JOIN operaciones_terrestres.OperacionTerrestre ot ON o.id_operacion = ot.id_operacion
JOIN operaciones_terrestres.Vehiculo v ON ot.id_vehiculo = v.id_vehiculo
JOIN operaciones_terrestres.Conductor c ON ot.id_conductor = c.id_conductor
JOIN operaciones_terrestres.Empleado emp ON c.id_empleado = emp.id_empleado
JOIN operaciones_terrestres.RutaTerrestre rt ON ot.id_ruta_terrestre = rt.id_ruta_terrestre
JOIN operaciones_terrestres.Ruta r ON rt.id_ruta = r.id_ruta
JOIN operaciones_terrestres.Usuario u ON i.id_usuario = u.id_usuario
JOIN operaciones_terrestres.Empleado u_rep ON u.id_empleado = u_rep.id_empleado
JOIN operaciones_terrestres.RolUsuario ur ON u.id_rol_usuario = ur.id_rol_usuario
WHERE i.fecha_hora >= CURRENT_DATE - INTERVAL '30 days'
AND ti.nombre IN ('Operacional', 'Seguridad')
AND i.grado_severidad >= 3
AND ei.nombre != 'Cerrada'
ORDER BY i.grado_severidad DESC, i.fecha_hora DESC
LIMIT 100;
```

## **1. CONSULTA 1: Dashboard de Planificación de Mantenimiento Preventivo**

**Descripción**: Consulta compleja para el dashboard que analiza activos que requieren mantenimiento urgente con múltiples JOINs y subconsultas.

### **Script de análisis**

```sql
EXPLAIN (ANALYZE, BUFFERS, FORMAT TEXT)
SELECT 
    a.codigo AS id_activo,
    a.nombre AS nombre_activo,
    ta.nombre AS tipo_activo,
    ea.nombre AS estado_activo,
    a.ubicacion,
    pm.fecha_inicio AS proximo_mantenimiento,
    CASE 
        WHEN pm.fecha_inicio <= CURRENT_DATE THEN 'Vencido'
        WHEN pm.fecha_inicio <= CURRENT_DATE + INTERVAL '7 days' THEN '7 Días Restantes'
        WHEN pm.fecha_inicio <= CURRENT_DATE + INTERVAL '30 days' THEN '27 Días Restantes'
        ELSE 'Programado'
    END AS estado_mantenimiento,
    (pm.fecha_inicio - CURRENT_DATE) AS dias_restantes
FROM mantenimiento_logistico.Activo a
JOIN mantenimiento_logistico.TipoActivo ta ON a.id_tipo_activo = ta.id_tipo_activo
JOIN mantenimiento_logistico.EstadoActivo ea ON a.id_estado_activo = ea.id_estado_activo
LEFT JOIN mantenimiento_logistico.PlanMantenimiento pm ON a.id_activo = pm.id_activo
WHERE ea.nombre IN ('En mantenimiento', 'Fuera de servicio')
   OR (pm.fecha_inicio IS NOT NULL AND pm.fecha_inicio <= CURRENT_DATE + INTERVAL '30 days')
ORDER BY 
    CASE 
        WHEN pm.fecha_inicio <= CURRENT_DATE THEN 1
        WHEN pm.fecha_inicio <= CURRENT_DATE + INTERVAL '7 days' THEN 2
        WHEN pm.fecha_inicio <= CURRENT_DATE + INTERVAL '30 days' THEN 3
        ELSE 4
    END,
    pm.fecha_inicio ASC NULLS LAST
LIMIT 50;
```

```sql
FROM mantenimiento_logistico.OrdenMantenimiento om
JOIN mantenimiento_logistico.OperacionMantenimiento opm ON om.id_operacion_mantenimiento = opm.id_operacion_mantenimiento
JOIN mantenimiento_logistico.SolicitudMantenimiento sm ON opm.id_solicitud_mantenimiento = sm.id_solicitud_mantenimiento
JOIN mantenimiento_logistico.Activo a ON sm.id_activo = a.id_activo
JOIN mantenimiento_logistico.Prioridad p ON sm.id_prioridad = p.id_prioridad
JOIN mantenimiento_logistico.EstadoOrden eo ON om.id_estado_orden = eo.id_estado_orden
LEFT JOIN mantenimiento_logistico.OperacionMantenimientoTecnico omt ON opm.id_operacion_mantenimiento = omt.id_operacion_mantenimiento
WHERE omt.id_tecnico IS NULL
AND p.nombre IN ('Crítica', 'Alta')
AND eo.nombre = 'Abierta'
AND om.fecha_generada >= CURRENT_DATE - INTERVAL '7 days'
ORDER BY 
    CASE p.nombre 
        WHEN 'Crítica' THEN 1
        WHEN 'Alta' THEN 2
        ELSE 3
    END,
    om.fecha_generada DESC
LIMIT 20;
```

### **Creación de Índices**

```sql
-- Índice para órdenes por fecha y estado
CREATE INDEX idx_orden_fecha_estado 
ON mantenimiento_logistico.OrdenMantenimiento (fecha_generada DESC, id_estado_orden) 
INCLUDE (codigo, id_operacion_mantenimiento);

-- Índice para solicitudes por prioridad
CREATE INDEX idx_solicitud_prioridad 
ON mantenimiento_logistico.SolicitudMantenimiento (id_prioridad) 
INCLUDE (id_activo, descripcion_problema, id_estado_solicitud);

-- Índice para asignación de técnicos
CREATE INDEX idx_op_mantenimiento_tecnico_completo 
ON mantenimiento_logistico.OperacionMantenimientoTecnico (id_operacion_mantenimiento, id_tecnico);
```
