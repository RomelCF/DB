```sql
-- =====================================
-- 1. INSERTAR PUERTOS BASE (2,000 puertos)
-- =====================================
DO $$
DECLARE
    v_start_time TIMESTAMP := clock_timestamp();
    v_paises TEXT[] := ARRAY['Perú', 'Chile', 'Ecuador', 'Colombia', 'México', 'Brasil', 'Argentina', 
                              'Uruguay', 'Venezuela', 'Panamá', 'Costa Rica', 'Estados Unidos', 
                              'Canadá', 'España', 'Portugal', 'Italia', 'Francia', 'Alemania', 
                              'Países Bajos', 'Bélgica', 'Reino Unido', 'China', 'Japón', 
                              'Corea del Sur', 'Singapur', 'India', 'Australia', 'Nueva Zelanda'];
    v_ciudades TEXT[] := ARRAY['Puerto', 'Bahía', 'Costa', 'Mar', 'Océano', 'Playa', 'Golfo', 
                                'Estrecho', 'Península', 'Cabo', 'Isla', 'Ensenada', 'Rada'];
BEGIN
    RAISE NOTICE '========================================';
    RAISE NOTICE 'INSERTANDO DATOS BASE - ALTA CAPACIDAD';
    RAISE NOTICE '========================================';
    RAISE NOTICE '1/3: Insertando 2,000 puertos...';
    
    -- Insertar 2,000 puertos
    INSERT INTO gestion_maritima.Puerto (id_puerto, codigo, nombre, pais, direccion)
    SELECT 
        gen_random_uuid(),
        CASE 
            WHEN i <= 26 THEN CHR(64 + i) || CHR(65 + (i % 26)) || '-' || LPAD((i % 100)::TEXT, 3, '0')
            ELSE CHR(65 + (i % 26)) || CHR(65 + ((i/26) % 26)) || '-' || LPAD(i::TEXT, 4, '0')
        END,
        v_ciudades[1 + (i % array_length(v_ciudades, 1))] || ' ' || 
        CASE 
            WHEN i % 5 = 0 THEN 'del Norte'
            WHEN i % 5 = 1 THEN 'del Sur'
            WHEN i % 5 = 2 THEN 'Central'
            WHEN i % 5 = 3 THEN 'Oriental'
            ELSE 'Occidental'
        END || ' ' || LPAD(i::TEXT, 4, '0'),
        v_paises[1 + (i % array_length(v_paises, 1))],
        'Av. Marítima ' || i || ', Zona Portuaria, CP ' || (10000 + i)::TEXT
    FROM generate_series(1, 2000) i;
    
    RAISE NOTICE '  ✓ Puertos completados (%.2f seg)', 
        EXTRACT(EPOCH FROM (clock_timestamp() - v_start_time));
    
    -- =====================================
    -- 2. INSERTAR MUELLES (2,000 muelles = 1 por puerto)
    -- =====================================
    RAISE NOTICE '2/3: Insertando 2,000 muelles...';
    
    INSERT INTO gestion_maritima.Muelle (id_muelle, codigo, ubicacion, capacidad, disponibilidad, id_puerto)
    SELECT 
        gen_random_uuid(),
        p.codigo || '-M' || LPAD((ROW_NUMBER() OVER (PARTITION BY p.id_puerto))::TEXT, 2, '0'),
        CASE (ROW_NUMBER() OVER (PARTITION BY p.id_puerto) % 8)
            WHEN 0 THEN 'Muelle Norte'
            WHEN 1 THEN 'Muelle Sur'
            WHEN 2 THEN 'Muelle Este'
            WHEN 3 THEN 'Muelle Oeste'
            WHEN 4 THEN 'Muelle Multipropósito'
            WHEN 5 THEN 'Muelle de Contenedores'
            WHEN 6 THEN 'Muelle de Carga General'
            ELSE 'Muelle Industrial'
        END,
        (2 + (random() * 8)::INT), -- Capacidad entre 2 y 10
        CASE WHEN random() < 0.85 THEN TRUE ELSE FALSE END, -- 85% disponible
        p.id_puerto
    FROM gestion_maritima.Puerto p;
    
    RAISE NOTICE '  ✓ Muelles completados (%.2f seg)', 
        EXTRACT(EPOCH FROM (clock_timestamp() - v_start_time));
    
END $$;

-- =====================================
-- 3. INSERTAR BUQUES BASE (2,000 buques)
-- =====================================
DO $$
DECLARE
    v_start_time TIMESTAMP := clock_timestamp();
    v_id_estado_disponible UUID;
    v_id_estado_operacion UUID;
    v_id_estado_mantenimiento UUID;
    v_id_estado_fuera_servicio UUID;
    v_prefijos TEXT[] := ARRAY['MV', 'MS', 'SS'];
    v_nombres TEXT[] := ARRAY['Pacific', 'Atlantic', 'Caribbean', 'Mediterranean', 'Baltic', 
                               'Explorer', 'Navigator', 'Pioneer', 'Voyager', 'Discovery',
                               'Enterprise', 'Triumph', 'Victory', 'Horizon', 'Spirit',
                               'Liberty', 'Freedom', 'Independence', 'Sovereign', 'Majesty',
                               'Celestial', 'Oceanic', 'Maritime', 'Marine', 'Nautical',
                               'Admiral', 'Captain', 'Commander', 'Commodore', 'Admiral'];
BEGIN
    RAISE NOTICE '3/3: Insertando 2,000 buques...';
    
    -- Obtener IDs de estados
    SELECT id_estado_embarcacion INTO v_id_estado_disponible 
    FROM gestion_maritima.EstadoEmbarcacion WHERE nombre = 'Disponible' LIMIT 1;
    
    SELECT id_estado_embarcacion INTO v_id_estado_operacion 
    FROM gestion_maritima.EstadoEmbarcacion WHERE nombre = 'En operación' LIMIT 1;
    
    SELECT id_estado_embarcacion INTO v_id_estado_mantenimiento 
    FROM gestion_maritima.EstadoEmbarcacion WHERE nombre = 'En mantenimiento' LIMIT 1;
    
    SELECT id_estado_embarcacion INTO v_id_estado_fuera_servicio 
    FROM gestion_maritima.EstadoEmbarcacion WHERE nombre = 'Fuera de servicio' LIMIT 1;
    
    -- Insertar 2,000 buques
    INSERT INTO gestion_maritima.Buque (
        id_buque, matricula, nombre, capacidad, 
        id_estado_embarcacion, peso, ubicacion_actual
    )
    SELECT 
        gen_random_uuid(),
        v_prefijos[1 + (i % array_length(v_prefijos, 1))] || '-' || LPAD(i::TEXT, 6, '0'),
        v_prefijos[1 + (i % array_length(v_prefijos, 1))] || ' ' || 
        v_nombres[1 + (i % array_length(v_nombres, 1))] || ' ' || 
        CASE 
            WHEN i <= 100 THEN i::TEXT
            ELSE ((i / 100) * 100)::TEXT
        END,
        CASE 
            WHEN i % 4 = 0 THEN 500 + (random() * 300)::INT    -- Buques pequeños: 500-800
            WHEN i % 4 = 1 THEN 800 + (random() * 400)::INT    -- Buques medianos: 800-1200
            WHEN i % 4 = 2 THEN 1200 + (random() * 800)::INT   -- Buques grandes: 1200-2000
            ELSE 2000 + (random() * 2000)::INT                 -- Buques muy grandes: 2000-4000
        END,
        CASE 
            WHEN random() < 0.50 THEN v_id_estado_disponible
            WHEN random() < 0.80 THEN v_id_estado_operacion
            WHEN random() < 0.95 THEN v_id_estado_mantenimiento
            ELSE v_id_estado_fuera_servicio
        END,
        (20000 + random() * 80000)::DECIMAL(15,2), -- Peso entre 20,000 y 100,000 toneladas
        CASE (random() * 15)::INT
            WHEN 0 THEN 'En Puerto'
            WHEN 1 THEN 'En Tránsito'
            WHEN 2 THEN 'En Alta Mar'
            WHEN 3 THEN 'Zona de Anclaje'
            WHEN 4 THEN 'En Reparación'
            WHEN 5 THEN 'En Dique Seco'
            WHEN 6 THEN 'Zona de Espera'
            WHEN 7 THEN 'Canal de Navegación'
            WHEN 8 THEN 'Área de Maniobras'
            WHEN 9 THEN 'Fondeadero'
            WHEN 10 THEN 'Rada Interior'
            WHEN 11 THEN 'Rada Exterior'
            WHEN 12 THEN 'En Cuarentena'
            WHEN 13 THEN 'Terminal Petrolero'
            ELSE 'Terminal de Contenedores'
        END
    FROM generate_series(1, 2000) i;
    
    RAISE NOTICE '  ✓ Buques completados (%.2f seg)', 
        EXTRACT(EPOCH FROM (clock_timestamp() - v_start_time));
    
END $$;


-- ============================================================
-- CARGA MASIVA DE DATOS
-- Schema: gestion_maritima
-- Tablas: Operacion, OperacionMaritima, Ruta, RutaMaritima
-- ============================================================

DO $$
DECLARE
    v_start_time TIMESTAMP := clock_timestamp();
    v_id_estado_en_progreso UUID;
    v_id_estado_completada UUID;
    v_id_estado_planificacion UUID;
    v_id_estatus_navegando UUID;
    v_id_estatus_puerto UUID;
    v_id_estatus_atracado UUID;
    v_buques UUID[];
    v_puertos UUID[];
    v_muelles UUID[];
    v_total_operaciones INT := 10000;
    v_total_rutas INT := 2000;
BEGIN
    RAISE NOTICE '========================================';
    RAISE NOTICE 'CARGA MASIVA - INICIO';
    RAISE NOTICE '========================================';
    
    -- =====================================
    -- OBTENER IDs DE CATÁLOGOS
    -- =====================================
    SELECT id_estado_operacion INTO v_id_estado_en_progreso 
    FROM gestion_maritima.EstadoOperacion WHERE nombre = 'En Progreso' LIMIT 1;
    
    SELECT id_estado_operacion INTO v_id_estado_completada 
    FROM gestion_maritima.EstadoOperacion WHERE nombre = 'Completada' LIMIT 1;
    
    SELECT id_estado_operacion INTO v_id_estado_planificacion 
    FROM gestion_maritima.EstadoOperacion WHERE nombre = 'En Planificación' LIMIT 1;
    
    SELECT id_estatus_navegacion INTO v_id_estatus_navegando 
    FROM gestion_maritima.EstatusNavegacion WHERE nombre = 'Navegando' LIMIT 1;
    
    SELECT id_estatus_navegacion INTO v_id_estatus_puerto 
    FROM gestion_maritima.EstatusNavegacion WHERE nombre = 'En Puerto' LIMIT 1;
    
    SELECT id_estatus_navegacion INTO v_id_estatus_atracado 
    FROM gestion_maritima.EstatusNavegacion WHERE nombre = 'Atracado' LIMIT 1;
    
    -- Obtener arrays de IDs
    SELECT ARRAY_AGG(id_buque) INTO v_buques FROM gestion_maritima.Buque;
    SELECT ARRAY_AGG(id_puerto) INTO v_puertos FROM gestion_maritima.Puerto;
    SELECT ARRAY_AGG(id_muelle) INTO v_muelles FROM gestion_maritima.Muelle;
    
    RAISE NOTICE 'Catálogos cargados: % buques, % puertos, % muelles', 
        array_length(v_buques, 1), array_length(v_puertos, 1), array_length(v_muelles, 1);
    
    -- =====================================
    -- 1. OPERACIONES (10,000)
    -- =====================================
    RAISE NOTICE '1/4: Insertando % operaciones...', v_total_operaciones;
    
    INSERT INTO gestion_maritima.Operacion (id_operacion, codigo, fecha_inicio, fecha_fin, id_estado_operacion)
    SELECT 
        gen_random_uuid(),
        'OP-2024-' || LPAD(i::TEXT, 6, '0'),
        fecha_inicio_temp,
        CASE 
            WHEN random() < 0.6 THEN fecha_inicio_temp + (random() * INTERVAL '30 days')
            ELSE NULL
        END,
        CASE 
            WHEN random() < 0.5 THEN v_id_estado_completada
            WHEN random() < 0.8 THEN v_id_estado_en_progreso
            ELSE v_id_estado_planificacion
        END
    FROM (
        SELECT 
            i,
            CURRENT_TIMESTAMP - (random() * INTERVAL '365 days') AS fecha_inicio_temp
        FROM generate_series(1, v_total_operaciones) i
    ) subquery;
    
    RAISE NOTICE '  ✓ Operaciones completadas (%.2f seg)', 
        EXTRACT(EPOCH FROM (clock_timestamp() - v_start_time));
    
    -- =====================================
    -- 2. OPERACIONES MARÍTIMAS (5,000 = 1 por operación)
    -- =====================================
    RAISE NOTICE '2/4: Insertando % operaciones marítimas...', v_total_operaciones;
    
    INSERT INTO gestion_maritima.OperacionMaritima (
        id_operacion_maritima, id_operacion, codigo, cantidad_contenedores, 
        id_estatus_navegacion, porcentaje_trayecto, id_buque
    )
    SELECT 
        gen_random_uuid(),
        o.id_operacion,
        'OM-2024-' || LPAD(ROW_NUMBER() OVER ()::TEXT, 6, '0'),
        (50 + random() * 450)::INT, -- Entre 50 y 500 contenedores
        CASE 
            WHEN random() < 0.4 THEN v_id_estatus_navegando
            WHEN random() < 0.7 THEN v_id_estatus_puerto
            ELSE v_id_estatus_atracado
        END,
        CASE 
            WHEN random() < 0.3 THEN (random() * 100)::DECIMAL(5,2)
            WHEN random() < 0.6 THEN 100.00
            ELSE (50 + random() * 50)::DECIMAL(5,2)
        END,
        v_buques[1 + floor(random() * array_length(v_buques, 1))]
    FROM gestion_maritima.Operacion o
    WHERE o.codigo LIKE 'OP-2024-%';
    
    RAISE NOTICE '  ✓ Operaciones marítimas completadas (%.2f seg)', 
        EXTRACT(EPOCH FROM (clock_timestamp() - v_start_time));
    
    -- =====================================
    -- 3. RUTAS (200)
    -- =====================================
    RAISE NOTICE '3/4: Insertando % rutas...', v_total_rutas;
    
    INSERT INTO gestion_maritima.Ruta (id_ruta, codigo, origen, destino, duracion, tarifa)
    SELECT 
        gen_random_uuid(),
        'RUT-' || LPAD(i::TEXT, 4, '0'),
        CASE (random() * 10)::INT
            WHEN 0 THEN 'Puerto del Callao'
            WHEN 1 THEN 'Puerto de Paita'
            WHEN 2 THEN 'Puerto de Matarani'
            WHEN 3 THEN 'Puerto de Salaverry'
            WHEN 4 THEN 'Puerto de Ilo'
            WHEN 5 THEN 'Puerto de Pisco'
            WHEN 6 THEN 'Puerto de Chimbote'
            WHEN 7 THEN 'Puerto de Iquitos'
            WHEN 8 THEN 'Puerto de Talara'
            ELSE 'Puerto San Juan de Marcona'
        END,
        CASE (random() * 10)::INT
            WHEN 0 THEN 'Puerto de Guayaquil'
            WHEN 1 THEN 'Puerto de Buenaventura'
            WHEN 2 THEN 'Puerto de Valparaíso'
            WHEN 3 THEN 'Puerto de San Antonio'
            WHEN 4 THEN 'Puerto de Manzanillo'
            WHEN 5 THEN 'Puerto de Los Ángeles'
            WHEN 6 THEN 'Puerto de Santos'
            WHEN 7 THEN 'Puerto de Montevideo'
            WHEN 8 THEN 'Puerto de Buenos Aires'
            ELSE 'Puerto de Cartagena'
        END,
        (24 + random() * 240)::INT, -- Duración entre 24 y 264 horas
        (5000 + random() * 45000)::DECIMAL(10,2) -- Tarifa entre 5,000 y 50,000
    FROM generate_series(1, v_total_rutas) i;
    
    RAISE NOTICE '  ✓ Rutas completadas (%.2f seg)', 
        EXTRACT(EPOCH FROM (clock_timestamp() - v_start_time));
    
    -- =====================================
    -- 4. RUTAS MARÍTIMAS (200 = 1 por ruta)
    -- =====================================
    RAISE NOTICE '4/4: Insertando % rutas marítimas...', v_total_rutas;
    
    -- Generar rutas marítimas asegurando puertos diferentes
    INSERT INTO gestion_maritima.RutaMaritima (
        id_ruta_maritima, id_ruta, codigo, distancia, 
        id_puerto_origen, id_puerto_destino
    )
    SELECT 
        gen_random_uuid(),
        r.id_ruta,
        'RM-' || LPAD(ROW_NUMBER() OVER ()::TEXT, 4, '0'),
        (500 + random() * 9500)::DECIMAL(10,2), -- Distancia entre 500 y 10,000 km
        puerto_origen.id_puerto,
        puerto_destino.id_puerto
    FROM gestion_maritima.Ruta r
    CROSS JOIN LATERAL (
        SELECT id_puerto 
        FROM gestion_maritima.Puerto 
        ORDER BY random() 
        LIMIT 1
    ) puerto_origen
    CROSS JOIN LATERAL (
        SELECT id_puerto 
        FROM gestion_maritima.Puerto 
        WHERE id_puerto != puerto_origen.id_puerto
        ORDER BY random() 
        LIMIT 1
    ) puerto_destino
    WHERE r.codigo LIKE 'RUT-%';
    
    RAISE NOTICE '  ✓ Rutas marítimas completadas (%.2f seg)', 
        EXTRACT(EPOCH FROM (clock_timestamp() - v_start_time));
    
    -- =====================================
    -- RESUMEN FINAL
    -- =====================================
    RAISE NOTICE '========================================';
    RAISE NOTICE 'CARGA COMPLETADA EN: %.2f segundos', 
        EXTRACT(EPOCH FROM (clock_timestamp() - v_start_time));
    RAISE NOTICE '========================================';
    RAISE NOTICE 'Datos insertados:';
    RAISE NOTICE '  - Operaciones: %', v_total_operaciones;
    RAISE NOTICE '  - Operaciones Marítimas: %', v_total_operaciones;
    RAISE NOTICE '  - Rutas: %', v_total_rutas;
    RAISE NOTICE '  - Rutas Marítimas: %', v_total_rutas;
    RAISE NOTICE '========================================';
    RAISE NOTICE 'Total de registros: %', (v_total_operaciones * 2 + v_total_rutas * 2);
    RAISE NOTICE '========================================';
    
END $$;
```

## **1. CONSULTA 1: Análisis Completo de Operaciones con Agregaciones Pesadas**

**Descripción**: Consulta compleja que cruza múltiples tablas con agregaciones y subconsultas. Sin índices, escanea millones de registros.

### **Script de análisis**

```sql
EXPLAIN (ANALYZE, BUFFERS, FORMAT TEXT)
SELECT 
    b.matricula,
    b.nombre AS buque,
    b.capacidad,
    ee.nombre AS estado_buque,
    COUNT(DISTINCT om.id_operacion_maritima) AS total_operaciones,
    COUNT(DISTINCT CASE 
        WHEN o.fecha_fin IS NULL THEN om.id_operacion_maritima 
    END) AS operaciones_activas,
    AVG(om.cantidad_contenedores)::DECIMAL(10,2) AS promedio_contenedores,
    SUM(om.cantidad_contenedores) AS total_contenedores_transportados,
    AVG(om.porcentaje_trayecto)::DECIMAL(5,2) AS promedio_avance,
    STRING_AGG(DISTINCT en.nombre, ', ') AS estados_navegacion,
    COUNT(DISTINCT CASE 
        WHEN o.fecha_inicio >= CURRENT_DATE - INTERVAL '30 days' 
        THEN om.id_operacion_maritima 
    END) AS operaciones_ultimo_mes,
    (
        SELECT COUNT(*)
        FROM gestion_maritima.OperacionMaritima om2
        JOIN gestion_maritima.Operacion o2 ON om2.id_operacion = o2.id_operacion
        WHERE om2.id_buque = b.id_buque
        AND o2.fecha_fin IS NOT NULL
        AND o2.fecha_fin >= o2.fecha_inicio
    ) AS operaciones_completadas,
    (
        SELECT AVG(rm.distancia)::DECIMAL(10,2)
        FROM gestion_maritima.OperacionMaritima om3
        JOIN gestion_maritima.OperacionRutaMaritima orm ON om3.id_operacion_maritima = orm.id_operacion_maritima
        JOIN gestion_maritima.RutaMaritima rm ON orm.id_ruta_maritima = rm.id_ruta_maritima
        JOIN gestion_maritima.Ruta r ON rm.id_ruta = r.id_ruta
        WHERE om3.id_buque = b.id_buque
    ) AS distancia_promedio,
    (
        SELECT STRING_AGG(DISTINCT p.nombre, ' -> ')
        FROM gestion_maritima.OperacionMaritima om4
        JOIN gestion_maritima.OperacionRutaMaritima orm2 ON om4.id_operacion_maritima = orm2.id_operacion_maritima
        JOIN gestion_maritima.RutaMaritima rm2 ON orm2.id_ruta_maritima = rm2.id_ruta_maritima
        JOIN gestion_maritima.Puerto p ON p.id_puerto IN (rm2.id_puerto_origen, rm2.id_puerto_destino)
        WHERE om4.id_buque = b.id_buque
        LIMIT 5
    ) AS puertos_visitados
FROM gestion_maritima.Buque b
JOIN gestion_maritima.EstadoEmbarcacion ee ON b.id_estado_embarcacion = ee.id_estado_embarcacion
LEFT JOIN gestion_maritima.OperacionMaritima om ON b.id_buque = om.id_buque
LEFT JOIN gestion_maritima.Operacion o ON om.id_operacion = o.id_operacion
LEFT JOIN gestion_maritima.EstatusNavegacion en ON om.id_estatus_navegacion = en.id_estatus_navegacion
WHERE b.capacidad >= 800
GROUP BY b.id_buque, b.matricula, b.nombre, b.capacidad, ee.nombre
HAVING COUNT(DISTINCT om.id_operacion_maritima) > 0
ORDER BY total_contenedores_transportados DESC, total_operaciones DESC
LIMIT 100;
```

## **2. CONSULTA 2: Análisis de Puertos con Estadísticas Complejas**

**Descripción**: Consulta masiva que analiza todos los puertos con múltiples subconsultas correlacionadas y agregaciones pesadas.

### **Script de análisis**

```sql
EXPLAIN (ANALYZE, BUFFERS, FORMAT TEXT)
SELECT 
    p.codigo AS codigo_puerto,
    p.nombre AS puerto,
    p.pais,
    p.direccion,
    (
        SELECT COUNT(*)
        FROM gestion_maritima.Muelle m
        WHERE m.id_puerto = p.id_puerto
    ) AS total_muelles,
    (
        SELECT COUNT(*)
        FROM gestion_maritima.Muelle m
        WHERE m.id_puerto = p.id_puerto AND m.disponibilidad = TRUE
    ) AS muelles_disponibles,
    (
        SELECT AVG(m.capacidad)::DECIMAL(10,2)
        FROM gestion_maritima.Muelle m
        WHERE m.id_puerto = p.id_puerto
    ) AS capacidad_promedio_muelles,
    (
        SELECT COUNT(DISTINCT op.id_operacion_portuaria)
        FROM gestion_maritima.Muelle m
        JOIN gestion_maritima.OperacionPortuaria op ON m.id_muelle = op.id_muelle
        WHERE m.id_puerto = p.id_puerto
    ) AS total_operaciones_portuarias,
    (
        SELECT COUNT(DISTINCT op.id_operacion_portuaria)
        FROM gestion_maritima.Muelle m
        JOIN gestion_maritima.OperacionPortuaria op ON m.id_muelle = op.id_muelle
        JOIN gestion_maritima.Operacion o ON op.id_operacion = o.id_operacion
        JOIN gestion_maritima.EstadoOperacion eo ON o.id_estado_operacion = eo.id_estado_operacion
        WHERE m.id_puerto = p.id_puerto
        AND eo.nombre = 'En Progreso'
    ) AS operaciones_en_progreso,
    (
        SELECT COUNT(DISTINCT b.id_buque)
        FROM gestion_maritima.Muelle m
        JOIN gestion_maritima.OperacionPortuaria op ON m.id_muelle = op.id_muelle
        JOIN gestion_maritima.Buque b ON op.id_buque = b.id_buque
        WHERE m.id_puerto = p.id_puerto
    ) AS buques_diferentes_atendidos,
    (
        SELECT COUNT(DISTINCT rm.id_ruta_maritima)
        FROM gestion_maritima.RutaMaritima rm
        WHERE rm.id_puerto_origen = p.id_puerto OR rm.id_puerto_destino = p.id_puerto
    ) AS rutas_maritimas_conectadas,
    (
        SELECT COUNT(*)
        FROM gestion_maritima.RutaMaritima rm
        WHERE rm.id_puerto_origen = p.id_puerto
    ) AS rutas_como_origen,
    (
        SELECT COUNT(*)
        FROM gestion_maritima.RutaMaritima rm
        WHERE rm.id_puerto_destino = p.id_puerto
    ) AS rutas_como_destino,
    (
        SELECT STRING_AGG(DISTINCT top.nombre, ', ')
        FROM gestion_maritima.Muelle m
        JOIN gestion_maritima.OperacionPortuaria op ON m.id_muelle = op.id_muelle
        JOIN gestion_maritima.TipoOperacionPortuaria top ON op.id_tipo_operacion_portuaria = top.id_tipo_operacion_portuaria
        WHERE m.id_puerto = p.id_puerto
    ) AS tipos_operaciones_realizadas,
    (
        SELECT AVG(EXTRACT(EPOCH FROM (o.fecha_fin - o.fecha_inicio))/3600)::DECIMAL(10,2)
        FROM gestion_maritima.Muelle m
        JOIN gestion_maritima.OperacionPortuaria op ON m.id_muelle = op.id_muelle
        JOIN gestion_maritima.Operacion o ON op.id_operacion = o.id_operacion
        WHERE m.id_puerto = p.id_puerto
        AND o.fecha_fin IS NOT NULL
    ) AS duracion_promedio_operaciones_horas
FROM gestion_maritima.Puerto p
ORDER BY p.codigo
LIMIT 500;
```
