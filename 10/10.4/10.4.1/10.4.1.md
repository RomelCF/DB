### Creación de vistas materializadas

### P05 — Dashboard de Reservas

#### Vista: `vm_dashboard_reservas`

**Descripción:** Métricas generales del sistema de reservas con KPIs de operaciones, clientes, buques y contenedores.

```sql
CREATE MATERIALIZED VIEW gestion_reserva.vm_dashboard_reservas AS
SELECT 
    -- Métricas de reservas
    COUNT(DISTINCT r.id_reserva) AS total_reservas,
    COUNT(DISTINCT CASE WHEN er.nombre = 'Confirmada' THEN r.id_reserva END) AS reservas_confirmadas,
    COUNT(DISTINCT CASE WHEN er.nombre = 'Activa' THEN r.id_reserva END) AS reservas_activas,
    COUNT(DISTINCT CASE WHEN er.nombre = 'Finalizada' THEN r.id_reserva END) AS reservas_finalizadas,
    COUNT(DISTINCT CASE WHEN er.nombre = 'Cancelada' THEN r.id_reserva END) AS reservas_canceladas,
    
    -- Métricas de clientes
    COUNT(DISTINCT c.id_cliente) AS total_clientes,
    COUNT(DISTINCT r.ruc_cliente) AS clientes_con_reservas,
    
    -- Métricas de agentes
    COUNT(DISTINCT ar.id_agente_reservas) AS total_agentes_activos,
    
    -- Métricas de buques
    COUNT(DISTINCT b.id_buque) AS total_buques,
    COUNT(DISTINCT CASE WHEN eb.nombre = 'Disponible' THEN b.id_buque END) AS buques_disponibles,
    COUNT(DISTINCT CASE WHEN eb.nombre = 'En operación' THEN b.id_buque END) AS buques_en_operacion,
    COUNT(DISTINCT CASE WHEN eb.nombre = 'Mantenimiento' THEN b.id_buque END) AS buques_en_mantenimiento,
    
    -- Métricas de contenedores
    COUNT(DISTINCT cont.id_contenedor) AS total_contenedores,
    COUNT(DISTINCT CASE WHEN ec.nombre = 'Disponible' THEN cont.id_contenedor END) AS contenedores_disponibles,
    COUNT(DISTINCT CASE WHEN ec.nombre = 'En tránsito' THEN cont.id_contenedor END) AS contenedores_en_transito,
    COUNT(DISTINCT CASE WHEN ec.nombre = 'Entregado' THEN cont.id_contenedor END) AS contenedores_entregados,
    
    -- Métricas de operaciones
    COUNT(DISTINCT o.id_operacion) AS total_operaciones,
    COUNT(DISTINCT CASE WHEN eo.nombre = 'En Progreso' THEN o.id_operacion END) AS operaciones_en_progreso,
    COUNT(DISTINCT CASE WHEN eo.nombre = 'Completada' THEN o.id_operacion END) AS operaciones_completadas,
    COUNT(DISTINCT om.id_operacion_maritima) AS operaciones_maritimas_total,
    COUNT(DISTINCT ot.id_operacion_terrestre) AS operaciones_terrestres_total,
    
    -- Métricas financieras
    COALESCE(SUM(r.pago_total), 0)::DECIMAL(12,2) AS ingresos_totales,
    COALESCE(AVG(r.pago_total), 0)::DECIMAL(10,2) AS ingreso_promedio_reserva,
    COALESCE(SUM(CASE WHEN er.nombre = 'Activa' THEN r.pago_total END), 0)::DECIMAL(12,2) AS ingresos_reservas_activas,
    
    -- Métricas de rutas
    COUNT(DISTINCT rt.id_ruta) AS total_rutas_disponibles,
    COALESCE(AVG(rt.duracion), 0)::DECIMAL(5,1) AS duracion_promedio_dias,
    COALESCE(AVG(rt.tarifa), 0)::DECIMAL(10,2) AS tarifa_promedio,
    
    -- Capacidad de buques
    COALESCE(SUM(b.capacidad), 0) AS capacidad_total_contenedores,
    COALESCE(AVG(b.capacidad), 0)::DECIMAL(10,1) AS capacidad_promedio_buque,
    
    -- Fecha de actualización
    CURRENT_TIMESTAMP AS fecha_actualizacion
    
FROM gestion_reserva.Reserva r
JOIN gestion_reserva.EstadoReserva er ON r.id_estado_reserva = er.id_estado_reserva

-- Clientes
LEFT JOIN gestion_reserva.Cliente c ON r.ruc_cliente = c.ruc

-- Agentes
LEFT JOIN gestion_reserva.AgenteReservas ar ON r.id_agente_reservas = ar.id_agente_reservas

-- Buques
LEFT JOIN gestion_reserva.Buque b ON r.matricula_buque = b.matricula
LEFT JOIN gestion_reserva.EstadoEmbarcacion eb ON b.id_estado_embarcacion = eb.id_estado_embarcacion

-- Rutas
LEFT JOIN gestion_reserva.Ruta rt ON r.id_ruta = rt.id_ruta

-- Contenedores (a través de ReservaContenedor)
LEFT JOIN gestion_reserva.ReservaContenedor rc ON r.id_reserva = rc.id_reserva
LEFT JOIN gestion_reserva.Contenedor cont ON rc.id_contenedor = cont.id_contenedor
LEFT JOIN gestion_reserva.EstadoContenedor ec ON cont.id_estado_contenedor = ec.id_estado_contenedor

-- Operaciones
CROSS JOIN gestion_reserva.Operacion o
LEFT JOIN gestion_reserva.EstadoOperacion eo ON o.id_estado_operacion = eo.id_estado_operacion
LEFT JOIN gestion_reserva.OperacionMaritima om ON o.id_operacion = om.id_operacion
LEFT JOIN gestion_reserva.OperacionTerrestre ot ON o.id_operacion = ot.id_operacion

WITH DATA;
```

### P02 — Listado de Reservas

#### Vista: `vm_listado_reservas_completo`

**Descripción:** Listado detallado de todas las reservas con información de cliente, buque, ruta y contenedores.

```sql
CREATE MATERIALIZED VIEW gestion_reserva.vm_listado_reservas_completo AS
SELECT 
    -- Información de la reserva
    r.id_reserva,
    r.codigo AS codigo_reserva,
    r.fecha_registro,
    er.nombre AS estado_reserva,
    r.pago_total,
    
    -- Información del cliente
    c.ruc AS ruc_cliente,
    c.razon_social,
    c.direccion AS direccion_cliente,
    c.email AS email_cliente,
    STRING_AGG(DISTINCT ct.telefono, ', ' ORDER BY ct.telefono) AS telefonos_cliente,
    
    -- Información del agente
    e.codigo AS codigo_agente,
    e.nombre || ' ' || e.apellido AS nombre_agente,
    ee.nombre AS especialidad_agente,
    
    -- Información del buque
    b.matricula,
    b.nombre AS nombre_buque,
    b.capacidad AS capacidad_buque,
    eb.nombre AS estado_buque,
    b.ubicacion_actual,
    
    -- Información de la ruta
    rt.codigo AS codigo_ruta,
    rt.origen,
    rt.destino,
    rt.duracion AS duracion_dias,
    rt.tarifa AS tarifa_ruta,
    
    -- Contenedores asociados
    COUNT(DISTINCT rc.id_contenedor) AS total_contenedores,
    SUM(rc.cantidad) AS cantidad_total_contenedores,
    STRING_AGG(DISTINCT cont.codigo, ', ' ORDER BY cont.codigo) AS codigos_contenedores,
    STRING_AGG(DISTINCT tc.nombre, ', ') AS tipos_contenedores,
    
    -- Operaciones asociadas
    COUNT(DISTINCT rom.id_operacion_maritima) AS operaciones_maritimas,
    COUNT(DISTINCT rot.id_operacion_terrestre) AS operaciones_terrestres,
    
    -- Días desde registro
    (CURRENT_DATE - r.fecha_registro) AS dias_desde_registro,
    
    -- Estado consolidado
    CASE 
        WHEN er.nombre = 'Activa' AND CURRENT_DATE - r.fecha_registro > 30 THEN 'Requiere atención'
        WHEN er.nombre = 'Confirmada' THEN 'En espera'
        WHEN er.nombre = 'Activa' THEN 'En curso'
        WHEN er.nombre = 'Finalizada' THEN 'Completada'
        ELSE er.nombre
    END AS estado_consolidado
    
FROM gestion_reserva.Reserva r
JOIN gestion_reserva.EstadoReserva er ON r.id_estado_reserva = er.id_estado_reserva

-- Cliente
JOIN gestion_reserva.Cliente c ON r.ruc_cliente = c.ruc
LEFT JOIN gestion_reserva.ClienteTelefono ct ON c.id_cliente = ct.id_cliente

-- Agente
JOIN gestion_reserva.AgenteReservas ar ON r.id_agente_reservas = ar.id_agente_reservas
JOIN gestion_reserva.Empleado e ON ar.id_empleado = e.id_empleado
LEFT JOIN gestion_reserva.EspecialidadEmpleado ee ON e.id_especialidad = ee.id_especialidad_empleado

-- Buque
JOIN gestion_reserva.Buque b ON r.matricula_buque = b.matricula
LEFT JOIN gestion_reserva.EstadoEmbarcacion eb ON b.id_estado_embarcacion = eb.id_estado_embarcacion

-- Ruta
JOIN gestion_reserva.Ruta rt ON r.id_ruta = rt.id_ruta

-- Contenedores
LEFT JOIN gestion_reserva.ReservaContenedor rc ON r.id_reserva = rc.id_reserva
LEFT JOIN gestion_reserva.Contenedor cont ON rc.id_contenedor = cont.id_contenedor
LEFT JOIN gestion_reserva.TipoContenedor tc ON cont.id_tipo_contenedor = tc.id_tipo_contenedor

-- Operaciones
LEFT JOIN gestion_reserva.ReservaOperacionMaritima rom ON r.id_reserva = rom.id_reserva
LEFT JOIN gestion_reserva.ReservaOperacionTerrestre rot ON r.id_reserva = rot.id_reserva

GROUP BY 
    r.id_reserva,
    r.codigo,
    r.fecha_registro,
    er.nombre,
    r.pago_total,
    c.ruc,
    c.razon_social,
    c.direccion,
    c.email,
    e.codigo,
    e.nombre,
    e.apellido,
    ee.nombre,
    b.matricula,
    b.nombre,
    b.capacidad,
    eb.nombre,
    b.ubicacion_actual,
    rt.codigo,
    rt.origen,
    rt.destino,
    rt.duracion,
    rt.tarifa

WITH DATA;
```

### P03 — Gestión de Clientes

#### Vista: `vm_clientes_historial_reservas`

**Descripción:** Perfil completo de clientes con historial de reservas y métricas de comportamiento.

```sql
CREATE MATERIALIZED VIEW gestion_reserva.vm_clientes_historial_reservas AS
SELECT 
    -- Información del cliente
    c.id_cliente,
    c.ruc,
    c.razon_social,
    c.direccion,
    c.email,
    STRING_AGG(DISTINCT ct.telefono, ', ' ORDER BY ct.telefono) AS telefonos,
    STRING_AGG(DISTINCT tt.nombre, ', ') AS tipos_telefono,
    
    -- Métricas de reservas
    COUNT(DISTINCT r.id_reserva) AS total_reservas,
    COUNT(DISTINCT CASE WHEN er.nombre = 'Confirmada' THEN r.id_reserva END) AS reservas_confirmadas,
    COUNT(DISTINCT CASE WHEN er.nombre = 'Activa' THEN r.id_reserva END) AS reservas_activas,
    COUNT(DISTINCT CASE WHEN er.nombre = 'Finalizada' THEN r.id_reserva END) AS reservas_finalizadas,
    COUNT(DISTINCT CASE WHEN er.nombre = 'Cancelada' THEN r.id_reserva END) AS reservas_canceladas,
    
    -- Métricas financieras
    COALESCE(SUM(r.pago_total), 0)::DECIMAL(12,2) AS ingresos_totales,
    COALESCE(AVG(r.pago_total), 0)::DECIMAL(10,2) AS ingreso_promedio,
    COALESCE(MAX(r.pago_total), 0)::DECIMAL(10,2) AS ingreso_maximo,
    COALESCE(MIN(r.pago_total), 0)::DECIMAL(10,2) AS ingreso_minimo,
    
    -- Fechas de actividad
    MIN(r.fecha_registro) AS primera_reserva,
    MAX(r.fecha_registro) AS ultima_reserva,
    (CURRENT_DATE - MAX(r.fecha_registro)) AS dias_desde_ultima_reserva,
    
    -- Preferencias de rutas
    STRING_AGG(DISTINCT rt.origen || ' → ' || rt.destino, ', ') AS rutas_utilizadas,
    COUNT(DISTINCT rt.id_ruta) AS total_rutas_diferentes,
    
    -- Buques utilizados
    COUNT(DISTINCT b.id_buque) AS total_buques_utilizados,
    STRING_AGG(DISTINCT b.nombre, ', ') AS buques_utilizados,
    
    -- Contenedores
    SUM(rc.cantidad) AS total_contenedores_reservados,
    COUNT(DISTINCT tc.id_tipo_contenedor) AS tipos_contenedores_diferentes,
    
    -- Agentes que atendieron
    COUNT(DISTINCT ar.id_agente_reservas) AS total_agentes_atendieron,
    STRING_AGG(DISTINCT e.nombre || ' ' || e.apellido, ', ') AS agentes_nombres,
    
    -- Atenciones registradas
    COUNT(DISTINCT ac.id_atencion_cliente) AS total_atenciones,
    MAX(ac.fecha_atencion) AS fecha_ultima_atencion,
    
    -- Clasificación del cliente
    CASE 
        WHEN COUNT(DISTINCT r.id_reserva) >= 10 AND COALESCE(SUM(r.pago_total), 0) >= 50000 THEN 'Premium'
        WHEN COUNT(DISTINCT r.id_reserva) >= 5 AND COALESCE(SUM(r.pago_total), 0) >= 25000 THEN 'Gold'
        WHEN COUNT(DISTINCT r.id_reserva) >= 2 THEN 'Silver'
        ELSE 'Bronze'
    END AS categoria_cliente,
    
    -- Estado del cliente
    CASE 
        WHEN MAX(r.fecha_registro) >= CURRENT_DATE - INTERVAL '30 days' THEN 'Activo'
        WHEN MAX(r.fecha_registro) >= CURRENT_DATE - INTERVAL '90 days' THEN 'Reciente'
        WHEN MAX(r.fecha_registro) >= CURRENT_DATE - INTERVAL '180 days' THEN 'Inactivo'
        ELSE 'Perdido'
    END AS estado_actividad
    
FROM gestion_reserva.Cliente c

-- Teléfonos
LEFT JOIN gestion_reserva.ClienteTelefono ct ON c.id_cliente = ct.id_cliente
LEFT JOIN gestion_reserva.TipoTelefono tt ON ct.id_tipo_telefono = tt.id_tipo_telefono

-- Reservas
LEFT JOIN gestion_reserva.Reserva r ON c.ruc = r.ruc_cliente
LEFT JOIN gestion_reserva.EstadoReserva er ON r.id_estado_reserva = er.id_estado_reserva

-- Rutas
LEFT JOIN gestion_reserva.Ruta rt ON r.id_ruta = rt.id_ruta

-- Buques
LEFT JOIN gestion_reserva.Buque b ON r.matricula_buque = b.matricula

-- Contenedores
LEFT JOIN gestion_reserva.ReservaContenedor rc ON r.id_reserva = rc.id_reserva
LEFT JOIN gestion_reserva.Contenedor cont ON rc.id_contenedor = cont.id_contenedor
LEFT JOIN gestion_reserva.TipoContenedor tc ON cont.id_tipo_contenedor = tc.id_tipo_contenedor

-- Agentes
LEFT JOIN gestion_reserva.AgenteReservas ar ON r.id_agente_reservas = ar.id_agente_reservas
LEFT JOIN gestion_reserva.Empleado e ON ar.id_empleado = e.id_empleado

-- Atenciones
LEFT JOIN gestion_reserva.AtencionCliente ac ON c.id_cliente = ac.id_cliente

GROUP BY 
    c.id_cliente,
    c.ruc,
    c.razon_social,
    c.direccion,
    c.email

WITH DATA;
```

### P01 — Registro de Reserva

#### Vista: `vm_disponibilidad_buques_rutas`

**Descripción:** Disponibilidad de buques y rutas para el registro de nuevas reservas.

```sql
CREATE MATERIALIZED VIEW gestion_reserva.vm_disponibilidad_buques_rutas AS
SELECT 
    -- Información del buque
    b.id_buque,
    b.matricula,
    b.nombre AS nombre_buque,
    b.capacidad,
    eb.nombre AS estado_buque,
    b.peso,
    b.ubicacion_actual,
    
    -- Capacidad disponible
    b.capacidad - COALESCE((
        SELECT SUM(om.cantidad_contenedores)
        FROM gestion_reserva.OperacionMaritima om
        JOIN gestion_reserva.Operacion o ON om.id_operacion = o.id_operacion
        WHERE om.id_buque = b.id_buque
        AND o.fecha_fin IS NULL
    ), 0) AS espacios_disponibles,
    
    -- Certificaciones
    COUNT(DISTINCT cb.id_certificacion) AS total_certificaciones,
    STRING_AGG(DISTINCT cert.nombre, ', ') AS certificaciones_activas,
    MIN(cb.fecha_vencimiento) AS proxima_vencimiento_certificacion,
    
    -- Rutas disponibles
    STRING_AGG(DISTINCT rt.codigo || ': ' || rt.origen || ' → ' || rt.destino, '; ') AS rutas_disponibles,
    COUNT(DISTINCT rt.id_ruta) AS total_rutas,
    
    -- Información de rutas
    AVG(rt.duracion)::DECIMAL(5,1) AS duracion_promedio_rutas,
    AVG(rt.tarifa)::DECIMAL(10,2) AS tarifa_promedio_rutas,
    MIN(rt.tarifa) AS tarifa_minima,
    MAX(rt.tarifa) AS tarifa_maxima,
    
    -- Reservas activas asociadas
    COUNT(DISTINCT r.id_reserva) AS reservas_activas_buque,
    
    -- Estado operacional
    CASE 
        WHEN eb.nombre = 'Disponible' AND 
             b.capacidad > COALESCE((
                 SELECT SUM(om.cantidad_contenedores)
                 FROM gestion_reserva.OperacionMaritima om
                 JOIN gestion_reserva.Operacion o ON om.id_operacion = o.id_operacion
                 WHERE om.id_buque = b.id_buque
                 AND o.fecha_fin IS NULL
             ), 0) THEN TRUE
        ELSE FALSE
    END AS buque_disponible,
    
    -- Fecha de actualización
    CURRENT_TIMESTAMP AS fecha_actualizacion
    
FROM gestion_reserva.Buque b
JOIN gestion_reserva.EstadoEmbarcacion eb ON b.id_estado_embarcacion = eb.id_estado_embarcacion

-- Certificaciones
LEFT JOIN gestion_reserva.CertificacionBuque cb ON b.id_buque = cb.id_buque
LEFT JOIN gestion_reserva.Certificacion cert ON cb.id_certificacion = cert.id_certificacion

-- Rutas (todas las rutas disponibles en el sistema)
CROSS JOIN gestion_reserva.Ruta rt

-- Reservas activas
LEFT JOIN gestion_reserva.Reserva r ON b.matricula = r.matricula_buque 
    AND r.id_estado_reserva IN (
        SELECT id_estado_reserva FROM gestion_reserva.EstadoReserva 
        WHERE nombre IN ('Activa', 'Confirmada')
    )

GROUP BY 
    b.id_buque,
    b.matricula,
    b.nombre,
    b.capacidad,
    eb.nombre,
    b.peso,
    b.ubicacion_actual

WITH DATA;
```
