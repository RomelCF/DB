### Creación de Triggers

#### 1. **Trigger: Actualizar disponibilidad de conductor al asignar operación**

```sql
-- Función que se ejecutará cuando se asigne un conductor a una operación
CREATE OR REPLACE FUNCTION operaciones_terrestres.trg_actualizar_disponibilidad_conductor()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
BEGIN
    -- Cuando se asigna un conductor a una operación terrestre, marcarlo como no disponible
    UPDATE operaciones_terrestres.Conductor
    SET disponibilidad = FALSE
    WHERE id_conductor = NEW.id_conductor;

    RAISE NOTICE 'Conductor % marcado como no disponible', NEW.id_conductor;
    RETURN NEW;
END;
$$;

-- Crear el trigger
CREATE TRIGGER trg_asignar_conductor
    AFTER INSERT ON operaciones_terrestres.OperacionTerrestre
    FOR EACH ROW
    EXECUTE FUNCTION operaciones_terrestres.trg_actualizar_disponibilidad_conductor();
```

#### 2. **Trigger: Liberar conductor al completar operación**

```sql
-- Función para liberar conductor cuando una operación se completa
CREATE OR REPLACE FUNCTION operaciones_terrestres.trg_liberar_conductor()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
DECLARE
    v_id_conductor CHAR(36);
BEGIN
    -- Solo liberar si el estado cambió a 'Completada' o 'Cancelada'
    IF NEW.id_estado_operacion != OLD.id_estado_operacion THEN
        -- Obtener el conductor de la operación terrestre
        SELECT id_conductor INTO v_id_conductor
        FROM operaciones_terrestres.OperacionTerrestre
        WHERE id_operacion = NEW.id_operacion;

        -- Si hay un conductor asignado y la operación está completada o cancelada
        IF v_id_conductor IS NOT NULL AND
           NEW.id_estado_operacion IN (
               SELECT id_estado_operacion
               FROM operaciones_terrestres.EstadoOperacion
               WHERE nombre IN ('Completada', 'Cancelada')
           ) THEN
            UPDATE operaciones_terrestres.Conductor
            SET disponibilidad = TRUE
            WHERE id_conductor = v_id_conductor;

            RAISE NOTICE 'Conductor % liberado y disponible nuevamente', v_id_conductor;
        END IF;
    END IF;

    RETURN NEW;
END;
$$;

-- Crear el trigger
CREATE TRIGGER trg_completar_operacion
    AFTER UPDATE ON operaciones_terrestres.Operacion
    FOR EACH ROW
    WHEN (NEW.id_estado_operacion IS DISTINCT FROM OLD.id_estado_operacion)
    EXECUTE FUNCTION operaciones_terrestres.trg_liberar_conductor();
```

#### 3. **Trigger: Validar capacidad del vehículo**

```sql
-- Función para validar que el vehículo tenga suficiente capacidad
CREATE OR REPLACE FUNCTION operaciones_terrestres.trg_validar_capacidad_vehiculo()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
DECLARE
    v_capacidad_vehiculo DECIMAL(10,2);
    v_estado_vehiculo VARCHAR(50);
BEGIN
    -- Obtener capacidad y estado del vehículo
    SELECT v.capacidad_ton, ev.nombre
    INTO v_capacidad_vehiculo, v_estado_vehiculo
    FROM operaciones_terrestres.Vehiculo v
    JOIN operaciones_terrestres.EstadoVehiculo ev ON v.id_estado_vehiculo = ev.id_estado_vehiculo
    WHERE v.id_vehiculo = NEW.id_vehiculo;

    -- Validar que el vehículo esté disponible
    IF v_estado_vehiculo != 'Disponible' THEN
        RAISE EXCEPTION 'El vehículo no está disponible para asignación. Estado actual: %', v_estado_vehiculo;
    END IF;

    -- Validar capacidad mínima (ejemplo: al menos 5 toneladas)
    IF v_capacidad_vehiculo < 5.00 THEN
        RAISE EXCEPTION 'El vehículo no tiene capacidad suficiente. Capacidad actual: % ton', v_capacidad_vehiculo;
    END IF;

    RETURN NEW;
END;
$$;

-- Crear el trigger
CREATE TRIGGER trg_validar_vehiculo_antes_asignar
    BEFORE INSERT ON operaciones_terrestres.OperacionTerrestre
    FOR EACH ROW
    EXECUTE FUNCTION operaciones_terrestres.trg_validar_capacidad_vehiculo();
```

#### 4. **Trigger: Registrar auditoría de incidencias**

```sql
-- Crear tabla de auditoría
CREATE TABLE IF NOT EXISTS operaciones_terrestres.AuditoriaIncidencia (
    id_auditoria CHAR(36) PRIMARY KEY DEFAULT gen_random_uuid()::TEXT,
    id_incidencia CHAR(36) NOT NULL,
    accion VARCHAR(20) NOT NULL,
    usuario VARCHAR(100),
    fecha_hora TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    estado_anterior VARCHAR(50),
    estado_nuevo VARCHAR(50),
    severidad_anterior INT,
    severidad_nueva INT,
    descripcion_cambio TEXT
);

-- Función para auditar incidencias
CREATE OR REPLACE FUNCTION operaciones_terrestres.trg_auditar_incidencia()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
DECLARE
    v_accion VARCHAR(20);
    v_estado_anterior VARCHAR(50);
    v_estado_nuevo VARCHAR(50);
BEGIN
    IF TG_OP = 'INSERT' THEN
        v_accion := 'INSERT';
        v_estado_nuevo := (SELECT nombre FROM operaciones_terrestres.EstadoIncidencia WHERE id_estado_incidencia = NEW.id_estado_incidencia);

        INSERT INTO operaciones_terrestres.AuditoriaIncidencia (
            id_incidencia, accion, usuario, estado_nuevo, severidad_nueva, descripcion_cambio
        ) VALUES (
            NEW.id_incidencia, v_accion, current_user, v_estado_nuevo, NEW.grado_severidad,
            'Nueva incidencia creada: ' || NEW.descripcion
        );

    ELSIF TG_OP = 'UPDATE' THEN
        v_accion := 'UPDATE';
        v_estado_anterior := (SELECT nombre FROM operaciones_terrestres.EstadoIncidencia WHERE id_estado_incidencia = OLD.id_estado_incidencia);
        v_estado_nuevo := (SELECT nombre FROM operaciones_terrestres.EstadoIncidencia WHERE id_estado_incidencia = NEW.id_estado_incidencia);

        INSERT INTO operaciones_terrestres.AuditoriaIncidencia (
            id_incidencia, accion, usuario, estado_anterior, estado_nuevo,
            severidad_anterior, severidad_nueva, descripcion_cambio
        ) VALUES (
            NEW.id_incidencia, v_accion, current_user, v_estado_anterior, v_estado_nuevo,
            OLD.grado_severidad, NEW.grado_severidad,
            'Incidencia actualizada'
        );

    ELSIF TG_OP = 'DELETE' THEN
        v_accion := 'DELETE';
        v_estado_anterior := (SELECT nombre FROM operaciones_terrestres.EstadoIncidencia WHERE id_estado_incidencia = OLD.id_estado_incidencia);

        INSERT INTO operaciones_terrestres.AuditoriaIncidencia (
            id_incidencia, accion, usuario, estado_anterior, severidad_anterior, descripcion_cambio
        ) VALUES (
            OLD.id_incidencia, v_accion, current_user, v_estado_anterior, OLD.grado_severidad,
            'Incidencia eliminada'
        );

        RETURN OLD;
    END IF;

    RETURN NEW;
END;
$$;

-- Crear el trigger
CREATE TRIGGER trg_auditar_incidencia_cambios
    AFTER INSERT OR UPDATE OR DELETE ON operaciones_terrestres.Incidencia
    FOR EACH ROW
    EXECUTE FUNCTION operaciones_terrestres.trg_auditar_incidencia();
```

### Procedimientos Almacenados

#### 1. **Procedimiento: Asignar operación terrestre completa**

```sql
CREATE OR REPLACE FUNCTION operaciones_terrestres.sp_asignar_operacion_terrestre(
    p_origen VARCHAR(100),
    p_destino VARCHAR(100),
    p_duracion INT,
    p_tarifa DECIMAL(10,2),
    p_id_vehiculo CHAR(36),
    p_id_conductor CHAR(36),
    p_costo_operacion DECIMAL(10,2)
)
RETURNS TABLE (
    id_operacion_creada CHAR(36),
    codigo_operacion VARCHAR(20),
    codigo_operacion_terrestre VARCHAR(20),
    codigo_ruta VARCHAR(20),
    mensaje TEXT
)
LANGUAGE plpgsql
AS $$
DECLARE
    v_id_operacion CHAR(36);
    v_id_ruta CHAR(36);
    v_id_ruta_terrestre CHAR(36);
    v_id_operacion_terrestre CHAR(36);
    v_codigo_operacion VARCHAR(20);
    v_codigo_operacion_terrestre VARCHAR(20);
    v_codigo_ruta VARCHAR(20);
    v_estado_pendiente CHAR(36);
BEGIN
    -- Obtener estado pendiente
    SELECT id_estado_operacion INTO v_estado_pendiente
    FROM operaciones_terrestres.EstadoOperacion
    WHERE nombre = 'Pendiente';

    -- Crear ruta
    v_codigo_ruta := operaciones_terrestres.generar_codigo_ruta();
    INSERT INTO operaciones_terrestres.Ruta (id_ruta, codigo, origen, destino, duracion, tarifa)
    VALUES (gen_random_uuid()::TEXT, v_codigo_ruta, p_origen, p_destino, p_duracion, p_tarifa)
    RETURNING id_ruta INTO v_id_ruta;

    -- Crear ruta terrestre
    INSERT INTO operaciones_terrestres.RutaTerrestre (id_ruta_terrestre, id_ruta)
    VALUES (gen_random_uuid()::TEXT, v_id_ruta)
    RETURNING id_ruta_terrestre INTO v_id_ruta_terrestre;

    -- Crear operación
    v_codigo_operacion := operaciones_terrestres.generar_codigo_operacion();
    INSERT INTO operaciones_terrestres.Operacion (id_operacion, codigo, fecha_inicio, id_estado_operacion)
    VALUES (gen_random_uuid()::TEXT, v_codigo_operacion, CURRENT_TIMESTAMP, v_estado_pendiente)
    RETURNING id_operacion INTO v_id_operacion;

    -- Crear operación terrestre
    v_codigo_operacion_terrestre := operaciones_terrestres.generar_codigo_operacion_terrestre();
    INSERT INTO operaciones_terrestres.OperacionTerrestre (
        id_operacion_terrestre, id_operacion, codigo, id_vehiculo, id_ruta_terrestre,
        id_conductor, costo_operacion_terrestre
    )
    VALUES (
        gen_random_uuid()::TEXT, v_id_operacion, v_codigo_operacion_terrestre,
        p_id_vehiculo, v_id_ruta_terrestre, p_id_conductor, p_costo_operacion
    )
    RETURNING id_operacion_terrestre INTO v_id_operacion_terrestre;

    -- Retornar resultados
    RETURN QUERY
    SELECT
        v_id_operacion,
        v_codigo_operacion,
        v_codigo_operacion_terrestre,
        v_codigo_ruta,
        'Operación terrestre creada exitosamente'::TEXT;

EXCEPTION
    WHEN OTHERS THEN
        RAISE EXCEPTION 'Error al crear operación terrestre: %', SQLERRM;
END;
$$;
```

**Ejemplo de uso:**
```sql

    (SELECT id_conductor FROM operaciones_terrestres.Conductor WHERE disponibilidad = TRUE LIMIT 1),
```

#### 2. **Procedimiento: Reportar incidencia en operación**

```sql
CREATE OR REPLACE FUNCTION operaciones_terrestres.sp_reportar_incidencia(
    p_id_operacion CHAR(36),
    p_tipo_incidencia VARCHAR(50),
    p_descripcion TEXT,
    p_grado_severidad INT,
    p_id_usuario CHAR(36)
)
RETURNS TABLE (
    id_incidencia_creada CHAR(36),
    codigo_incidencia VARCHAR(20),
    mensaje TEXT
)
LANGUAGE plpgsql
AS $$
DECLARE
    v_id_incidencia CHAR(36);
    v_codigo_incidencia VARCHAR(20);
    v_id_tipo_incidencia CHAR(36);
    v_id_estado_incidencia CHAR(36);
BEGIN
    -- Validar severidad
    IF p_grado_severidad NOT BETWEEN 1 AND 5 THEN
        RAISE EXCEPTION 'El grado de severidad debe estar entre 1 y 5';
    END IF;

    -- Obtener tipo de incidencia
    SELECT id_tipo_incidencia INTO v_id_tipo_incidencia
    FROM operaciones_terrestres.TipoIncidencia
    WHERE nombre = p_tipo_incidencia;

    IF v_id_tipo_incidencia IS NULL THEN
        RAISE EXCEPTION 'Tipo de incidencia no válido: %', p_tipo_incidencia;
    END IF;

    -- Obtener estado 'Reportada'
    SELECT id_estado_incidencia INTO v_id_estado_incidencia
    FROM operaciones_terrestres.EstadoIncidencia
    WHERE nombre = 'Reportada';

    -- Generar código
    v_codigo_incidencia := operaciones_terrestres.generar_codigo_incidencia();

    -- Crear incidencia
    INSERT INTO operaciones_terrestres.Incidencia (
        id_incidencia, codigo, id_tipo_incidencia, descripcion, grado_severidad,
        fecha_hora, id_estado_incidencia, id_operacion, id_usuario
    )
    VALUES (
        gen_random_uuid()::TEXT, v_codigo_incidencia, v_id_tipo_incidencia, p_descripcion,
        p_grado_severidad, CURRENT_TIMESTAMP, v_id_estado_incidencia, p_id_operacion, p_id_usuario
    )
    RETURNING id_incidencia INTO v_id_incidencia;

    -- Retornar resultado
    RETURN QUERY
    SELECT
        v_id_incidencia,
        v_codigo_incidencia,
        'Incidencia reportada exitosamente'::TEXT;

EXCEPTION
    WHEN OTHERS THEN
        RAISE EXCEPTION 'Error al reportar incidencia: %', SQLERRM;
END;
$$;
```

**Ejemplo de uso:**
```sql
SELECT * FROM operaciones_terrestres.sp_reportar_incidencia(
    (SELECT id_operacion FROM operaciones_terrestres.Operacion WHERE codigo LIKE 'OP-%' LIMIT 1),
    'Operacional',
    'Falla mecánica en el vehículo durante el transporte',
    4,
    (SELECT id_usuario FROM operaciones_terrestres.Usuario LIMIT 1)
);
```

### Vistas Materializadas

#### 1. **Vista Materializada: Dashboard de operaciones terrestres**

```sql
CREATE MATERIALIZED VIEW operaciones_terrestres.vm_dashboard_operaciones AS
SELECT
    -- Métricas de operaciones
    COUNT(DISTINCT o.id_operacion) AS total_operaciones,
    COUNT(DISTINCT CASE WHEN eo.nombre = 'En curso' THEN o.id_operacion END) AS operaciones_en_curso,
    COUNT(DISTINCT CASE WHEN eo.nombre = 'Completada' THEN o.id_operacion END) AS operaciones_completadas,
    COUNT(DISTINCT CASE WHEN eo.nombre = 'Pendiente' THEN o.id_operacion END) AS operaciones_pendientes,
    COUNT(DISTINCT CASE WHEN eo.nombre = 'Cancelada' THEN o.id_operacion END) AS operaciones_canceladas,

    -- Métricas de vehículos
    COUNT(DISTINCT v.id_vehiculo) AS total_vehiculos,
    COUNT(DISTINCT CASE WHEN ev.nombre = 'Disponible' THEN v.id_vehiculo END) AS vehiculos_disponibles,
    COUNT(DISTINCT CASE WHEN ev.nombre = 'En ruta' THEN v.id_vehiculo END) AS vehiculos_en_ruta,
    COUNT(DISTINCT CASE WHEN ev.nombre = 'Mantenimiento' THEN v.id_vehiculo END) AS vehiculos_en_mantenimiento,

    -- Métricas de conductores
    COUNT(DISTINCT c.id_conductor) AS total_conductores,
    COUNT(DISTINCT CASE WHEN c.disponibilidad = TRUE THEN c.id_conductor END) AS conductores_disponibles,
    COUNT(DISTINCT CASE WHEN c.disponibilidad = FALSE THEN c.id_conductor END) AS conductores_ocupados,

    -- Métricas de rutas
    COUNT(DISTINCT r.id_ruta) AS total_rutas,
    AVG(r.duracion)::DECIMAL(10,2) AS duracion_promedio_horas,
    AVG(r.tarifa)::DECIMAL(10,2) AS tarifa_promedio,

    -- Métricas de incidencias
    COUNT(DISTINCT i.id_incidencia) AS total_incidencias,
    COUNT(DISTINCT CASE WHEN ei.nombre = 'Reportada' THEN i.id_incidencia END) AS incidencias_reportadas,
    COUNT(DISTINCT CASE WHEN ei.nombre = 'En investigación' THEN i.id_incidencia END) AS incidencias_investigacion,
    COUNT(DISTINCT CASE WHEN ei.nombre = 'Resuelta' THEN i.id_incidencia END) AS incidencias_resueltas,
    COUNT(DISTINCT CASE WHEN ei.nombre = 'Cerrada' THEN i.id_incidencia END) AS incidencias_cerradas,
    AVG(CASE WHEN ei.nombre = 'Resuelta' THEN i.grado_severidad END)::DECIMAL(3,2) AS severidad_promedio_resueltas,

    -- Métricas financieras
    SUM(ot.costo_operacion_terrestre)::DECIMAL(12,2) AS costos_totales,
    SUM(r.tarifa)::DECIMAL(12,2) AS ingresos_totales,
    (SUM(r.tarifa) - SUM(ot.costo_operacion_terrestre))::DECIMAL(12,2) AS utilidad_neta,

    -- Capacidad total
    SUM(v.capacidad_ton)::DECIMAL(12,2) AS capacidad_total_vehiculos_ton,

    -- Fecha de actualización
    CURRENT_TIMESTAMP AS fecha_actualizacion

FROM operaciones_terrestres.Operacion o
JOIN operaciones_terrestres.EstadoOperacion eo ON o.id_estado_operacion = eo.id_estado_operacion
LEFT JOIN operaciones_terrestres.OperacionTerrestre ot ON o.id_operacion = ot.id_operacion
LEFT JOIN operaciones_terrestres.Vehiculo v ON ot.id_vehiculo = v.id_vehiculo
LEFT JOIN operaciones_terrestres.EstadoVehiculo ev ON v.id_estado_vehiculo = ev.id_estado_vehiculo
LEFT JOIN operaciones_terrestres.Conductor c ON ot.id_conductor = c.id_conductor
LEFT JOIN operaciones_terrestres.RutaTerrestre rt ON ot.id_ruta_terrestre = rt.id_ruta_terrestre
LEFT JOIN operaciones_terrestres.Ruta r ON rt.id_ruta = r.id_ruta
LEFT JOIN operaciones_terrestres.Incidencia i ON o.id_operacion = i.id_operacion
LEFT JOIN operaciones_terrestres.EstadoIncidencia ei ON i.id_estado_incidencia = ei.id_estado_incidencia

WITH DATA;

-- Crear índice para mejorar rendimiento
CREATE INDEX idx_vm_dashboard_fecha ON operaciones_terrestres.vm_dashboard_operaciones(fecha_actualizacion);
```

#### 2. **Vista Materializada: Análisis de incidencias por tipo y severidad**

```sql
CREATE MATERIALIZED VIEW operaciones_terrestres.vm_analisis_incidencias AS
SELECT
    -- Información del tipo de incidencia
    ti.nombre AS tipo_incidencia,

    -- Métricas por tipo
    COUNT(i.id_incidencia) AS total_incidencias,
    AVG(i.grado_severidad)::DECIMAL(3,2) AS severidad_promedio,

    -- Distribución por severidad
    COUNT(CASE WHEN i.grado_severidad = 1 THEN 1 END) AS severidad_1_minima,
    COUNT(CASE WHEN i.grado_severidad = 2 THEN 1 END) AS severidad_2_baja,
    COUNT(CASE WHEN i.grado_severidad = 3 THEN 1 END) AS severidad_3_media,
    COUNT(CASE WHEN i.grado_severidad = 4 THEN 1 END) AS severidad_4_alta,
    COUNT(CASE WHEN i.grado_severidad = 5 THEN 1 END) AS severidad_5_critica,

    -- Distribución por estado
    COUNT(CASE WHEN ei.nombre = 'Reportada' THEN 1 END) AS estado_reportada,
    COUNT(CASE WHEN ei.nombre = 'En investigación' THEN 1 END) AS estado_investigacion,
    COUNT(CASE WHEN ei.nombre = 'Resuelta' THEN 1 END) AS estado_resuelta,
    COUNT(CASE WHEN ei.nombre = 'Cerrada' THEN 1 END) AS estado_cerrada,

    -- Tiempo promedio de resolución (en días)
    AVG(
        CASE
            WHEN ei.nombre IN ('Resuelta', 'Cerrada') THEN
                EXTRACT(EPOCH FROM (CURRENT_TIMESTAMP - i.fecha_hora)) / 86400
            ELSE NULL
        END
    )::DECIMAL(10,2) AS dias_promedio_resolucion,

    -- Incidencias últimos 30 días
    COUNT(CASE WHEN i.fecha_hora >= CURRENT_DATE - INTERVAL '30 days' THEN 1 END) AS incidencias_ultimos_30_dias,

    -- Incidencias pendientes de resolver
    COUNT(CASE WHEN ei.nombre IN ('Reportada', 'En investigación') THEN 1 END) AS incidencias_pendientes,

    -- Fecha de actualización
    CURRENT_TIMESTAMP AS fecha_actualizacion

FROM operaciones_terrestres.TipoIncidencia ti
LEFT JOIN operaciones_terrestres.Incidencia i ON ti.id_tipo_incidencia = i.id_tipo_incidencia
LEFT JOIN operaciones_terrestres.EstadoIncidencia ei ON i.id_estado_incidencia = ei.id_estado_incidencia

GROUP BY ti.id_tipo_incidencia, ti.nombre
ORDER BY total_incidencias DESC

WITH DATA;

-- Crear índice
CREATE INDEX idx_vm_analisis_incidencias ON operaciones_terrestres.vm_analisis_incidencias(tipo_incidencia);
```

```sql
-- Función para actualizar stock de repuestos
CREATE OR REPLACE FUNCTION mantenimiento_logistico.trg_actualizar_stock_repuesto()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
DECLARE
    v_stock_actual INT;
    v_stock_minimo INT;
    v_nombre_repuesto VARCHAR(100);
BEGIN
    -- Obtener stock actual y mínimo
    SELECT stock, stock_minimo, nombre
    INTO v_stock_actual, v_stock_minimo, v_nombre_repuesto
    FROM mantenimiento_logistico.Repuesto
    WHERE id_repuesto = NEW.id_repuesto;

    -- Validar que hay stock suficiente
    IF v_stock_actual < NEW.cantidad THEN
        RAISE EXCEPTION 'Stock insuficiente para repuesto %. Stock actual: %, Cantidad solicitada: %',
            v_nombre_repuesto, v_stock_actual, NEW.cantidad;
    END IF;

    -- Actualizar stock
    UPDATE mantenimiento_logistico.Repuesto
    SET stock = stock - NEW.cantidad
    WHERE id_repuesto = NEW.id_repuesto;

    -- Advertencia si el stock queda por debajo del mínimo
    IF (v_stock_actual - NEW.cantidad) <= v_stock_minimo THEN
        RAISE WARNING 'ALERTA: Stock de repuesto % está por debajo del mínimo. Stock actual: %, Mínimo: %',
            v_nombre_repuesto, (v_stock_actual - NEW.cantidad), v_stock_minimo;
    END IF;

    RAISE NOTICE 'Stock actualizado para repuesto %. Nuevo stock: %', v_nombre_repuesto, (v_stock_actual - NEW.cantidad);
    RETURN NEW;
END;
$$;

-- Crear el trigger
CREATE TRIGGER trg_usar_repuesto
    BEFORE INSERT ON mantenimiento_logistico.OperacionMantenimientoRepuesto
    FOR EACH ROW
    EXECUTE FUNCTION mantenimiento_logistico.trg_actualizar_stock_repuesto();
```

#### 2. **Trigger: Cambiar estado de activo al iniciar mantenimiento**

```sql
-- Función para cambiar estado del activo a "En mantenimiento"
CREATE OR REPLACE FUNCTION mantenimiento_logistico.trg_cambiar_estado_activo_mantenimiento()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
DECLARE
    v_id_activo CHAR(36);
    v_id_estado_mantenimiento CHAR(36);
BEGIN
    -- Obtener el activo relacionado con la operación de mantenimiento
    SELECT COALESCE(pm.id_activo, sm.id_activo)
    INTO v_id_activo
    FROM mantenimiento_logistico.OperacionMantenimiento om
    LEFT JOIN mantenimiento_logistico.PlanMantenimiento pm ON om.id_plan_mantenimiento = pm.id_plan_mantenimiento
    LEFT JOIN mantenimiento_logistico.SolicitudMantenimiento sm ON om.id_solicitud_mantenimiento = sm.id_solicitud_mantenimiento
    WHERE om.id_operacion_mantenimiento = NEW.id_operacion_mantenimiento;

    -- Obtener ID del estado "En mantenimiento"
    SELECT id_estado_activo INTO v_id_estado_mantenimiento
    FROM mantenimiento_logistico.EstadoActivo
    WHERE nombre = 'En mantenimiento';

    -- Actualizar estado del activo si se encuentra el activo
    IF v_id_activo IS NOT NULL AND v_id_estado_mantenimiento IS NOT NULL THEN
        UPDATE mantenimiento_logistico.Activo
        SET id_estado_activo = v_id_estado_mantenimiento
        WHERE id_activo = v_id_activo;

        RAISE NOTICE 'Activo % cambiado a estado "En mantenimiento"', v_id_activo;
    END IF;

    RETURN NEW;
END;
$$;

-- Crear el trigger
CREATE TRIGGER trg_iniciar_mantenimiento_activo
    AFTER INSERT ON mantenimiento_logistico.OrdenMantenimiento
    FOR EACH ROW
    WHEN (NEW.id_estado_orden IN (SELECT id_estado_orden FROM mantenimiento_logistico.EstadoOrden WHERE nombre IN ('Abierta', 'En ejecución')))
    EXECUTE FUNCTION mantenimiento_logistico.trg_cambiar_estado_activo_mantenimiento();
```

#### 3. **Trigger: Restaurar estado de activo al completar mantenimiento**

```sql
-- Función para restaurar estado del activo a "Operativo"
CREATE OR REPLACE FUNCTION mantenimiento_logistico.trg_restaurar_estado_activo()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
DECLARE
    v_id_activo CHAR(36);
    v_id_estado_operativo CHAR(36);
BEGIN
    -- Solo ejecutar si el estado cambió a 'Completada'
    IF NEW.id_estado_orden != OLD.id_estado_orden THEN
        -- Obtener el activo relacionado
        SELECT COALESCE(pm.id_activo, sm.id_activo)
        INTO v_id_activo
        FROM mantenimiento_logistico.OperacionMantenimiento om
        LEFT JOIN mantenimiento_logistico.PlanMantenimiento pm ON om.id_plan_mantenimiento = pm.id_plan_mantenimiento
        LEFT JOIN mantenimiento_logistico.SolicitudMantenimiento sm ON om.id_solicitud_mantenimiento = sm.id_solicitud_mantenimiento
        WHERE om.id_operacion_mantenimiento = NEW.id_operacion_mantenimiento;

        -- Obtener ID del estado "Operativo"
        SELECT id_estado_activo INTO v_id_estado_operativo
        FROM mantenimiento_logistico.EstadoActivo
        WHERE nombre = 'Operativo';

        -- Si la orden está completada, restaurar el activo a operativo
        IF v_id_activo IS NOT NULL AND v_id_estado_operativo IS NOT NULL AND
           NEW.id_estado_orden IN (SELECT id_estado_orden FROM mantenimiento_logistico.EstadoOrden WHERE nombre = 'Completada') THEN
            UPDATE mantenimiento_logistico.Activo
            SET id_estado_activo = v_id_estado_operativo
            WHERE id_activo = v_id_activo;

            RAISE NOTICE 'Activo % restaurado a estado "Operativo"', v_id_activo;
        END IF;
    END IF;

    RETURN NEW;
END;
$$;

-- Crear el trigger
CREATE TRIGGER trg_completar_mantenimiento_activo
    AFTER UPDATE ON mantenimiento_logistico.OrdenMantenimiento
    FOR EACH ROW
    WHEN (NEW.id_estado_orden IS DISTINCT FROM OLD.id_estado_orden)
    EXECUTE FUNCTION mantenimiento_logistico.trg_restaurar_estado_activo();
```

```sql
-- Crear tabla de auditoría
CREATE TABLE IF NOT EXISTS mantenimiento_logistico.AuditoriaOrdenMantenimiento (

    id_orden CHAR(36) NOT NULL,

    usuario VARCHAR(100),
    fecha_hora TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    estado_anterior VARCHAR(50),

    fecha_programada_anterior DATE,
    fecha_programada_nueva DATE,
    fecha_cierre_anterior DATE,
    fecha_cierre_nueva DATE,
    descripcion_cambio TEXT
);

-- Función para auditar órdenes de mantenimiento
CREATE OR REPLACE FUNCTION mantenimiento_logistico.trg_auditar_orden_mantenimiento()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
DECLARE
    v_accion VARCHAR(20);
    v_estado_anterior VARCHAR(50);
    v_estado_nuevo VARCHAR(50);
BEGIN
    IF TG_OP = 'INSERT' THEN
        v_accion := 'INSERT';
        v_estado_nuevo := (SELECT nombre FROM mantenimiento_logistico.EstadoOrden WHERE id_estado_orden = NEW.id_estado_orden);

        INSERT INTO mantenimiento_logistico.AuditoriaOrdenMantenimiento (
            id_orden, accion, usuario, estado_nuevo, fecha_programada_nueva, descripcion_cambio
        ) VALUES (
            NEW.id_orden, v_accion, current_user, v_estado_nuevo, NEW.fecha_programada,
            'Nueva orden de mantenimiento creada: ' || NEW.codigo
        );

    ELSIF TG_OP = 'UPDATE' THEN
        v_accion := 'UPDATE';
        v_estado_anterior := (SELECT nombre FROM mantenimiento_logistico.EstadoOrden WHERE id_estado_orden = OLD.id_estado_orden);
        v_estado_nuevo := (SELECT nombre FROM mantenimiento_logistico.EstadoOrden WHERE id_estado_orden = NEW.id_estado_orden);

        INSERT INTO mantenimiento_logistico.AuditoriaOrdenMantenimiento (
            id_orden, accion, usuario, estado_anterior, estado_nuevo,
            fecha_programada_anterior, fecha_programada_nueva,
            fecha_cierre_anterior, fecha_cierre_nueva, descripcion_cambio
        ) VALUES (
            NEW.id_orden, v_accion, current_user, v_estado_anterior, v_estado_nuevo,
            OLD.fecha_programada, NEW.fecha_programada,
            OLD.fecha_cierre, NEW.fecha_cierre,
            'Orden de mantenimiento actualizada'
        );

    ELSIF TG_OP = 'DELETE' THEN
        v_accion := 'DELETE';
        v_estado_anterior := (SELECT nombre FROM mantenimiento_logistico.EstadoOrden WHERE id_estado_orden = OLD.id_estado_orden);

        INSERT INTO mantenimiento_logistico.AuditoriaOrdenMantenimiento (
            id_orden, accion, usuario, estado_anterior, fecha_programada_anterior,
            fecha_cierre_anterior, descripcion_cambio
        ) VALUES (
            OLD.id_orden, v_accion, current_user, v_estado_anterior,
            OLD.fecha_programada, OLD.fecha_cierre,
            'Orden de mantenimiento eliminada'
        );

        RETURN OLD;
    END IF;

    RETURN NEW;
END;
$$;

-- Crear el trigger
CREATE TRIGGER trg_auditar_orden_cambios
    AFTER INSERT OR UPDATE OR DELETE ON mantenimiento_logistico.OrdenMantenimiento
    FOR EACH ROW
```

```sql
CREATE OR REPLACE FUNCTION mantenimiento_logistico.sp_crear_solicitud_mantenimiento(
    p_descripcion_problema TEXT,
    p_id_activo CHAR(36),
    p_id_responsable CHAR(36),
    p_prioridad VARCHAR(50)
)
RETURNS TABLE (
    id_solicitud_creada CHAR(36),
    codigo_solicitud VARCHAR(20),
    mensaje TEXT
)
LANGUAGE plpgsql
AS $$
DECLARE
    v_id_solicitud CHAR(36);
    v_codigo_solicitud VARCHAR(20);
    v_id_prioridad CHAR(36);
    v_id_estado_pendiente CHAR(36);
BEGIN
    -- Obtener ID de prioridad
    SELECT id_prioridad INTO v_id_prioridad
    FROM mantenimiento_logistico.Prioridad
    WHERE nombre = p_prioridad;

    IF v_id_prioridad IS NULL THEN
        RAISE EXCEPTION 'Prioridad no válida: %. Valores válidos: Baja, Media, Alta, Crítica', p_prioridad;
    END IF;

    -- Obtener estado pendiente
    SELECT id_estado_solicitud INTO v_id_estado_pendiente
    FROM mantenimiento_logistico.EstadoSolicitud
    WHERE nombre = 'Pendiente';

    -- Generar código
    v_codigo_solicitud := mantenimiento_logistico.generar_codigo_solicitud_mantenimiento();

    -- Crear solicitud
    INSERT INTO mantenimiento_logistico.SolicitudMantenimiento (
        id_solicitud_mantenimiento, codigo, descripcion_problema, fecha_solicitud,
        id_prioridad, id_estado_solicitud, id_responsable_solicitud, id_activo
    )
    VALUES (
        gen_random_uuid()::TEXT, v_codigo_solicitud, p_descripcion_problema, CURRENT_DATE,
        v_id_prioridad, v_id_estado_pendiente, p_id_responsable, p_id_activo
    )
    RETURNING id_solicitud_mantenimiento INTO v_id_solicitud;

    -- Retornar resultado
    RETURN QUERY
    SELECT
        v_id_solicitud,
        v_codigo_solicitud,
        'Solicitud de mantenimiento creada exitosamente con prioridad: ' || p_prioridad;

EXCEPTION
    WHEN OTHERS THEN
        RAISE EXCEPTION 'Error al crear solicitud de mantenimiento: %', SQLERRM;
END;
$$;
```

#### 2. **Procedimiento: Generar orden de mantenimiento desde solicitud**

```sql
CREATE OR REPLACE FUNCTION mantenimiento_logistico.sp_generar_orden_desde_solicitud(
    p_id_solicitud CHAR(36),
    p_tipo_mantenimiento VARCHAR(50),
    p_fecha_programada DATE,
    p_id_tecnico CHAR(36)
)
RETURNS TABLE (
    id_orden_creada CHAR(36),
    codigo_orden VARCHAR(20),
    codigo_operacion VARCHAR(20),
    mensaje TEXT
)
LANGUAGE plpgsql
AS $$
DECLARE
    v_id_operacion CHAR(36);
    v_id_operacion_mantenimiento CHAR(36);
    v_id_orden CHAR(36);
    v_codigo_operacion VARCHAR(20);
    v_codigo_orden VARCHAR(20);
    v_id_tipo_mantenimiento CHAR(36);
    v_id_estado_operacion CHAR(36);
    v_id_estado_orden CHAR(36);
BEGIN
    -- Validar que la solicitud existe
    IF NOT EXISTS (SELECT 1 FROM mantenimiento_logistico.SolicitudMantenimiento WHERE id_solicitud_mantenimiento = p_id_solicitud) THEN
        RAISE EXCEPTION 'La solicitud de mantenimiento no existe';
    END IF;

    -- Obtener tipo de mantenimiento
    SELECT id_tipo_mantenimiento INTO v_id_tipo_mantenimiento
    FROM mantenimiento_logistico.TipoMantenimiento
    WHERE nombre = p_tipo_mantenimiento;

    IF v_id_tipo_mantenimiento IS NULL THEN
        RAISE EXCEPTION 'Tipo de mantenimiento no válido: %', p_tipo_mantenimiento;
    END IF;

    -- Obtener estados
    SELECT id_estado_operacion INTO v_id_estado_operacion
    FROM mantenimiento_logistico.EstadoOperacion
    WHERE nombre = 'Pendiente';

    SELECT id_estado_orden INTO v_id_estado_orden
    FROM mantenimiento_logistico.EstadoOrden
    WHERE nombre = 'Abierta';

    -- Crear operación
    v_codigo_operacion := mantenimiento_logistico.generar_codigo_operacion();
    INSERT INTO mantenimiento_logistico.Operacion (id_operacion, codigo, fecha_inicio, id_estado_operacion)
    VALUES (gen_random_uuid()::TEXT, v_codigo_operacion, CURRENT_TIMESTAMP, v_id_estado_operacion)
    RETURNING id_operacion INTO v_id_operacion;

    -- Crear operación de mantenimiento
    INSERT INTO mantenimiento_logistico.OperacionMantenimiento (
        id_operacion_mantenimiento, id_operacion, id_plan_mantenimiento, id_solicitud_mantenimiento
    )
    VALUES (gen_random_uuid()::TEXT, v_id_operacion, NULL, p_id_solicitud)
    RETURNING id_operacion_mantenimiento INTO v_id_operacion_mantenimiento;

    -- Crear orden de mantenimiento
    v_codigo_orden := mantenimiento_logistico.generar_codigo_orden_mantenimiento();
    INSERT INTO mantenimiento_logistico.OrdenMantenimiento (
        id_orden, codigo, fecha_generada, fecha_programada, id_tipo_mantenimiento,
        id_estado_orden, id_operacion_mantenimiento
    )
    VALUES (
        gen_random_uuid()::TEXT, v_codigo_orden, CURRENT_DATE, p_fecha_programada,
        v_id_tipo_mantenimiento, v_id_estado_orden, v_id_operacion_mantenimiento
    )
    RETURNING id_orden INTO v_id_orden;

    -- Asignar técnico
    IF p_id_tecnico IS NOT NULL THEN
        INSERT INTO mantenimiento_logistico.OperacionMantenimientoTecnico (
            id_operacion_mantenimiento, id_tecnico, fecha_asignacion
        )
        VALUES (v_id_operacion_mantenimiento, p_id_tecnico, CURRENT_DATE);
    END IF;

    -- Actualizar estado de la solicitud
    UPDATE mantenimiento_logistico.SolicitudMantenimiento
    SET id_estado_solicitud = (SELECT id_estado_solicitud FROM mantenimiento_logistico.EstadoSolicitud WHERE nombre = 'En ejecución')
    WHERE id_solicitud_mantenimiento = p_id_solicitud;

    -- Retornar resultado
    RETURN QUERY
    SELECT
        v_id_orden,
        v_codigo_orden,
        v_codigo_operacion,
        'Orden de mantenimiento generada exitosamente'::TEXT;

EXCEPTION
    WHEN OTHERS THEN
        RAISE EXCEPTION 'Error al generar orden de mantenimiento: %', SQLERRM;
END;
$$;
```

**Ejemplo de uso:**
```sql
SELECT * FROM mantenimiento_logistico.sp_generar_orden_desde_solicitud(
    (SELECT id_solicitud_mantenimiento FROM mantenimiento_logistico.SolicitudMantenimiento WHERE codigo LIKE 'SOL-%' LIMIT 1),
    'Correctivo',
    CURRENT_DATE + INTERVAL '2 days',
    (SELECT id_tecnico FROM mantenimiento_logistico.Tecnico LIMIT 1)
```

```sql
CREATE MATERIALIZED VIEW mantenimiento_logistico.vm_dashboard_mantenimiento AS
SELECT
    -- Métricas de activos
    COUNT(DISTINCT a.id_activo) AS total_activos,
    COUNT(DISTINCT CASE WHEN ea.nombre = 'Operativo' THEN a.id_activo END) AS activos_operativos,
    COUNT(DISTINCT CASE WHEN ea.nombre = 'En mantenimiento' THEN a.id_activo END) AS activos_en_mantenimiento,
    COUNT(DISTINCT CASE WHEN ea.nombre = 'Fuera de servicio' THEN a.id_activo END) AS activos_fuera_servicio,
    COUNT(DISTINCT CASE WHEN ea.nombre = 'Dado de baja' THEN a.id_activo END) AS activos_dados_baja,

    -- Porcentaje de disponibilidad
    ROUND(
        (COUNT(DISTINCT CASE WHEN ea.nombre = 'Operativo' THEN a.id_activo END) * 100.0 /
        NULLIF(COUNT(DISTINCT a.id_activo), 0)), 2
    ) AS porcentaje_disponibilidad,

    -- Métricas de órdenes
    COUNT(DISTINCT om.id_orden) AS total_ordenes,
    COUNT(DISTINCT CASE WHEN eo.nombre = 'Abierta' THEN om.id_orden END) AS ordenes_abiertas,
    COUNT(DISTINCT CASE WHEN eo.nombre = 'En ejecución' THEN om.id_orden END) AS ordenes_en_ejecucion,
    COUNT(DISTINCT CASE WHEN eo.nombre = 'Completada' THEN om.id_orden END) AS ordenes_completadas,
    COUNT(DISTINCT CASE WHEN eo.nombre = 'Cancelada' THEN om.id_orden END) AS ordenes_canceladas,

    -- Métricas por tipo de mantenimiento
    COUNT(DISTINCT CASE WHEN tm.nombre = 'Preventivo' THEN om.id_orden END) AS mantenimientos_preventivos,
    COUNT(DISTINCT CASE WHEN tm.nombre = 'Correctivo' THEN om.id_orden END) AS mantenimientos_correctivos,
    COUNT(DISTINCT CASE WHEN tm.nombre = 'Predictivo' THEN om.id_orden END) AS mantenimientos_predictivos,
    COUNT(DISTINCT CASE WHEN tm.nombre = 'Emergencia' THEN om.id_orden END) AS mantenimientos_emergencia,

    -- Métricas de solicitudes
    COUNT(DISTINCT sm.id_solicitud_mantenimiento) AS total_solicitudes,
    COUNT(DISTINCT CASE WHEN es.nombre = 'Pendiente' THEN sm.id_solicitud_mantenimiento END) AS solicitudes_pendientes,
    COUNT(DISTINCT CASE WHEN es.nombre = 'Aprobada' THEN sm.id_solicitud_mantenimiento END) AS solicitudes_aprobadas,
    COUNT(DISTINCT CASE WHEN es.nombre = 'En ejecución' THEN sm.id_solicitud_mantenimiento END) AS solicitudes_en_ejecucion,
    COUNT(DISTINCT CASE WHEN es.nombre = 'Rechazada' THEN sm.id_solicitud_mantenimiento END) AS solicitudes_rechazadas,

    -- Métricas de planes
    COUNT(DISTINCT pm.id_plan_mantenimiento) AS total_planes,
    COUNT(DISTINCT CASE WHEN ep.nombre = 'Activo' THEN pm.id_plan_mantenimiento END) AS planes_activos,

    -- Métricas de repuestos
    (SELECT COUNT(*) FROM mantenimiento_logistico.Repuesto) AS total_repuestos,
    (SELECT COUNT(*) FROM mantenimiento_logistico.Repuesto WHERE stock <= stock_minimo) AS repuestos_stock_critico,

    -- Métricas de costos
    SUM(omr.cantidad * omr.precio_unitario)::DECIMAL(12,2) AS costos_materiales_totales,
    AVG(omr.cantidad * omr.precio_unitario)::DECIMAL(10,2) AS costo_promedio_materiales,

    -- Métricas de técnicos
    (SELECT COUNT(*) FROM mantenimiento_logistico.Tecnico) AS total_tecnicos,
    COUNT(DISTINCT omt.id_tecnico) AS tecnicos_asignados,

    -- Tiempo promedio de resolución (días)
    AVG(
        CASE
            WHEN om.fecha_cierre IS NOT NULL AND om.fecha_generada IS NOT NULL THEN
                EXTRACT(EPOCH FROM (om.fecha_cierre - om.fecha_generada)) / 86400.0
            ELSE NULL
        END
    )::DECIMAL(10,2) AS dias_promedio_resolucion,

    -- Fecha de actualización
    CURRENT_TIMESTAMP AS fecha_actualizacion

FROM mantenimiento_logistico.Activo a
JOIN mantenimiento_logistico.EstadoActivo ea ON a.id_estado_activo = ea.id_estado_activo
LEFT JOIN mantenimiento_logistico.PlanMantenimiento pm ON a.id_activo = pm.id_activo
LEFT JOIN mantenimiento_logistico.EstadoPlan ep ON pm.id_estado_plan = ep.id_estado_plan
LEFT JOIN mantenimiento_logistico.SolicitudMantenimiento sm ON a.id_activo = sm.id_activo
LEFT JOIN mantenimiento_logistico.EstadoSolicitud es ON sm.id_estado_solicitud = es.id_estado_solicitud
LEFT JOIN mantenimiento_logistico.OperacionMantenimiento opm ON (opm.id_plan_mantenimiento = pm.id_plan_mantenimiento OR opm.id_solicitud_mantenimiento = sm.id_solicitud_mantenimiento)
LEFT JOIN mantenimiento_logistico.OrdenMantenimiento om ON opm.id_operacion_mantenimiento = om.id_operacion_mantenimiento
LEFT JOIN mantenimiento_logistico.EstadoOrden eo ON om.id_estado_orden = eo.id_estado_orden
LEFT JOIN mantenimiento_logistico.TipoMantenimiento tm ON om.id_tipo_mantenimiento = tm.id_tipo_mantenimiento
LEFT JOIN mantenimiento_logistico.OperacionMantenimientoTecnico omt ON opm.id_operacion_mantenimiento = omt.id_operacion_mantenimiento
LEFT JOIN mantenimiento_logistico.OperacionMantenimientoRepuesto omr ON opm.id_operacion_mantenimiento = omr.id_operacion_mantenimiento
```

#### 2. **Vista Materializada: Análisis de inventario de repuestos**

```sql
CREATE MATERIALIZED VIEW mantenimiento_logistico.vm_analisis_inventario_repuestos AS
SELECT
    -- Información del repuesto
    r.id_repuesto,
    r.codigo,
    r.nombre,
    r.stock,
    r.stock_minimo,

    -- Margen de seguridad
    (r.stock - r.stock_minimo) AS margen_seguridad,

    -- Estado del stock
    CASE
        WHEN r.stock = 0 THEN 'Agotado'
        WHEN r.stock <= r.stock_minimo THEN 'Crítico'
        WHEN r.stock <= r.stock_minimo * 1.5 THEN 'Bajo'
        ELSE 'Satisfactorio'
    END AS estado_stock,

    -- Requiere reabastecimiento
    CASE WHEN r.stock <= r.stock_minimo THEN TRUE ELSE FALSE END AS requiere_reabastecimiento,

    -- Estadísticas de uso
    COALESCE(SUM(omr.cantidad), 0) AS total_utilizado,
    COALESCE(COUNT(DISTINCT omr.id_operacion_mantenimiento), 0) AS veces_utilizado,

    -- Uso últimos 30 días
    COALESCE(SUM(CASE WHEN omr.fecha_uso >= CURRENT_DATE - INTERVAL '30 days' THEN omr.cantidad ELSE 0 END), 0) AS utilizado_ultimos_30_dias,

    -- Costos
    COALESCE(AVG(omr.precio_unitario), 0)::DECIMAL(10,2) AS precio_promedio,
    COALESCE(SUM(omr.cantidad * omr.precio_unitario), 0)::DECIMAL(12,2) AS costo_total_utilizado,

    -- Valor en inventario
    (r.stock * COALESCE(AVG(omr.precio_unitario), 0))::DECIMAL(10,2) AS valor_inventario,

    -- Fechas
    MAX(omr.fecha_uso) AS fecha_ultimo_uso,
    MIN(omr.fecha_uso) AS fecha_primer_uso,

    -- Tasa de uso mensual
    CASE
        WHEN COUNT(DISTINCT DATE_TRUNC('month', omr.fecha_uso)) > 0 THEN
            COALESCE(SUM(omr.cantidad), 0) / COUNT(DISTINCT DATE_TRUNC('month', omr.fecha_uso))
        ELSE 0
    END::DECIMAL(10,2) AS uso_mensual_promedio,

    -- Proyección de meses restantes
    CASE
        WHEN (COALESCE(SUM(omr.cantidad), 0) / NULLIF(COUNT(DISTINCT DATE_TRUNC('month', omr.fecha_uso)), 0)) > 0 THEN
            r.stock / (COALESCE(SUM(omr.cantidad), 0) / NULLIF(COUNT(DISTINCT DATE_TRUNC('month', omr.fecha_uso)), 0))
        ELSE NULL
    END::DECIMAL(10,1) AS meses_restantes_stock,

    -- Alerta
    CASE
        WHEN r.stock = 0 THEN 'Stock Agotado - Pedido Urgente'
        WHEN r.stock <= r.stock_minimo THEN 'Stock Crítico - Reabastecer'
        WHEN r.stock <= r.stock_minimo * 2 THEN 'Stock Bajo - Monitorear'
        ELSE 'Stock Adecuado'
    END AS alerta_inventario,

    -- Cantidad sugerida de reorden
    CASE
        WHEN r.stock <= r.stock_minimo THEN GREATEST(r.stock_minimo * 3 - r.stock, 10)
        ELSE 0
    END AS cantidad_sugerida_reorden,

    -- Fecha de actualización
    CURRENT_TIMESTAMP AS fecha_actualizacion

FROM mantenimiento_logistico.Repuesto r
LEFT JOIN mantenimiento_logistico.OperacionMantenimientoRepuesto omr ON r.id_repuesto = omr.id_repuesto

GROUP BY r.id_repuesto, r.codigo, r.nombre, r.stock, r.stock_minimo
ORDER BY
    CASE
        WHEN r.stock = 0 THEN 1
        WHEN r.stock <= r.stock_minimo THEN 2
        WHEN r.stock <= r.stock_minimo * 1.5 THEN 3
        ELSE 4
    END,
    r.codigo

WITH DATA;

-- Crear índice
CREATE INDEX idx_vm_inventario_estado ON mantenimiento_logistico.vm_analisis_inventario_repuestos(estado_stock);
CREATE INDEX idx_vm_inventario_codigo ON mantenimiento_logistico.vm_analisis_inventario_repuestos(codigo);
```
