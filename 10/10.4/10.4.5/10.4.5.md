# 10.4.5. Módulo 5 / Integrante 5

## Módulo de Monitoreo de Entrega

- **Nota:** Utilizar los [scripts de creación de tablas para este módulo](/scripts/ddl/tablas/integrante5.sql) y el [poblamiento inicial de datos](/scripts/dml/datos-iniciales/integrante5.sql)

### Creación de vistas materializadas

1. **Pantalla: P03 - Dashboard principal de monitoreo**
    - Vista 1: vm_dashboard_operaciones
    - Tablas incluidas: Operacion, Contenedor, Sensor, Notificacion, Incidencia

    **Script de creación de vista materializada**

```sql
CREATE MATERIALIZED VIEW monitoreo.vm_dashboard_operaciones AS
SELECT 
    -- Métricas de operaciones
    COUNT(DISTINCT o.id_operacion) AS total_operaciones,
    COUNT(DISTINCT CASE WHEN eo.nombre = 'En curso' THEN o.id_operacion END) AS operaciones_en_curso,
    COUNT(DISTINCT CASE WHEN eo.nombre = 'Completada' THEN o.id_operacion END) AS operaciones_completadas,
    COUNT(DISTINCT CASE WHEN eo.nombre = 'Pendiente' THEN o.id_operacion END) AS operaciones_pendientes,
    COUNT(DISTINCT CASE WHEN eo.nombre = 'Cancelada' THEN o.id_operacion END) AS operaciones_canceladas,
    
    -- Métricas de contenedores
    COUNT(DISTINCT c.id_contenedor) AS total_contenedores,
    COUNT(DISTINCT CASE WHEN ec.nombre = 'En tránsito' THEN c.id_contenedor END) AS contenedores_en_transito,
    COUNT(DISTINCT CASE WHEN ec.nombre = 'Disponible' THEN c.id_contenedor END) AS contenedores_disponibles,
    COUNT(DISTINCT CASE WHEN ec.nombre = 'Entregado' THEN c.id_contenedor END) AS contenedores_entregados,
    COUNT(DISTINCT CASE WHEN ec.nombre = 'En mantenimiento' THEN c.id_contenedor END) AS contenedores_en_mantenimiento,
    
    -- Métricas de sensores
    COUNT(DISTINCT s.id_sensor) AS total_sensores_activos,
    COUNT(DISTINCT CASE WHEN ts.nombre = 'Temperatura' THEN s.id_sensor END) AS sensores_temperatura,
    COUNT(DISTINCT CASE WHEN ts.nombre = 'Humedad' THEN s.id_sensor END) AS sensores_humedad,
    COUNT(DISTINCT CASE WHEN ts.nombre = 'GPS' THEN s.id_sensor END) AS sensores_gps,
    
    -- Métricas de notificaciones (últimas 24 horas)
    COUNT(DISTINCT CASE 
        WHEN tn.nombre = 'Crítica' AND n.fecha_hora >= CURRENT_TIMESTAMP - INTERVAL '24 hours' 
        THEN n.id_notificacion 
    END) AS notificaciones_criticas_24h,
    COUNT(DISTINCT CASE 
        WHEN tn.nombre = 'Advertencia' AND n.fecha_hora >= CURRENT_TIMESTAMP - INTERVAL '24 hours' 
        THEN n.id_notificacion 
    END) AS notificaciones_advertencia_24h,
    COUNT(DISTINCT CASE 
        WHEN tn.nombre = 'Informativa' AND n.fecha_hora >= CURRENT_TIMESTAMP - INTERVAL '24 hours' 
        THEN n.id_notificacion 
    END) AS notificaciones_informativas_24h,
    
    -- Métricas de incidencias
    COUNT(DISTINCT i.id_incidencia) AS total_incidencias,
    COUNT(DISTINCT CASE WHEN ei.nombre = 'Abierta' THEN i.id_incidencia END) AS incidencias_abiertas,
    COUNT(DISTINCT CASE WHEN ei.nombre = 'En proceso' THEN i.id_incidencia END) AS incidencias_en_proceso,
    COUNT(DISTINCT CASE WHEN ei.nombre = 'Resuelta' THEN i.id_incidencia END) AS incidencias_resueltas,
    
    -- Métricas de entregas
    COUNT(DISTINCT ent.id_entrega) AS total_entregas,
    COUNT(DISTINCT CASE WHEN ee.nombre = 'Completada' THEN ent.id_entrega END) AS entregas_completadas,
    COUNT(DISTINCT CASE WHEN ee.nombre = 'Pendiente' THEN ent.id_entrega END) AS entregas_pendientes,
    
    -- Métricas de vehículos y buques
    COUNT(DISTINCT v.id_vehiculo) AS total_vehiculos,
    COUNT(DISTINCT CASE WHEN ev.nombre = 'Disponible' THEN v.id_vehiculo END) AS vehiculos_disponibles,
    COUNT(DISTINCT b.id_buque) AS total_buques,
    COUNT(DISTINCT CASE WHEN eb.nombre = 'Disponible' THEN b.id_buque END) AS buques_disponibles,
    
    -- Tiempos promedio
    AVG(EXTRACT(EPOCH FROM (COALESCE(o.fecha_fin, CURRENT_TIMESTAMP) - o.fecha_inicio)) / 3600)::DECIMAL(10,2) AS duracion_promedio_operacion_horas,
    
    -- Fecha de última actualización
    CURRENT_TIMESTAMP AS fecha_actualizacion
    
FROM monitoreo.Operacion o
JOIN monitoreo.EstadoOperacion eo ON o.id_estado_operacion = eo.id_estado_operacion

-- Contenedores
CROSS JOIN monitoreo.Contenedor c
JOIN monitoreo.EstadoContenedor ec ON c.id_estado_contenedor = ec.id_estado_contenedor

-- Sensores
LEFT JOIN monitoreo.Sensor s ON s.id_contenedor = c.id_contenedor
LEFT JOIN monitoreo.TipoSensor ts ON s.id_tipo_sensor = ts.id_tipo_sensor

-- Notificaciones
LEFT JOIN monitoreo.Notificacion n ON n.id_sensor = s.id_sensor
LEFT JOIN monitoreo.TipoNotificacion tn ON n.id_tipo_notificacion = tn.id_tipo_notificacion

-- Incidencias
LEFT JOIN monitoreo.Incidencia i ON i.id_operacion = o.id_operacion
LEFT JOIN monitoreo.EstadoIncidencia ei ON i.id_estado_incidencia = ei.id_estado_incidencia

-- Entregas
LEFT JOIN monitoreo.Entrega ent ON ent.id_contenedor = c.id_contenedor
LEFT JOIN monitoreo.EstadoEntrega ee ON ent.id_estado_entrega = ee.id_estado_entrega

-- Vehículos
CROSS JOIN monitoreo.Vehiculo v
JOIN monitoreo.EstadoVehiculo ev ON v.id_estado_vehiculo = ev.id_estado_vehiculo

-- Buques
CROSS JOIN monitoreo.Buque b
JOIN monitoreo.EstadoEmbarcacion eb ON b.id_estado_embarcacion = eb.id_estado_embarcacion

WITH DATA;
```

2. **Pantalla: P05, P06 - Vista detallada de contenedor**
    - Vista 2: vm_contenedores_detalle
    - Tablas incluidas: Contenedor, Sensor, Notificacion, Documentacion

    **Script de creación de vista materializada**

```sql
CREATE MATERIALIZED VIEW monitoreo.vm_contenedores_detalle AS
SELECT 
    -- Información del contenedor
    c.id_contenedor,
    c.codigo,
    c.peso,
    c.capacidad,
    c.dimensiones,
    tc.nombre AS tipo_contenedor,
    tc.codigo AS codigo_tipo_contenedor,
    tc.costo AS costo_tipo,
    ec.nombre AS estado_contenedor,
    
    -- Porcentaje de carga
    ROUND((c.peso / NULLIF(c.capacidad, 0) * 100)::NUMERIC, 2) AS porcentaje_carga,
    
    -- Información de sensores
    COUNT(DISTINCT s.id_sensor) AS total_sensores,
    COUNT(DISTINCT CASE WHEN ts.nombre = 'Temperatura' THEN s.id_sensor END) AS sensores_temperatura,
    COUNT(DISTINCT CASE WHEN ts.nombre = 'Humedad' THEN s.id_sensor END) AS sensores_humedad,
    COUNT(DISTINCT CASE WHEN ts.nombre = 'GPS' THEN s.id_sensor END) AS sensores_gps,
    COUNT(DISTINCT CASE WHEN ts.nombre = 'Vibración' THEN s.id_sensor END) AS sensores_vibracion,
    COUNT(DISTINCT CASE WHEN ts.nombre = 'Apertura' THEN s.id_sensor END) AS sensores_apertura,
    
    -- Códigos de sensores (para referencia)
    STRING_AGG(DISTINCT s.codigo, ', ' ORDER BY s.codigo) AS codigos_sensores,
    
    -- Notificaciones asociadas
    COUNT(DISTINCT n.id_notificacion) AS total_notificaciones,
    COUNT(DISTINCT CASE WHEN tn.nombre = 'Crítica' THEN n.id_notificacion END) AS notificaciones_criticas,
    COUNT(DISTINCT CASE WHEN tn.nombre = 'Advertencia' THEN n.id_notificacion END) AS notificaciones_advertencia,
    
    -- Última notificación
    MAX(n.fecha_hora) AS fecha_ultima_notificacion,
    (SELECT tn2.nombre 
     FROM monitoreo.Notificacion n2
     JOIN monitoreo.TipoNotificacion tn2 ON n2.id_tipo_notificacion = tn2.id_tipo_notificacion
     JOIN monitoreo.Sensor s2 ON n2.id_sensor = s2.id_sensor
     WHERE s2.id_contenedor = c.id_contenedor
     ORDER BY n2.fecha_hora DESC
     LIMIT 1) AS tipo_ultima_notificacion,
    
    -- Documentación asociada
    COUNT(DISTINCT dc.id_documentacion) AS total_documentos,
    STRING_AGG(DISTINCT d.nombre, ', ') AS documentos_nombres,
    STRING_AGG(DISTINCT td.nombre, ', ') AS tipos_documentos,
    
    -- Entrega asociada (si existe)
    ent.codigo AS codigo_entrega,
    ent.fecha_entrega,
    ent.lugar_entrega,
    ee.nombre AS estado_entrega,
    imp.razon_social AS importador,
    imp.ruc AS ruc_importador,
    
    -- Estado operacional consolidado
    CASE 
        WHEN ec.nombre = 'No disponible' THEN 'No disponible'
        WHEN ec.nombre = 'En mantenimiento' THEN 'En mantenimiento'
        WHEN ec.nombre = 'Entregado' THEN 'Entregado'
        WHEN ec.nombre = 'En tránsito' THEN 'En tránsito'
        WHEN ec.nombre = 'Disponible' THEN 'Disponible'
        ELSE 'Otro'
    END AS estado_operacional
    
FROM monitoreo.Contenedor c
JOIN monitoreo.TipoContenedor tc ON c.id_tipo_contenedor = tc.id_tipo_contenedor
JOIN monitoreo.EstadoContenedor ec ON c.id_estado_contenedor = ec.id_estado_contenedor

-- Sensores
LEFT JOIN monitoreo.Sensor s ON s.id_contenedor = c.id_contenedor
LEFT JOIN monitoreo.TipoSensor ts ON s.id_tipo_sensor = ts.id_tipo_sensor

-- Notificaciones
LEFT JOIN monitoreo.Notificacion n ON n.id_sensor = s.id_sensor
LEFT JOIN monitoreo.TipoNotificacion tn ON n.id_tipo_notificacion = tn.id_tipo_notificacion

-- Documentación
LEFT JOIN monitoreo.DocumentacionContenedor dc ON dc.id_contenedor = c.id_contenedor
LEFT JOIN monitoreo.Documentacion d ON d.id_documentacion = dc.id_documentacion
LEFT JOIN monitoreo.TipoDocumento td ON d.id_tipo_documento = td.id_tipo_documento

-- Entrega
LEFT JOIN monitoreo.Entrega ent ON ent.id_contenedor = c.id_contenedor
LEFT JOIN monitoreo.EstadoEntrega ee ON ent.id_estado_entrega = ee.id_estado_entrega
LEFT JOIN monitoreo.Importador imp ON ent.id_importador = imp.id_importador

GROUP BY 
    c.id_contenedor,
    c.codigo,
    c.peso,
    c.capacidad,
    c.dimensiones,
    tc.nombre,
    tc.codigo,
    tc.costo,
    ec.nombre,
    ent.codigo,
    ent.fecha_entrega,
    ent.lugar_entrega,
    ee.nombre,
    imp.razon_social,
    imp.ruc

WITH DATA;
```

3. **Pantalla: P09 - Generación de reportes y auditoría**
    - Vista 3: vm_reportes_operaciones
    - Tablas incluidas: Operacion, Incidencia, Reporte, Notificacion

    **Script de creación de vista materializada**

```sql
CREATE MATERIALIZED VIEW monitoreo.vm_reportes_operaciones AS
SELECT 
    -- Información de la operación
    o.id_operacion,
    o.codigo AS codigo_operacion,
    o.fecha_inicio,
    o.fecha_fin,
    eo.nombre AS estado_operacion,
    
    -- Duración
    CASE 
        WHEN o.fecha_fin IS NOT NULL THEN 
            EXTRACT(EPOCH FROM (o.fecha_fin - o.fecha_inicio)) / 3600
        ELSE 
            EXTRACT(EPOCH FROM (CURRENT_TIMESTAMP - o.fecha_inicio)) / 3600
    END AS duracion_horas,
    
    -- Estado de la operación
    CASE 
        WHEN o.fecha_fin IS NULL THEN 'Activa'
        WHEN o.fecha_fin IS NOT NULL THEN 'Finalizada'
    END AS estado_temporal,
    
    -- Incidencias asociadas
    COUNT(DISTINCT i.id_incidencia) AS total_incidencias,
    COUNT(DISTINCT CASE WHEN ei.nombre = 'Abierta' THEN i.id_incidencia END) AS incidencias_abiertas,
    COUNT(DISTINCT CASE WHEN ei.nombre = 'En proceso' THEN i.id_incidencia END) AS incidencias_en_proceso,
    COUNT(DISTINCT CASE WHEN ei.nombre = 'Resuelta' THEN i.id_incidencia END) AS incidencias_resueltas,
    COUNT(DISTINCT CASE WHEN ei.nombre = 'Cerrada' THEN i.id_incidencia END) AS incidencias_cerradas,
    
    -- Tipos de incidencias
    STRING_AGG(DISTINCT ti.nombre, ', ') AS tipos_incidencias,
    
    -- Severidad promedio
    AVG(i.grado_severidad)::DECIMAL(3,1) AS severidad_promedio,
    MAX(i.grado_severidad) AS severidad_maxima,
    
    -- Códigos de incidencias
    STRING_AGG(DISTINCT i.codigo, ', ' ORDER BY i.codigo) AS codigos_incidencias,
    
    -- Reportes asociados
    COUNT(DISTINCT r.id_reporte) AS total_reportes,
    STRING_AGG(DISTINCT r.codigo, ', ' ORDER BY r.codigo) AS codigos_reportes,
    MAX(r.fecha_reporte) AS fecha_ultimo_reporte,
    
    -- Notificaciones generadas (a través de reportes)
    COUNT(DISTINCT n.id_notificacion) AS total_notificaciones_reportadas,
    COUNT(DISTINCT CASE WHEN tn.nombre = 'Crítica' THEN n.id_notificacion END) AS notificaciones_criticas_reportadas,
    
    -- Usuarios que reportaron incidencias
    STRING_AGG(DISTINCT u.correo_electronico, ', ') AS usuarios_reportaron,
    COUNT(DISTINCT u.id_usuario) AS total_usuarios_involucrados,
    
    -- Métricas de calidad
    CASE 
        WHEN COUNT(DISTINCT i.id_incidencia) = 0 THEN 'Excelente'
        WHEN COUNT(DISTINCT i.id_incidencia) <= 2 AND AVG(i.grado_severidad) < 5 THEN 'Bueno'
        WHEN COUNT(DISTINCT i.id_incidencia) <= 5 AND AVG(i.grado_severidad) < 7 THEN 'Regular'
        ELSE 'Deficiente'
    END AS calificacion_operacion,
    
    -- Cumplimiento de plazo (7 días estándar)
    CASE 
        WHEN o.fecha_fin IS NOT NULL AND o.fecha_fin <= o.fecha_inicio + INTERVAL '7 days' THEN TRUE
        WHEN o.fecha_fin IS NULL AND CURRENT_TIMESTAMP <= o.fecha_inicio + INTERVAL '7 days' THEN TRUE
        ELSE FALSE
    END AS cumplimiento_plazo,
    
    -- Días transcurridos o de duración
    CASE 
        WHEN o.fecha_fin IS NOT NULL THEN 
            EXTRACT(DAY FROM (o.fecha_fin - o.fecha_inicio))
        ELSE 
            EXTRACT(DAY FROM (CURRENT_TIMESTAMP - o.fecha_inicio))
    END::INTEGER AS dias_duracion
    
FROM monitoreo.Operacion o
JOIN monitoreo.EstadoOperacion eo ON o.id_estado_operacion = eo.id_estado_operacion

-- Incidencias
LEFT JOIN monitoreo.Incidencia i ON i.id_operacion = o.id_operacion
LEFT JOIN monitoreo.EstadoIncidencia ei ON i.id_estado_incidencia = ei.id_estado_incidencia
LEFT JOIN monitoreo.TipoIncidencia ti ON i.id_tipo_incidencia = ti.id_tipo_incidencia
LEFT JOIN monitoreo.Usuario u ON i.id_usuario = u.id_usuario

-- Reportes (relacionados con incidencias)
LEFT JOIN monitoreo.IncidenciaReporte ir ON ir.id_incidencia = i.id_incidencia
LEFT JOIN monitoreo.Reporte r ON r.id_reporte = ir.id_reporte

-- Notificaciones (a través de reportes)
LEFT JOIN monitoreo.Notificacion n ON n.id_reporte = r.id_reporte
LEFT JOIN monitoreo.TipoNotificacion tn ON n.id_tipo_notificacion = tn.id_tipo_notificacion

GROUP BY 
    o.id_operacion, 
    o.codigo, 
    o.fecha_inicio, 
    o.fecha_fin, 
    eo.nombre

WITH DATA;
```
