## Módulo de Gestión de Operaciones Terrestres

### Creación de secuencias

### 1. **Secuencia para Operacion (OP-YYYY-NNNNNN)**
```sql
-- Secuencia para códigos de Operacion
CREATE SEQUENCE operaciones_terrestres.seq_operacion_codigo
    START WITH 1
    INCREMENT BY 1
    MINVALUE 1
    MAXVALUE 999999
    CYCLE
    CACHE 20;

-- Inicializar secuencia basada en datos existentes
DO $$
DECLARE
    v_max_numero INTEGER;
BEGIN
    SELECT COALESCE(
        MAX(CAST(SUBSTRING(codigo FROM '[0-9]+$') AS INTEGER)), 0
    ) INTO v_max_numero
    FROM operaciones_terrestres.Operacion
    WHERE codigo ~ '^OP-[0-9]{4}-[0-9]{6}$';

    PERFORM setval('operaciones_terrestres.seq_operacion_codigo', v_max_numero + 1);
    RAISE NOTICE 'Secuencia operacion inicializada en: %', v_max_numero + 1;
END $$;
```

## **Funciones de Generación de Códigos**

### 1. **Función para código de Operacion**
```sql
CREATE OR REPLACE FUNCTION operaciones_terrestres.generar_codigo_operacion()
RETURNS VARCHAR(20)
LANGUAGE plpgsql
AS $$
DECLARE
    v_codigo VARCHAR(20);
    v_numero INTEGER;
    v_anio VARCHAR(4);
BEGIN
    v_anio := EXTRACT(YEAR FROM CURRENT_DATE)::VARCHAR;
    v_numero := nextval('operaciones_terrestres.seq_operacion_codigo');
    v_codigo := 'OP-' || v_anio || '-' || LPAD(v_numero::TEXT, 6, '0');
    RETURN v_codigo;
END;
$$;
```

## **Ejemplos de Uso**

### 1. **Insertar nueva Operacion**
```sql
INSERT INTO operaciones_terrestres.Operacion (id_operacion, codigo, fecha_inicio, id_estado_operacion)
VALUES (
    gen_random_uuid()::TEXT,
    operaciones_terrestres.generar_codigo_operacion(),  -- Genera: OP-2025-000001
    CURRENT_TIMESTAMP,
    (SELECT id_estado_operacion FROM operaciones_terrestres.EstadoOperacion WHERE nombre = 'Pendiente')
);
```

### 2. **Insertar nueva OperacionTerrestre**
```sql
-- Primero crear la operación base
INSERT INTO operaciones_terrestres.Operacion (id_operacion, codigo, fecha_inicio, id_estado_operacion)
VALUES (
    gen_random_uuid()::TEXT,
    operaciones_terrestres.generar_codigo_operacion(),
    CURRENT_TIMESTAMP,
    (SELECT id_estado_operacion FROM operaciones_terrestres.EstadoOperacion WHERE nombre = 'En curso')
)
RETURNING id_operacion;

-- Luego crear la operación terrestre asociada
INSERT INTO operaciones_terrestres.OperacionTerrestre (
    id_operacion_terrestre, id_operacion, codigo, id_vehiculo, id_ruta_terrestre, id_conductor, costo_operacion_terrestre
)
VALUES (
    gen_random_uuid()::TEXT,
    (SELECT id_operacion FROM operaciones_terrestres.Operacion ORDER BY fecha_inicio DESC LIMIT 1),
    operaciones_terrestres.generar_codigo_operacion_terrestre(),  -- Genera: OT-2025-000001
    (SELECT id_vehiculo FROM operaciones_terrestres.Vehiculo
     WHERE id_estado_vehiculo = (SELECT id_estado_vehiculo FROM operaciones_terrestres.EstadoVehiculo WHERE nombre = 'Disponible')
     LIMIT 1),
    (SELECT id_ruta_terrestre FROM operaciones_terrestres.RutaTerrestre ORDER BY random() LIMIT 1),
    (SELECT id_conductor FROM operaciones_terrestres.Conductor WHERE disponibilidad = TRUE LIMIT 1),
    1500.00
);
```

### 3. **Insertar nueva Ruta**
```sql
INSERT INTO operaciones_terrestres.Ruta (id_ruta, codigo, origen, destino, duracion, tarifa)
VALUES (
    gen_random_uuid()::TEXT,
    operaciones_terrestres.generar_codigo_ruta(),  -- Genera: RUT-0001
    'Callao, Perú',
    'Arequipa, Perú',
    18,
    2500.00
);

-- Crear ruta terrestre asociada
INSERT INTO operaciones_terrestres.RutaTerrestre (id_ruta_terrestre, id_ruta)
VALUES (
    gen_random_uuid()::TEXT,
    (SELECT id_ruta FROM operaciones_terrestres.Ruta ORDER BY id_ruta DESC LIMIT 1)
);
```

### 4. **Insertar nuevo Empleado y Conductor**
```sql
-- Primero crear un contrato
INSERT INTO operaciones_terrestres.Contrato (id_contrato, fecha_emision, fecha_vencimiento, id_estado_contrato)
VALUES (
    gen_random_uuid()::TEXT,
    CURRENT_DATE,
    CURRENT_DATE + INTERVAL '2 years',
    (SELECT id_estado_contrato FROM operaciones_terrestres.EstadoContrato WHERE nombre = 'Vigente')
)
RETURNING id_contrato;

-- Luego insertar el empleado
INSERT INTO operaciones_terrestres.Empleado (id_empleado, codigo, dni, nombre, apellido, direccion, id_especialidad_empleado, id_contrato)
VALUES (
    gen_random_uuid()::TEXT,
    operaciones_terrestres.generar_codigo_empleado(),  -- Genera: EMP-0001
    '78901234',
    'Carlos',
    'Rodríguez López',
    'Av. Principal 789, Lima',
    (SELECT id_especialidad_empleado FROM operaciones_terrestres.EspecialidadEmpleado WHERE nombre = 'Conductor'),
    (SELECT id_contrato FROM operaciones_terrestres.Contrato
     WHERE id_contrato NOT IN (SELECT id_contrato FROM operaciones_terrestres.Empleado)
     ORDER BY fecha_emision DESC LIMIT 1)
)
RETURNING id_empleado;

-- Finalmente crear el conductor
INSERT INTO operaciones_terrestres.Conductor (id_conductor, id_empleado, licencia, categoria, disponibilidad)
VALUES (
    gen_random_uuid()::TEXT,
    (SELECT id_empleado FROM operaciones_terrestres.Empleado ORDER BY id_empleado DESC LIMIT 1),
    'LIC-789012',
    'A-IIIb',
    TRUE
);
```

### 5. **Insertar nueva Incidencia**
```sql
INSERT INTO operaciones_terrestres.Incidencia (
    id_incidencia, codigo, id_tipo_incidencia, descripcion, grado_severidad,
    fecha_hora, id_estado_incidencia, id_operacion, id_usuario
)
VALUES (
    gen_random_uuid()::TEXT,
    operaciones_terrestres.generar_codigo_incidencia(),  -- Genera: INC-2025-000001
    (SELECT id_tipo_incidencia FROM operaciones_terrestres.TipoIncidencia WHERE nombre = 'Operacional'),
    'Retraso en la ruta debido a condiciones climáticas adversas',
    3,
    CURRENT_TIMESTAMP,
    (SELECT id_estado_incidencia FROM operaciones_terrestres.EstadoIncidencia WHERE nombre = 'Reportada'),
    (SELECT id_operacion FROM operaciones_terrestres.Operacion
     WHERE id_estado_operacion = (SELECT id_estado_operacion FROM operaciones_terrestres.EstadoOperacion WHERE nombre = 'En curso')
     ORDER BY random() LIMIT 1),
    (SELECT id_usuario FROM operaciones_terrestres.Usuario ORDER BY random() LIMIT 1)
);
```
