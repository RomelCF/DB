# 10.3.1. Módulo 1 / Integrante 1

## Módulo de Gestión de Reservas

### Creación de secuencias

### 1. **Secuencia para Empleado (EMP-NNN)**
```sql
-- Secuencia para códigos de Empleado
CREATE SEQUENCE gestion_reserva.seq_empleado_codigo
    START WITH 1
    INCREMENT BY 1
    MINVALUE 1
    MAXVALUE 999
    CYCLE
    CACHE 10;

-- Inicializar secuencia basada en datos existentes
DO $$
DECLARE
    v_max_numero INTEGER;
BEGIN
    SELECT COALESCE(
        MAX(CAST(SUBSTRING(codigo FROM '[0-9]+$') AS INTEGER)), 0
    ) INTO v_max_numero
    FROM gestion_reserva.Empleado 
    WHERE codigo ~ '^EMP-[0-9]{3}$';
    
    PERFORM setval('gestion_reserva.seq_empleado_codigo', v_max_numero + 1);
    RAISE NOTICE 'Secuencia empleado inicializada en: %', v_max_numero + 1;
END $$;
```

## **Funciones de Generación de Códigos**

### 1. **Función para código de Empleado**
```sql
CREATE OR REPLACE FUNCTION gestion_reserva.generar_codigo_empleado()
RETURNS VARCHAR(20)
LANGUAGE plpgsql
AS $$
DECLARE
    v_codigo VARCHAR(20);
    v_numero INTEGER;
BEGIN
    v_numero := nextval('gestion_reserva.seq_empleado_codigo');
    v_codigo := 'EMP-' || LPAD(v_numero::TEXT, 3, '0');
    RETURN v_codigo;
END;
$$;
```

AS $$
DECLARE
    v_codigo VARCHAR(20);
    v_numero INTEGER;
    v_anio VARCHAR(4);
BEGIN
    v_anio := EXTRACT(YEAR FROM CURRENT_DATE)::VARCHAR;
    v_numero := nextval('gestion_reserva.seq_operacion_codigo');
    v_codigo := 'OP-' || v_anio || '-' || LPAD(v_numero::TEXT, 3, '0');
    RETURN v_codigo;
END;
$$;

## **Ejemplos de Uso**

### 1. **Insertar nuevo Empleado**
```sql
-- Primero crear un nuevo contrato para el empleado
INSERT INTO gestion_reserva.Contrato (id_contrato, fecha_emision, fecha_vencimiento, id_estado_contrato)
VALUES (
    gen_random_uuid(),
    CURRENT_DATE,
    CURRENT_DATE + INTERVAL '2 years',
    (SELECT id_estado_contrato FROM gestion_reserva.EstadoContrato WHERE nombre = 'Activo')
)
RETURNING id_contrato;

-- Luego insertar el empleado con el contrato creado
INSERT INTO gestion_reserva.Empleado (id_empleado, codigo, dni, nombre, apellido, direccion, id_especialidad, id_contrato)
VALUES (
    gen_random_uuid(),
    gestion_reserva.generar_codigo_empleado(),  -- Genera: EMP-007
    '78901234',
    'Pedro',
    'González Martínez',
    'Av. Los Olivos 456, Lima',
    (SELECT id_especialidad_empleado FROM gestion_reserva.EspecialidadEmpleado WHERE nombre = 'Agente de Reservas'),
    -- Usar el último contrato creado que no esté asignado
    (SELECT id_contrato FROM gestion_reserva.Contrato 
     WHERE id_contrato NOT IN (SELECT id_contrato FROM gestion_reserva.Empleado)
     ORDER BY fecha_emision DESC LIMIT 1)
);
```

### 2. **Insertar nuevo Contenedor**
```sql
INSERT INTO gestion_reserva.Contenedor (id_contenedor, codigo, peso, capacidad, dimensiones, id_estado_contenedor, id_tipo_contenedor)
VALUES (
    gen_random_uuid(),
    gestion_reserva.generar_codigo_contenedor(),  -- Genera: CONT-007
    3500.00,
    30000.00,
    '40x8x9.5 ft',
    (SELECT id_estado_contenedor FROM gestion_reserva.EstadoContenedor WHERE nombre = 'Disponible'),
    (SELECT id_tipo_contenedor FROM gestion_reserva.TipoContenedor WHERE codigo = 'T-002')
);
```

### 3. **Insertar nueva Ruta**
```sql
INSERT INTO gestion_reserva.Ruta (id_ruta, codigo, origen, destino, duracion, tarifa)
VALUES (
    gen_random_uuid(),

```

### 4. **Insertar nueva Operacion**
```sql

VALUES (
    gen_random_uuid(),

```

### 5. **Insertar nueva OperacionTerrestre**
```sql

    950.00

```
